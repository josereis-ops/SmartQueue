            <input type="checkbox" id="check-todos-users" onchange="selecionarTodos('chk-user', this.checked)"> <label for="check-todos-users" style="font-size:12px; color:#9CA3AF;">Selecionar Todos</label>

    <script>
      window.onload = function() {
        carregarDados();
      };

      function carregarDados() {
        document.getElementById('loader').style.display = 'block';
        document.getElementById('interface-principal').style.display = 'none';
        document.getElementById('mensagem').innerText = "";

        google.script.run.withSuccessHandler(function(res) {
          document.getElementById('loader').style.display = 'none';
          document.getElementById('interface-principal').style.display = 'block';

          var htmlUsers = "";
          res.users.forEach(function(u) {
            htmlUsers += '<label class="item-lista">' +
                         '<input type="checkbox" class="chk-user" value="' + u.email + '">' +
                         '<div><div class="nome-agente">' + u.nome + '</div>' +
                         '<div class="skills-atuais">Skills: ' + (u.skills || 'Nenhuma') + '</div></div></label>';
          });
          document.getElementById('lista-users').innerHTML = htmlUsers;

          var htmlSkills = "";
          res.skills.forEach(function(sk) {
            htmlSkills += '<label class="item-lista">' +
                          '<input type="checkbox" class="chk-skill" value="' + sk + '">' +
                          '<div class="nome-agente">' + sk + '</div></label>';
          });
          document.getElementById('lista-skills').innerHTML = htmlSkills;
        }).obterDadosGestorSkills();
      }

      function selecionarTodos(classe, estado) {
        var checkboxes = document.querySelectorAll('.' + classe);
        checkboxes.forEach(function(cb) { cb.checked = estado; });
      }

      function aplicarAlteracoes(acao) {
        var usersCheck = document.querySelectorAll('.chk-user:checked');
        var emails = Array.from(usersCheck).map(function(cb) { return cb.value; });

        var skillsCheck = document.querySelectorAll('.chk-skill:checked');
        var skills = Array.from(skillsCheck).map(function(cb) { return cb.value; });

        if(emails.length === 0 || skills.length === 0) {
           document.getElementById('mensagem').style.color = "var(--alerta-amarelo)";
           document.getElementById('mensagem').innerText = "Seleciona pelo menos 1 operador e 1 skill.";
           return;
        }

        document.getElementById('mensagem').style.color = "white";
        document.getElementById('mensagem').innerText = "A processar...";

        google.script.run.withSuccessHandler(function(res) {
           if(res.sucesso) {
              document.getElementById('mensagem').style.color = "var(--sucesso)";
              document.getElementById('mensagem').innerText = res.mensagem;
              setTimeout(carregarDados, 1500); // Recarrega para mostrar as novas skills na UI
           } else {
              document.getElementById('mensagem').style.color = "var(--alerta-vermelho)";
              document.getElementById('mensagem').innerText = res.mensagem;
           }
        }).atualizarSkillsEmMassa(emails, skills, acao);
      }
    </script>
