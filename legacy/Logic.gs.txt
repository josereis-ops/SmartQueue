/** Segmentos = diferença entre marcas na ordem dada; _apos_ultima_marca_ms = até ao momento da chamada. */
function sqDebugPerfBuildReport_(ctx, orderedLabels) {
  if (!ctx) return null;
  var cum = ctx.cum;
  var seg = {};
  var prev = 0;
  for (var i = 0; i < orderedLabels.length; i++) {
    var lab = orderedLabels[i];
    var t = cum[lab];
    if (typeof t !== 'number') continue;
    seg[lab + '_ms'] = t - prev;
    prev = t;
  }
  var wall = Date.now() - ctx.t0;
  seg._apos_ultima_marca_ms = wall - prev;
  seg._total_ms = wall;
  return { cumulativo_ms: cum, segmentos_ms: seg };
}

/** Formata segundos como "H:MM:SS" ou "MM:SS" (igual à lógica antiga de registarLogEstado). */
function _formatarDuracaoLogEstado_(diffSegundos) {
  var h = Math.floor(diffSegundos / 3600);
  var m = Math.floor((diffSegundos % 3600) / 60).toString().padStart(2, '0');
  var s = (diffSegundos % 60).toString().padStart(2, '0');
  return (h > 0 ? h + ':' : '') + m + ':' + s;
}

/** Fallback: últimas linhas da folha, de baixo para cima (sem ler a folha toda). lastRowHint: opcional (evita 2× getLastRow no mesmo registarLogEstado). */
function _registarLogEstadoFecharLinhaLegacy_(sheetLogEst, email, dataHora, lastRowHint) {
  var lastR = (typeof lastRowHint === 'number' && lastRowHint >= 1) ? lastRowHint : sheetLogEst.getLastRow();
  if (lastR < 2) return;
  var startR = Math.max(2, lastR - LOG_ESTADO_LEGACY_SCAN_MAX + 1);
  var chunk = sheetLogEst.getRange(startR, 1, lastR, 4).getValues();
  var em = email.toString().toLowerCase().trim();
  for (var ii = chunk.length - 1; ii >= 0; ii--) {
    if (!chunk[ii][0] || chunk[ii][0].toString().toLowerCase().trim() !== em) continue;
    var dataInicio = new Date(chunk[ii][2]);
    if (isNaN(dataInicio.getTime())) break;
    var diffSegundos = Math.floor((dataHora.getTime() - dataInicio.getTime()) / 1000);
    if (diffSegundos >= 0) {
      var realRow = startR + ii;
      sheetLogEst.getRange(realRow, 4).setValue(_formatarDuracaoLogEstado_(diffSegundos));
    }
    break;
  }
}

// --- NOVA FUNÇÃO: PARSER UNIVERSAL DE DATAS ---
function parseDataSegura(valor) {
  if (!valor) return null;
  if (valor instanceof Date) {
    return isNaN(valor.getTime()) ? null : valor;
  }
  if (typeof valor === 'number') {
    var d = new Date(valor);
    return isNaN(d.getTime()) ? null : d;
  }
  if (typeof valor === 'string') {
    var vStr = valor.trim().replace('T', ' '); // Normaliza o 'T' do HTML5 para espaço
    // Quando o valor chega como texto numérico (epoch em ms), converter explicitamente.
    // new Date("1713859200000") pode ser inválido em alguns contextos.
    if (/^\d{10,16}$/.test(vStr)) {
      var nTs = Number(vStr);
      var dNumStr = new Date(nTs);
      return isNaN(dNumStr.getTime()) ? null : dNumStr;
    }
    vStr = vStr.replace(/-/g, '/'); // Normaliza hífens para barras para o construtor Date
    var regexEuro = /^(\d{1,2})\/(\d{1,2})\/(\d{4})(?: (\d{1,2}):(\d{1,2})(?::(\d{1,2}))?)?$/;
    var match = vStr.match(regexEuro);
    if (match) {
      var dia = parseInt(match[1], 10);
      var mes = parseInt(match[2], 10) - 1;
      var ano = parseInt(match[3], 10);
      var hora = match[4] ? parseInt(match[4], 10) : 0;
      var min = match[5] ? parseInt(match[5], 10) : 0;
      var sec = match[6] ? parseInt(match[6], 10) : 0;
      var dEuro = new Date(ano, mes, dia, hora, min, sec);
      if (!isNaN(dEuro.getTime())) return dEuro;
    }
    var dStr = new Date(vStr);
    return isNaN(dStr.getTime()) ? null : dStr;
  }
  return null;
}

// --- FUNÇÃO AUXILIAR: CALCULAR DIAS ÚTEIS ---
function calcularDiasUteis(dataInicial, dataFinal) {
  var count = 0;
  var cur = new Date(dataInicial.getTime());
  cur.setHours(0,0,0,0);
  var fim = new Date(dataFinal.getTime());
  fim.setHours(0,0,0,0);

  while (cur < fim) {
    var day = cur.getDay();
    if (day !== 0 && day !== 6) { // 0 = Domingo, 6 = Sábado
      count++;
    }
    cur.setDate(cur.getDate() + 1);
  }
  return count;
}

/**
 * @param {string} novoEstado
 * @param {{ t0: number, cum: Object }|null} [perfCtxExterno] Se definido (ex.: arranque com perf), marcas vão para este contexto; `ret.debugOfflineMs` leva o relatório (serializável).
 * @param {string=} [motivoSuspensaoOpt] Motivo custom para suspensão automática de casos em tratamento.
 */
function atualizarEstadoOperador(novoEstado, perfCtxExterno, motivoSuspensaoOpt) {
  var startTsHealth = new Date().getTime();
  var perfOff = perfCtxExterno || (SQ_DEBUG_PERF_PAYLOAD ? sqDebugPerfStart_() : null);
  try {
    var email = Session.getActiveUser().getEmail();
    if (!email || email === "") {
      return { sucesso: false, codigoErro: "SQ_SESSAO_EXPIRADA", mensagem: "Sessão expirada. Por favor, faz F5 na página." };
    }
    email = email.toLowerCase().trim();

    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName('Utilizadores');
    var data = sheet.getDataRange().getValues();
    var agora = new Date();

    for (var i = 1; i < data.length; i++) {
      if (data[i][0] && data[i][0].toString().toLowerCase().trim() === email) {
        var estadoAtual = data[i][6] ? data[i][6].toString().trim() : "Offline";
        // Short-circuit: evita custo pesado quando o estado já é o mesmo (ex.: arranque já Offline).
        if (estadoAtual === novoEstado) {
          var retIgual = { sucesso: true, semAlteracao: true };
          if (perfOff) {
            if (!perfOff.cum.util) sqDebugPerfMark_(perfOff, 'util');
            if (!perfOff.cum.log) sqDebugPerfMark_(perfOff, 'log');
            if (!perfOff.cum.susp) sqDebugPerfMark_(perfOff, 'susp');
            var repIgual = sqDebugPerfBuildReport_(perfOff, ['util', 'log', 'susp']);
            retIgual.debugOfflineMs = repIgual;
            if (perfCtxExterno) perfCtxExterno._sqOfflinePerf = repIgual;
          }
          return retIgual;
        }

        // 1. Utilizadores: G–K num único setValues (preserva colunas I–J)
        sheet.getRange(i + 1, 7, 1, 5).setValues([[
          novoEstado,
          agora,
          data[i][8],
          data[i][9],
          agora.getTime()
        ]]);
        if (perfOff) sqDebugPerfMark_(perfOff, 'util');

        registarLogEstado(email, novoEstado, agora);
        if (perfOff) sqDebugPerfMark_(perfOff, 'log');

        // Evita stats pesadas stale imediatamente após troca de estado.
        try { _limparCacheStatsOperador_(email); } catch (_eStats) {}

        // Proteção server-side: ao sair de estados operacionais, suspende sempre casos em tratamento.
        // Evita ficar "em pausa/refeição/etc" com caso ainda atribuído.
        var manterCasoAtivo = (novoEstado === "Disponível" || novoEstado === "Trabalho manual");
        if (!manterCasoAtivo) {
          suspenderCasosEmTratamentoServidor(email, motivoSuspensaoOpt, data);
        }
        if (perfOff) sqDebugPerfMark_(perfOff, 'susp');

        var ret = { sucesso: true };
        if (perfOff) {
          var repOff = sqDebugPerfBuildReport_(perfOff, ['util', 'log', 'susp']);
          ret.debugOfflineMs = repOff;
          if (perfCtxExterno) {
            perfCtxExterno._sqOfflinePerf = repOff;
          }
        }
        return ret;
      }
    }
    return { sucesso: false, mensagem: "Aviso: O teu e-mail não foi encontrado na base de dados." };
  } catch (e) { 
    return { sucesso: false, mensagem: e.message }; 
  } finally {
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("atualizarEstadoOperador", new Date().getTime() - startTsHealth);
  }
}

// Fluxo otimizado para primeira queda: altera para "Disponível" e tenta auto-atribuir no mesmo pedido.
function atualizarEstadoEAutoAtribuirPrimeiraTarefa() {
  var t0Auto = Date.now();
  var dbgAuto = SQ_DEBUG_PERF_PAYLOAD ? { marks: {} } : null;
  var markAuto = function(nome) {
    if (!dbgAuto) return;
    dbgAuto.marks[nome] = Date.now() - t0Auto;
  };
  try {
    markAuto("start");
    var ss = getSpreadsheet();
    var email = Session.getActiveUser().getEmail();
    if (!email || email === "") {
      return { sucesso: false, mensagem: "Sessão expirada. Por favor, faz F5 na página." };
    }
    email = email.toLowerCase().trim();

    var sheetU = ss.getSheetByName('Utilizadores');
    var lrU = sheetU ? sheetU.getLastRow() : 0;
    var dataU = (sheetU && lrU > 0) ? sheetU.getRange(1, 1, lrU, 13).getValues() : [[]];

    var idxUser = -1;
    for (var i = 1; i < dataU.length; i++) {
      var mail = dataU[i][0] ? dataU[i][0].toString().toLowerCase().trim() : "";
      if (mail === email) {
        idxUser = i;
        break;
      }
    }
    if (idxUser === -1) {
      return { sucesso: false, codigoErro: "SQ_UTILIZADOR_NAO_ENCONTRADO", mensagem: "Aviso: O teu e-mail não foi encontrado na base de dados." };
    }

    var agora = new Date();
    var estadoAtual = dataU[idxUser][6] ? dataU[idxUser][6].toString().trim() : "Offline";
    if (estadoAtual !== "Disponível") {
      markAuto("before_update_utilizador");
      // Atualização mínima e imediata para priorizar auto-atribuição da primeira tarefa.
      sheetU.getRange(idxUser + 1, 7, 1, 5).setValues([[
        "Disponível",
        agora,
        dataU[idxUser][8],
        dataU[idxUser][9],
        agora.getTime()
      ]]);
      markAuto("after_setValues_utilizador");
      markAuto("before_registarLogEstado");
      // Hot-path de 1ª queda: quando saímos de Offline, evitar fecho de duração legado reduz latência.
      // Para outros estados mantém-se a lógica completa de durações.
      if (estadoAtual === "Offline") _registarLogEstadoAppendRapido_(email, "Disponível", agora);
      else registarLogEstado(email, "Disponível", agora);
      markAuto("after_registarLogEstado");
      // Hot path de 1ª queda: evita custo extra de invalidação imediata.
      // As stats já têm TTL curto e serão atualizadas no background.

      // Mantém matriz coerente para reutilizar já no pedirNovaTarefa (evita segunda leitura da folha).
      dataU[idxUser][6] = "Disponível";
      dataU[idxUser][7] = agora;
      dataU[idxUser][10] = agora.getTime();
      markAuto("after_update_utilizador");
    }
    markAuto("before_pedirNovaTarefa");
    var pedido = pedirNovaTarefa(ss, {
      dataU: dataU,
      debugPerf: SQ_DEBUG_PERF_PAYLOAD === true
    });
    markAuto("after_pedirNovaTarefa");

    if (pedido && pedido.sucesso) {
      var retOk = { sucesso: true, tarefa: pedido.tarefa, autoAtribuida: true };
      if (dbgAuto) {
        dbgAuto.total_ms = Date.now() - t0Auto;
        dbgAuto.pedirNovaTarefa = pedido.debugPedirNovaTarefa || null;
        retOk.debugAutoPrimeiraTarefa = dbgAuto;
      }
      return retOk;
    }

    var retSem = {
      sucesso: true,
      autoAtribuida: false,
      mensagem: (pedido && pedido.mensagem) ? pedido.mensagem : "Sem tarefas disponíveis para o teu perfil."
    };
    if (dbgAuto) {
      dbgAuto.total_ms = Date.now() - t0Auto;
      dbgAuto.pedirNovaTarefa = (pedido && pedido.debugPedirNovaTarefa) ? pedido.debugPedirNovaTarefa : null;
      retSem.debugAutoPrimeiraTarefa = dbgAuto;
    }
    return retSem;
  } catch (e) {
    return { sucesso: false, mensagem: e.message };
  }
}

// Endpoint leve para sincronizar apenas estado remoto (evita usar stats pesadas no timer de 15s).
function obterEstadoAtualOperadorRapido() {
  try {
    var email = Session.getActiveUser().getEmail();
    if (!email || email === "") return { sucesso: false, mensagem: "Sessão expirada." };
    email = email.toLowerCase().trim();

    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName('Utilizadores');
    if (!sheet) return { sucesso: false, mensagem: "Folha 'Utilizadores' não encontrada." };

    var lr = sheet.getLastRow();
    if (lr < 2) return { sucesso: false, mensagem: "Base de utilizadores vazia." };

    // Só lê colunas mínimas necessárias: A (email) e G (estado atual).
    var dados = sheet.getRange(2, 1, lr - 1, 7).getValues();
    for (var i = 0; i < dados.length; i++) {
      var mail = dados[i][0] ? dados[i][0].toString().toLowerCase().trim() : "";
      if (mail === email) {
        return { sucesso: true, estadoAtualServidor: (dados[i][6] || "").toString().trim() };
      }
    }
    return { sucesso: false, mensagem: "Utilizador não encontrado." };
  } catch (e) {
    return { sucesso: false, mensagem: e.message };
  }
}

// Leitura incremental do Log_Performance desde uma data mínima (reduz scans totais em folhas grandes).
function _lerLogPerformanceDesde_(sheetLog, dataMin) {
  if (!sheetLog) return [];
  var lr = sheetLog.getLastRow();
  if (lr < 2) return [];

  var chunk = 1000;
  var encontrou = false;
  var startRow = lr + 1;
  var stop = false;

  for (var end = lr; end >= 2 && !stop; end -= chunk) {
    var bs = Math.max(2, end - chunk + 1);
    var qtd = end - bs + 1;
    var datas = sheetLog.getRange(bs, 4, qtd, 1).getValues(); // Col D = Data Fim

    for (var i = qtd - 1; i >= 0; i--) {
      var dFim = parseDataSegura(datas[i][0]);
      if (dFim && dFim >= dataMin) {
        encontrou = true;
        startRow = bs + i;
      } else if (encontrou) {
        stop = true; // cruzou a fronteira temporal, já temos o ponto de corte
        break;
      }
    }
  }

  if (!encontrou || startRow > lr) return [];
  // A–H: ID, Operador, Início, Fim, Estado, Duração texto, Segundos, Skill (H).
  return sheetLog.getRange(startRow, 1, lr - startRow + 1, 8).getValues();
}

// Leitura incremental do Log_Estados desde uma data mínima (normalmente início do dia).
function _lerLogEstadosDesde_(sheetEst, dataMin) {
  if (!sheetEst) return [];
  var lr = sheetEst.getLastRow();
  if (lr < 2) return [];

  var chunk = 1000;
  var encontrou = false;
  var startRow = lr + 1;
  var stop = false;

  for (var end = lr; end >= 2 && !stop; end -= chunk) {
    var bs = Math.max(2, end - chunk + 1);
    var qtd = end - bs + 1;
    var datas = sheetEst.getRange(bs, 3, qtd, 1).getValues(); // Col C = Data/Hora registo

    for (var i = qtd - 1; i >= 0; i--) {
      var dReg = parseDataSegura(datas[i][0]);
      if (dReg && dReg >= dataMin) {
        encontrou = true;
        startRow = bs + i;
      } else if (encontrou) {
        stop = true;
        break;
      }
    }
  }

  if (!encontrou || startRow > lr) return [];
  return sheetEst.getRange(startRow, 1, lr - startRow + 1, 4).getValues();
}

/// --- SUSPENDER CASOS EM TRATAMENTO (MANTÉM O DONO, FECHA TMT): scan J:K; por segmento 1× getValues A:M + 1× setValues; Log batch; 1× flush ---
/** @param {*} [matrizUtilizadoresOpt] Se definida (ex. matriz já lida), evita nova leitura da folha Utilizadores. */
function suspenderCasosEmTratamentoServidor(emailOperador, motivoForce, matrizUtilizadoresOpt) {
  try {
    var ss = getSpreadsheet();
    var sheetDist = ss.getSheetByName(ABA_DISTRIBUICAO);
    if (!sheetDist) return;

    var lrD = sheetDist.getLastRow();
    if (lrD < 2) return;

    var emailLimpo = emailOperador.toString().toLowerCase().trim();
    // Scan leve apenas J:K (Estado + Responsável).
    var jk = sheetDist.getRange(1, 10, lrD, 2).getValues();
    var linhasOrd = [];
    for (var i = 1; i < jk.length; i++) {
      var estadoScan = jk[i][0] ? jk[i][0].toString().trim() : "";
      var responsavelScan = jk[i][1] ? jk[i][1].toString().toLowerCase().trim() : "";
      if (estadoScan === "Em Tratamento" && responsavelScan === emailLimpo) {
        linhasOrd.push(i + 1);
      }
    }
    if (linhasOrd.length === 0) return;

    var textoMotivo = motivoForce ? motivoForce : "Operador Offline";
    var novaObsLinha = "[Auto] Caso Suspenso (" + textoMotivo + ") - Retido para o operador.";

    var timezone = Session.getScriptTimeZone();
    var dataFim = new Date();
    var carimbo = Utilities.formatDate(dataFim, timezone, "dd/MM HH:mm");

    var mapaEmailLoja = {};
    var du = null;
    if (matrizUtilizadoresOpt != null && matrizUtilizadoresOpt.length > 0) {
      du = matrizUtilizadoresOpt;
    } else {
      var sheetU = ss.getSheetByName("Utilizadores");
      if (sheetU) du = sheetU.getDataRange().getValues();
    }
    if (du) {
      for (var ux = 1; ux < du.length; ux++) {
        var emk = du[ux][0] ? du[ux][0].toString().toLowerCase().trim() : "";
        if (emk) mapaEmailLoja[emk] = du[ux][2] ? du[ux][2].toString().trim() : "";
      }
    }

    var sheetLog = ss.getSheetByName(ABA_LOG);
    var logRows = [];

    linhasOrd.sort(function (a, b) { return a - b; });
    var houveEscritaDist = false;
    var segStart = linhasOrd[0];
    var segEnd = linhasOrd[0];
    for (var z = 1; z < linhasOrd.length; z++) {
      if (linhasOrd[z] === segEnd + 1) {
        segEnd = linhasOrd[z];
      } else {
        if (_suspenderProcessarSegmentoLinhas_(sheetDist, segStart, segEnd, emailLimpo, mapaEmailLoja, dataFim, carimbo, novaObsLinha, logRows)) {
          houveEscritaDist = true;
        }
        segStart = linhasOrd[z];
        segEnd = linhasOrd[z];
      }
    }
    if (_suspenderProcessarSegmentoLinhas_(sheetDist, segStart, segEnd, emailLimpo, mapaEmailLoja, dataFim, carimbo, novaObsLinha, logRows)) {
      houveEscritaDist = true;
    }

    if (logRows.length > 0 && sheetLog) {
      var lr = sheetLog.getLastRow();
      sheetLog.getRange(lr + 1, 1, logRows.length, 8).setValues(logRows);
    }

    if (houveEscritaDist) {
      SpreadsheetApp.flush();
    }
  } catch (e) {
    Logger.log("Erro ao suspender casos zombie: " + e.message);
  }
}

/**
 * Lê A:O uma vez em [r0..r1], revalida Em Tratamento + responsável, aplica Suspenso (mantém dono col. K), Log_Performance em logRows.
 * @returns {boolean} true se fez setValues na Distribuição.
 */
function _suspenderProcessarSegmentoLinhas_(sheetDist, r0, r1, emailLimpo, mapaEmailLoja, dataFim, carimbo, novaObsLinha, logRows) {
  var n = r1 - r0 + 1;
  var block = sheetDist.getRange(r0, 1, n, 15).getValues();
  var alterou = false;
  for (var jj = 0; jj < n; jj++) {
    var row = block[jj];
    var estado = row[9] ? row[9].toString().trim() : "";
    var responsavel = row[10] ? row[10].toString().toLowerCase().trim() : "";
    if (estado !== "Em Tratamento" || responsavel !== emailLimpo) continue;

    var emailDaLinha = row[10];
    var emailKey = emailDaLinha ? emailDaLinha.toString().toLowerCase().trim() : "";
    var dataInicio = row[11];
    var obsAntigas = row[8] ? row[8].toString().trim() : "";
    var userAbrev = (emailDaLinha ? emailDaLinha.toString() : "").split("@")[0];
    var novaEntrada = "[" + carimbo + " - " + userAbrev + "]\n" + novaObsLinha;
    var obsFinal = obsAntigas === "" ? novaEntrada : novaEntrada + "\n\n" + obsAntigas;
    var lojaOp = mapaEmailLoja[emailKey] || "";

    row[0] = lojaOp;
    row[8] = obsFinal;
    row[9] = "Suspenso";
    row[10] = emailDaLinha;
    row[11] = "";
    row[12] = dataFim;
    alterou = true;

    var diffS = 0;
    var duracaoFormatada = "00:00";
    if (dataInicio && dataFim) {
      var dInicio = new Date(dataInicio);
      if (!isNaN(dInicio.getTime())) {
        diffS = Math.floor((dataFim.getTime() - dInicio.getTime()) / 1000);
        if (diffS < 0) diffS = 0;
        var h = Math.floor(diffS / 3600);
        var mm = Math.floor((diffS % 3600) / 60).toString().padStart(2, "0");
        var ssf = (diffS % 60).toString().padStart(2, "0");
        duracaoFormatada = (h > 0 ? h + ":" : "") + mm + ":" + ssf;
      }
    }
    var idCaso = row[1];
    var idLog = _normIdCaso_(idCaso);
    var skillLinha = row[14] ? row[14].toString().trim() : "";
    logRows.push([idLog, emailDaLinha, dataInicio, dataFim, "Suspenso", duracaoFormatada, diffS, skillLinha]);
  }
  if (alterou) {
    sheetDist.getRange(r0, 1, n, 15).setValues(block);
  }
  return alterou;
}

// --- MS-23: Libertar suspensos fantasma em background (não corre em pedirNovaTarefa) ---

/** Texto abaixo do bloco auto «Retido para o operador» = tratamento (ex. justificação Pendente/Agendado). */
function _obsSubstantivasAntesRetido_(obsText) {
  var t = (obsText || "").toString().trim();
  if (t === "") return false;
  var sep = t.indexOf("\n\n");
  if (sep < 0) return false;
  return t.substring(sep + 2).trim() !== "";
}

/** Mapa idCaso → diffS da última linha Suspenso no Log_Performance. */
function _mapaTmtSuspensaoLog_(sheetLog) {
  var mapa = {};
  if (!sheetLog) return mapa;
  var lr = sheetLog.getLastRow();
  if (lr < 2) return mapa;
  var data = sheetLog.getRange(2, 1, lr - 1, 7).getValues();
  for (var i = 0; i < data.length; i++) {
    var acao = data[i][4] ? data[i][4].toString().trim() : "";
    if (acao !== "Suspenso") continue;
    var id = _normIdCaso_(data[i][0]);
    if (!id) continue;
    var diffS = parseInt(data[i][6], 10);
    if (isNaN(diffS) || diffS < 0) diffS = 0;
    mapa[id] = diffS;
  }
  return mapa;
}

/**
 * Job background (trigger 5 min): libertar suspensos sem tratamento efectivo.
 * Fantasma = sem obs antes do retido E TMT < FANTASMA_TMT_MAX_SEG no log de suspensão.
 */
function libertarSuspensosFantasmaBackground() {
  var startTsHealth = new Date().getTime();
  var lock = LockService.getScriptLock();
  var libertados = 0;
  var ids = [];
  try {
    lock.waitLock(30000);
    var ss = getSpreadsheet();
    var sheetDist = ss.getSheetByName(ABA_DISTRIBUICAO);
    if (!sheetDist) {
      return { sucesso: false, mensagem: "Aba Distribuição não encontrada.", libertados: 0 };
    }
    var lr = sheetDist.getLastRow();
    if (lr < 2) {
      return { sucesso: true, mensagem: "Sem linhas.", libertados: 0, ids: [] };
    }

    var mapaTmt = _mapaTmtSuspensaoLog_(ss.getSheetByName(ABA_LOG));
    var bloco = sheetDist.getRange(2, 1, lr - 1, 15).getValues();
    var alterou = false;
    var carimbo = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "dd/MM HH:mm");
    var notaLiberta = "[Auto] Libertado — suspensão fantasma (job background).";

    for (var r = 0; r < bloco.length; r++) {
      var row = bloco[r];
      var estado = row[9] ? row[9].toString().trim() : "";
      var responsavel = row[10] ? row[10].toString().toLowerCase().trim() : "";
      if (estado !== "Suspenso" || responsavel === "") continue;

      if (_obsSubstantivasAntesRetido_(row[8])) continue;

      var idCaso = _normIdCaso_(row[1]);
      var diffS = mapaTmt.hasOwnProperty(idCaso) ? mapaTmt[idCaso] : null;
      if (diffS === null || diffS >= FANTASMA_TMT_MAX_SEG) continue;

      var userAbrev = responsavel.split("@")[0];
      var novaEntrada = "[" + carimbo + " - " + userAbrev + "]\n" + notaLiberta;
      var obsAntigas = row[8] ? row[8].toString().trim() : "";
      row[0] = "";
      row[8] = obsAntigas === "" ? novaEntrada : novaEntrada + "\n\n" + obsAntigas;
      row[9] = "Livre";
      row[10] = "";
      row[11] = "";
      alterou = true;
      libertados++;
      ids.push(idCaso);
    }

    if (alterou) {
      sheetDist.getRange(2, 1, bloco.length, 15).setValues(bloco);
      SpreadsheetApp.flush();
    }

    return {
      sucesso: true,
      mensagem: libertados > 0
        ? libertados + " caso(s) fantasma libertado(s)."
        : "Nenhum suspenso fantasma encontrado.",
      libertados: libertados,
      ids: ids
    };
  } catch (e) {
    return { sucesso: false, mensagem: e.message, libertados: libertados, ids: ids };
  } finally {
    try { lock.releaseLock(); } catch (_e) {}
    if (typeof logAcaoHealthCheck === "function") {
      logAcaoHealthCheck("libertarSuspensosFantasmaBackground", new Date().getTime() - startTsHealth);
    }
  }
}

/** Cria trigger de 5 em 5 min para libertarSuspensosFantasmaBackground. */

  var alvo = "libertarSuspensosFantasmaBackground";
  var triggers = ScriptApp.getProjectTriggers();

    mensagem: jaExiste ? "Trigger fantasma já existia." : "Trigger fantasma criado (5 min).",
    handler: alvo,
    frequencia: "5 minutos"
  };
}

/** Normaliza uma linha da Distribuição para escrita batch A:P (16 colunas). */

/**
 * @param {GoogleAppsScript.Spreadsheet.Spreadsheet|null} ssAtivo
 * @param {Object=} opcoes Opcional: { dataU, dataDist, skipSpreadsheetFlush }
 *   dataU — matriz A:M (13 cols) de Utilizadores já lida nesta execução.
 *   dataDist — matriz da Distribuição (16 cols) alinhada a getLastRow(); atualizada em RAM pela finalização.
 *   skipSpreadsheetFlush — se true, não chama SpreadsheetApp.flush() (ex.: fluxo 2-em-1; o caller faz flush único).
 */
function pedirNovaTarefa(ssAtivo, opcoes) {
  var startTsHealth = new Date().getTime();
  opcoes = opcoes || {};
  var permitirRetryConflito = (opcoes.retryConflito !== false);
  var perfDetalheAtivo = (opcoes.debugPerf === true);
  var diagAtivo = (opcoes.debugDiag === true || SQ_DEBUG_DIAG_QUEDA === true);
  var perfDetalhe = perfDetalheAtivo ? { marks: {}, total_ms: 0 } : null;
  var diag = diagAtivo ? {
    tsIso: new Date(startTsHealth).toISOString(),
    emailOperador: "",
    estadoOperadorAtual: "",
    lojaOperador: "",
    skillsOperadorCount: 0,
    contadores: {
      linhasTotais: 0,
      semId: 0,
      fechado: 0,
      filtroLoja: 0,
      skillVazia: 0,
      filtroSkill: 0,
      semTier: 0,
      elegiveis: 0
    },
    candidato: null,
    revalidacao: null,
    motivoBloqueio: ""
  } : null;
  var _perfMark = function(nome) {
    if (!perfDetalhe) return;
    perfDetalhe.marks[nome] = Date.now() - startTsHealth;
  };
  _perfMark("start");
  // Fast-path local: evita o parser universal quando o valor já vem em formato nativo.
  var parseDataRapida = function(v) {
    if (v instanceof Date) return isNaN(v.getTime()) ? null : v;
    if (v === null || v === undefined || v === "") return null;
    if (typeof v === "number" && !isNaN(v)) {
      var dn = new Date(v);
      return isNaN(dn.getTime()) ? null : dn;
    }
    return parseDataSegura(v);
  };
  
  // =================================================================
  // 1. FASE DE PREPARAÇÃO: FORA DO CADEADO (Multi-Thread)
  // Tudo o que não sofre alterações de outros operadores ao mesmo tempo
  // é calculado aqui para não entupir a fila de espera.
  // =================================================================
  var emailOperador = Session.getActiveUser().getEmail();
  
  if (!emailOperador || emailOperador === "") {
    var retSemSessao = { sucesso: false, codigoErro: "SQ_SESSAO_INVALIDA", mensagem: "Sessão não identificada. Atualiza a página (F5)." };
    if (diag) {
      diag.motivoBloqueio = "SESSAO_VAZIA";
      retSemSessao.debugDiagPedirNovaTarefa = diag;
    }
    return retSemSessao;
  }
  emailOperador = emailOperador.toLowerCase().trim();
  if (diag) diag.emailOperador = emailOperador;

  var ss = ssAtivo || getSpreadsheet();
  var agora = new Date(); 
  var timezone = Session.getScriptTimeZone();
  
  // MAPEAMENTO DOS AGENTES AUSENTES, OFFLINE E SKILLS DO OPERADOR
  var sheetU = ss.getSheetByName('Utilizadores');
  var dataU;
  if (opcoes.dataU) {
    dataU = opcoes.dataU;
  } else if (sheetU && sheetU.getLastRow() > 0) {
    dataU = sheetU.getRange(1, 1, sheetU.getLastRow(), 13).getValues();
  } else {
    dataU = [[]];
  }
  var agentesAusentes = {}; 
  var agentesOffline = {}; 
  var skillsOperador = []; 
  var skillsOperadorMap = {};
  var lojaOperador = "";
  var estadoOperadorAtual = "Offline";
  
  for (var u = 1; u < dataU.length; u++) {
    var mail = dataU[u][0] ? dataU[u][0].toString().toLowerCase().trim() : "";
    if (mail === "") continue;

    if (mail === emailOperador) {
       lojaOperador = dataU[u][2] ? dataU[u][2].toString().trim() : "";
       estadoOperadorAtual = dataU[u][6] ? dataU[u][6].toString().trim() : "Offline";
       var skStr = dataU[u][12] ? dataU[u][12].toString().toLowerCase().trim() : "";
       skillsOperador = skStr !== "" ? skStr.split(",").map(function(s) { return s.trim(); }) : [];
       for (var sk = 0; sk < skillsOperador.length; sk++) {
         if (skillsOperador[sk] !== "") skillsOperadorMap[skillsOperador[sk]] = true;
       }
    }

    var est = dataU[u][6] ? dataU[u][6].toString().trim() : "Offline";
    var dataUltimaMudanca = parseDataSegura(dataU[u][7]);
    var dataPingReal = parseDataSegura(dataU[u][10]);
    var teveAtividadeHoje = false;
    var inicioHoje = new Date();
    inicioHoje.setHours(0,0,0,0);

    if (dataPingReal && dataPingReal >= inicioHoje) {
        teveAtividadeHoje = true;
    }

    // Registo de estados para libertação de casos
    if (est === "Offline") {
      if (!teveAtividadeHoje) {
        agentesOffline[mail] = true;
      }
      // Ausente só após 3 dias úteis completos e sem qualquer atividade hoje.
      if (!teveAtividadeHoje && dataUltimaMudanca && calcularDiasUteis(dataUltimaMudanca, agora) >= 3) {
        agentesAusentes[mail] = true; 
      }
    }
  }

  // Fronteira de limite para as RQS "até hoje" (23:59:59) e SLA
  var limiteHoje = new Date();
  limiteHoje.setHours(23, 59, 59, 999);
  var dataLimiteSLA = calcularDataLimiteSLA();
  var passouDas14 = (agora.getHours() >= 14);

  // =================================================================
  // 2. FASE CRÍTICA: ENTRADA NO CADEADO
  // O sistema bloqueia apenas para Ler a Fila, Escolher e Gravar o "Em Tratamento".
  // =================================================================
  var lock = LockService.getScriptLock();
  var assumeLocked = (opcoes.assumeLocked === true);
  
  try {
    var lockWaitStart = Date.now();
    if (!assumeLocked) {
      lock.waitLock(30000);
    }
    _perfMark("after_waitLock");
    if (perfDetalhe) perfDetalhe.lock_wait_ms = Date.now() - lockWaitStart;

    // Blindagem server-side: só auto-atribui se o operador estiver efetivamente Disponível.
    if (estadoOperadorAtual !== "Disponível") {
      var retNaoDisponivel = { sucesso: false, codigoErro: "SQ_OPERADOR_NAO_DISPONIVEL", mensagem: "Auto-atribuição bloqueada: operador não está Disponível." };
      if (diag) {
        diag.estadoOperadorAtual = estadoOperadorAtual;
        diag.lojaOperador = lojaOperador;
        diag.skillsOperadorCount = skillsOperador.length;
        diag.motivoBloqueio = "OPERADOR_NAO_DISPONIVEL";
        retNaoDisponivel.debugDiagPedirNovaTarefa = diag;
      }
      return retNaoDisponivel;
    }

    var sheet = ss.getSheetByName(ABA_DISTRIBUICAO);
    var ultimaLinha = sheet.getLastRow();
    
    if (ultimaLinha < 2) return { sucesso: false, mensagem: "A grelha está vazia." };
    
    var data;
    var injDist = opcoes.dataDist;
    var usarInjDist = injDist && injDist.length === ultimaLinha;
    if (usarInjDist) {
      for (var pr = 0; pr < ultimaLinha; pr++) {
        if (!injDist[pr] || injDist[pr].length < 16) {
          usarInjDist = false;
          break;
        }
      }
    }
    if (usarInjDist) {
      data = injDist;
    } else {
      _perfMark("before_sheet_read");
      data = sheet.getRange(1, 1, ultimaLinha, 16).getValues();
      _perfMark("after_sheet_read");
    }
    var estadosComAgendamento = { "Agendado": true, "Pendente": true, "Por tratar": true, "Outro": true };
    _perfMark("before_scan");
    var melhorCaso = null;
    var melhorTier = 99;
    var melhorTsCriacao = 0;
    var melhorTsDespertador = 0;
    var melhorIsRqsUrgente = false;

    for (var i = 1; i < data.length; i++) {
      if (diag) diag.contadores.linhasTotais++;
      if (!data[i][1] || data[i][1].toString().trim() === "") {
        if (diag) diag.contadores.semId++;
        continue;
      }

      var estadoLimpo = data[i][9] ? data[i][9].toString().trim() : ""; 
      var responsavel = data[i][10] ? data[i][10].toString().toLowerCase().trim() : ""; 
      var lojaDoCaso = data[i][0] ? data[i][0].toString().trim() : "";

      // 🛡️ CADEADO 1: Ignora sumariamente casos já fechados (Impede Zombie Cases às 14H)
      if (estadoLimpo === "Concluído" || estadoLimpo === "Cancelado") {
        if (diag) diag.contadores.fechado++;
        continue;
      }

      // 🛡️ CADEADO 2: Blindagem universal de Loja (Impede Fuga de Lojas)
      if (lojaDoCaso !== "" && lojaDoCaso !== lojaOperador && responsavel !== emailOperador) {
        if (diag) diag.contadores.filtroLoja++;
        continue;
      }

      var skillDoCaso = data[i][14] ? data[i][14].toString().toLowerCase().trim() : "";
      if (skillDoCaso === "") {
        if (diag) diag.contadores.skillVazia++;
        continue;
      }
      if (!skillsOperadorMap[skillDoCaso] && responsavel !== emailOperador) {
        if (diag) diag.contadores.filtroSkill++;
        continue;
      }

      var isFlashPriority = data[i][15] === true || data[i][15].toString().toUpperCase() === "SIM";
      var dCriaBruta = data[i][5];
      var dRqsBruta = data[i][6];
      var dAgendamentoBruta = data[i][12]; // Coluna M
      
      var dataCriacao = parseDataRapida(dCriaBruta) || agora;
      var dataRqsReal = parseDataRapida(dRqsBruta); // Validação estrita
      var dataRqs = dataRqsReal || agora; // Fallback apenas para cálculos de Tier
      var agendamentoValido = parseDataRapida(dAgendamentoBruta);
      var dataDespertador = agendamentoValido || dataRqs;
      var isPorTratarComRespSemAgenda = (estadoLimpo === "Por tratar" && responsavel !== "" && !agendamentoValido);
      var estadoParaTier = isPorTratarComRespSemAgenda ? "Livre" : estadoLimpo;

      var donoAusente = (responsavel !== "" && agentesAusentes[responsavel] === true);

      // [BLINDAGEM RQS] Só consideramos elegível para queda às 14h se a data for real e válida (até hoje)
      var isRqsHojeOuAtrasada = (dataRqsReal !== null && dataRqsReal <= limiteHoje);
      
      // Validação robusta de Intercalar (Define a Urgência da RQS para Desempate)
      var semIntercalar = true;
      if (data[i][7]) {
          var valInt = data[i][7].toString().trim();
          if (valInt !== "" && valInt !== "null" && valInt !== "undefined") semIntercalar = false;
      }
      
      // Proteção de Agendamento: Aplica-se APENAS aos estados onde o operador define ativamente uma data futura
      var isAgendadoFuturo = (estadosComAgendamento[estadoParaTier] === true && dataDespertador > agora);

      // Só liberta às 14h se: O DONO ESTIVER OFFLINE E não tiver Intercalar E não for agendamento futuro
      var donoEstaOffline = (responsavel !== "" && agentesOffline[responsavel] === true);
      var libertarAs14h = (donoEstaOffline && estadoLimpo !== "Em Tratamento" && estadoLimpo !== "Suspenso" && isRqsHojeOuAtrasada && passouDas14 && semIntercalar && !isAgendadoFuturo);

      // REGRA DONO AUSENTE OU LIBERTADO ÀS 14H: O caso só passa se for da mesma loja do operador a pedir
      if ((donoAusente || libertarAs14h) && responsavel !== emailOperador && lojaDoCaso !== lojaOperador) {
          continue;
      }

      // Validação do SLA para o ecrã do Operador
      var dCriaZero = new Date(dataCriacao.getTime());
      dCriaZero.setHours(0,0,0,0);
      var isSlaAtrasado = (dCriaZero <= dataLimiteSLA);
      
      var tier = 99; 
      
      // Permite que o caso seja apanhado pela loja
      var souODonoOuLivre = (responsavel === "" || responsavel === emailOperador || donoAusente || libertarAs14h);

      // --- AS REGRAS BLINDADAS DE DISTRIBUIÇÃO ---
      
      // 1º ABSOLUTO (Tier -3): Recuperação de sessão (Nunca interromper o que está literalmente aberto agora)
      if (estadoParaTier === "Em Tratamento" && responsavel === emailOperador) tier = -3;
      
      // 2º ABSOLUTO (Tier -2): Retomar o trabalho a meio (Terminar o Suspenso antes de levar com novos casos)
      else if (estadoParaTier === "Suspenso" && responsavel === emailOperador) tier = -2;
      
      // 3º LUGAR (Tier -1): Prioridade Máxima Manual (Flash)
      // Fura a fila de todos os casos Livres e Agendados, MAS respeita o trabalho que o operador tem a meio.
      else if (isFlashPriority && (estadoParaTier === "Livre" || estadoParaTier === "" || responsavel === emailOperador) && souODonoOuLivre) tier = -1;
      
      else if (estadoParaTier === "Suspenso" && responsavel === emailOperador) tier = 0;
      
      // TIER 1 & 2: Agendamentos Expirados E Casos Libertados às 14H
      else if (((estadoParaTier === "Agendado" || estadoParaTier === "Pendente" || estadoParaTier === "Por tratar" || estadoParaTier === "Outro" || estadoParaTier === "Suspenso" || estadoParaTier === "Suspender") && dataDespertador <= agora) || libertarAs14h) {
         if (responsavel === emailOperador) tier = 1; 
         else if (responsavel === "" || donoAusente || libertarAs14h) tier = 2; // Colega apanha de imediato no Tier 2
      }
      
      // TIER 3 & 4: Casos Livres (Atenção à data limite de hoje)
      else if ((estadoParaTier === "Livre" || estadoParaTier === "") && souODonoOuLivre) {
          if (dataRqs <= limiteHoje) {
              tier = 3; 
          } else {
              tier = 4; 
          }
      }

      if (tier !== 99) {
        if (diag) diag.contadores.elegiveis++;
        var tsCriacao = dataCriacao.getTime();
        var tsDespertador = dataDespertador.getTime();
        var isRqsUrgente = (isRqsHojeOuAtrasada && semIntercalar);
        var deveTrocar = false;
        if (!melhorCaso || tier < melhorTier) {
          deveTrocar = true;
        } else if (tier === melhorTier) {
          if (tier === 4 && tsCriacao !== melhorTsCriacao) {
            deveTrocar = tsCriacao < melhorTsCriacao;
          } else if (tsDespertador !== melhorTsDespertador) {
            deveTrocar = tsDespertador < melhorTsDespertador;
          } else if (isRqsUrgente !== melhorIsRqsUrgente) {
            deveTrocar = isRqsUrgente === true;
          } else {
            deveTrocar = tsCriacao < melhorTsCriacao;
          }
        }
        if (deveTrocar) {
          melhorTier = tier;
          melhorTsCriacao = tsCriacao;
          melhorTsDespertador = tsDespertador;
          melhorIsRqsUrgente = isRqsUrgente;
          melhorCaso = {
            linhaIndex: i, loja: data[i][0], idUnico: data[i][1], canal: data[i][2],
            pn: data[i][4] ? data[i][4].toString().trim() : "-",
            dataRqs: dataRqs, dataCriacao: dataCriacao, dataDespertador: dataDespertador, tier: tier,
            intercalar: data[i][7] ? data[i][7].toString() : "",
            observacoes: data[i][8] ? data[i][8].toString() : "",
            isSlaAtrasado: isSlaAtrasado
          };
        }
      } else if (diag) {
        diag.contadores.semTier++;
      }
    }
    _perfMark("after_scan");
    
    if (!melhorCaso) {
      var retSemElegiveis = { sucesso: false, codigoErro: "SQ_SEM_ELEGIVEIS", mensagem: "Sem tarefas disponíveis para o teu perfil." }; // [Ajustada a mensagem]
      if (diag) {
        diag.estadoOperadorAtual = estadoOperadorAtual;
        diag.lojaOperador = lojaOperador;
        diag.skillsOperadorCount = skillsOperador.length;
        diag.motivoBloqueio = "SEM_ELEGIVEIS_APOS_FILTROS";
        retSemElegiveis.debugDiagPedirNovaTarefa = diag;
      }
      return retSemElegiveis;
    }

    var t = melhorCaso; 
    var linhaAtribuida = t.linhaIndex + 1; 
    var linhaAtualReal = sheet.getRange(linhaAtribuida, 1, 1, 16).getValues()[0];
    var estadoAtualReal = linhaAtualReal[9] ? linhaAtualReal[9].toString().trim() : "";
    var responsavelAtualReal = linhaAtualReal[10] ? linhaAtualReal[10].toString().toLowerCase().trim() : "";
    var idAtualReal = _normIdCaso_(linhaAtualReal[1]);
    var idEscolhido = _normIdCaso_(t.idUnico);
    var linhaAindaLivre = (estadoAtualReal === "" || estadoAtualReal === "Livre");
    var linhaAindaMinha = (estadoAtualReal === "Em Tratamento" && responsavelAtualReal === emailOperador);
    var linhaSuspensaMinha = (estadoAtualReal === "Suspenso" && responsavelAtualReal === emailOperador);
    // Alinha a revalidação final com estados já elegíveis no scan (tier 1/2) quando o caso é do próprio operador.
    // Não altera prioridade nem regras de distribuição; apenas evita falso "outro operador" no mesmo responsável.
    var estadosRevalidarMinhaFila = {
      "Agendado": true, "Pendente": true, "Por tratar": true, "Outro": true, "Suspender": true
    };
    var linhaMinhaElegivelNoMesmoDono = (responsavelAtualReal === emailOperador && estadosRevalidarMinhaFila[estadoAtualReal] === true);
    var estadosElegiveisDonoAusente = {
      "Suspenso": true, "Agendado": true, "Pendente": true, "Por tratar": true, "Outro": true, "Suspender": true, "Livre": true, "": true
    };
    var donoAtualAusente = (responsavelAtualReal !== "" && agentesAusentes[responsavelAtualReal] === true);
    var dRqsRevalidacao = parseDataRapida(linhaAtualReal[6]);
    var dAgendRevalidacao = parseDataRapida(linhaAtualReal[12]);
    var semIntercalarRevalidacao = true;
    if (linhaAtualReal[7]) {
      var valIntRevalid = linhaAtualReal[7].toString().trim();
      if (valIntRevalid !== "" && valIntRevalid !== "null" && valIntRevalid !== "undefined") semIntercalarRevalidacao = false;
    }
    var isRqsHojeOuAtrasadaRevalidacao = (dRqsRevalidacao !== null && dRqsRevalidacao <= limiteHoje);
    var dataDespertadorRevalidacao = dAgendRevalidacao || dRqsRevalidacao || agora;
    var isAgendadoFuturoRevalidacao = (estadosComAgendamento[estadoAtualReal] === true && dataDespertadorRevalidacao > agora);
    var libertarAs14hRevalidacao = (
      responsavelAtualReal !== "" &&
      agentesOffline[responsavelAtualReal] === true &&
      estadoAtualReal !== "Em Tratamento" &&
      estadoAtualReal !== "Suspenso" &&
      isRqsHojeOuAtrasadaRevalidacao &&
      passouDas14 &&
      semIntercalarRevalidacao &&
      !isAgendadoFuturoRevalidacao
    );
    var mesmaLojaRevalidacao = ((linhaAtualReal[0] ? linhaAtualReal[0].toString().trim() : "") === lojaOperador);
    // Alinha a revalidação com a regra de scan (dono ausente OU libertação às 14h) na mesma loja.
    // Não altera prioridade; só evita falso conflito na reserva final.
    var linhaElegivelPorDonoAusente = (
      (donoAtualAusente || libertarAs14hRevalidacao) &&
      responsavelAtualReal !== emailOperador &&
      mesmaLojaRevalidacao &&
      estadosElegiveisDonoAusente[estadoAtualReal] === true
    );
    var linhaAtribuivel = (idAtualReal === idEscolhido) && (linhaAindaLivre || linhaAindaMinha || linhaSuspensaMinha || linhaMinhaElegivelNoMesmoDono || linhaElegivelPorDonoAusente);

    // Revalidação robusta: se a linha mudou de posição entretanto, relocaliza pelo idUnico.
    // Não altera regras/tier; apenas evita falso conflito quando a linha foi deslocada por outra escrita.
    if (!linhaAtribuivel && idAtualReal !== idEscolhido && idEscolhido !== "") {
      var dadosRevalidacao = sheet.getRange(1, 2, ultimaLinha, 10).getValues(); // B:K (ID + estado/responsável)
      var linhaRelocalizada = -1;
      for (var rr = 1; rr < dadosRevalidacao.length; rr++) {
        if (_normIdCaso_(dadosRevalidacao[rr][0]) === idEscolhido) {
          linhaRelocalizada = rr + 1;
          break;
        }
      }
      if (linhaRelocalizada > 0 && linhaRelocalizada !== linhaAtribuida) {
        linhaAtribuida = linhaRelocalizada;
        linhaAtualReal = sheet.getRange(linhaAtribuida, 1, 1, 16).getValues()[0];
        estadoAtualReal = linhaAtualReal[9] ? linhaAtualReal[9].toString().trim() : "";
        responsavelAtualReal = linhaAtualReal[10] ? linhaAtualReal[10].toString().toLowerCase().trim() : "";
        idAtualReal = _normIdCaso_(linhaAtualReal[1]);
        linhaAindaLivre = (estadoAtualReal === "" || estadoAtualReal === "Livre");
        linhaAindaMinha = (estadoAtualReal === "Em Tratamento" && responsavelAtualReal === emailOperador);
        linhaSuspensaMinha = (estadoAtualReal === "Suspenso" && responsavelAtualReal === emailOperador);
        linhaMinhaElegivelNoMesmoDono = (responsavelAtualReal === emailOperador && estadosRevalidarMinhaFila[estadoAtualReal] === true);
        donoAtualAusente = (responsavelAtualReal !== "" && agentesAusentes[responsavelAtualReal] === true);
        dRqsRevalidacao = parseDataRapida(linhaAtualReal[6]);
        dAgendRevalidacao = parseDataRapida(linhaAtualReal[12]);
        semIntercalarRevalidacao = true;
        if (linhaAtualReal[7]) {
          var valIntRevalid2 = linhaAtualReal[7].toString().trim();
          if (valIntRevalid2 !== "" && valIntRevalid2 !== "null" && valIntRevalid2 !== "undefined") semIntercalarRevalidacao = false;
        }
        isRqsHojeOuAtrasadaRevalidacao = (dRqsRevalidacao !== null && dRqsRevalidacao <= limiteHoje);
        dataDespertadorRevalidacao = dAgendRevalidacao || dRqsRevalidacao || agora;
        isAgendadoFuturoRevalidacao = (estadosComAgendamento[estadoAtualReal] === true && dataDespertadorRevalidacao > agora);
        libertarAs14hRevalidacao = (
          responsavelAtualReal !== "" &&
          agentesOffline[responsavelAtualReal] === true &&
          estadoAtualReal !== "Em Tratamento" &&
          estadoAtualReal !== "Suspenso" &&
          isRqsHojeOuAtrasadaRevalidacao &&
          passouDas14 &&
          semIntercalarRevalidacao &&
          !isAgendadoFuturoRevalidacao
        );
        mesmaLojaRevalidacao = ((linhaAtualReal[0] ? linhaAtualReal[0].toString().trim() : "") === lojaOperador);
        linhaElegivelPorDonoAusente = (
          (donoAtualAusente || libertarAs14hRevalidacao) &&
          responsavelAtualReal !== emailOperador &&
          mesmaLojaRevalidacao &&
          estadosElegiveisDonoAusente[estadoAtualReal] === true
        );
        linhaAtribuivel = (idAtualReal === idEscolhido) && (linhaAindaLivre || linhaAindaMinha || linhaSuspensaMinha || linhaMinhaElegivelNoMesmoDono || linhaElegivelPorDonoAusente);
        if (diag) {
          diag.relocalizacaoReserva = {
            linhaOriginal: t.linhaIndex + 1,
            linhaRelocalizada: linhaRelocalizada,
            idEscolhido: idEscolhido,
            idAtualRelocalizado: idAtualReal
          };
        }
      }
    }
    var houveTrocaId = (idAtualReal !== idEscolhido);
    var houveDonoDiferente = (responsavelAtualReal !== "" && responsavelAtualReal !== emailOperador);
    var evidenciaConcorrencia = (houveTrocaId || houveDonoDiferente);
    if (!linhaAtribuivel && permitirRetryConflito) {
      // Revalidação falhou por corrida: atualiza snapshot da linha e tenta novo candidato.
      // Mantém regras de distribuição; apenas evita falha precoce quando há concorrência alta.
      data[t.linhaIndex] = linhaAtualReal;
      var maxTentativasConflito = 5;
      for (var tentConf = 1; tentConf <= maxTentativasConflito; tentConf++) {
        // Pequeno jitter para reduzir colisões repetidas entre vários operadores.
        Utilities.sleep(60 + Math.floor(Math.random() * 120));
        var tentativaConflito = pedirNovaTarefa(ss, {
          assumeLocked: true,
          skipSpreadsheetFlush: opcoes.skipSpreadsheetFlush === true,
          dataU: dataU,
          retryConflito: false
        });
        if (tentativaConflito && tentativaConflito.sucesso) {
          return tentativaConflito;
        }
        var msgTentativaConflito = tentativaConflito && tentativaConflito.mensagem ? tentativaConflito.mensagem.toString().toLowerCase() : "";
        if (msgTentativaConflito.indexOf("fila atualizada por outro operador") === -1) {
          return tentativaConflito;
        }
      }

      var retConflito = {
        sucesso: false,
        codigoErro: "SQ_RESERVA_FALHOU_PERSISTENTE",
        detalheReserva: evidenciaConcorrencia ? "CONCORRENCIA_REAL" : "INCONSISTENCIA_LOCAL",
        conflitoDetalhe: {
          idEscolhido: idEscolhido,
          idAtualReal: idAtualReal,
          estadoAtualReal: estadoAtualReal,
          responsavelAtualReal: responsavelAtualReal,
          linhaAtribuida: linhaAtribuida
        },
        mensagem: "Não foi possível reservar um caso elegível neste momento. Mantém-te Disponível: vamos continuar a tentar automaticamente."
      };
      if (diag) {
        diag.estadoOperadorAtual = estadoOperadorAtual;
        diag.lojaOperador = lojaOperador;
        diag.skillsOperadorCount = skillsOperador.length;
        diag.candidato = {
          id: t.idUnico,
          linha: linhaAtribuida,
          tier: t.tier
        };
        diag.revalidacao = {
          idAtualReal: idAtualReal,
          idEscolhido: idEscolhido,
          estadoAtualReal: estadoAtualReal,
          responsavelAtualReal: responsavelAtualReal,
          linhaAindaLivre: linhaAindaLivre,
          linhaAindaMinha: linhaAindaMinha,
          linhaSuspensaMinha: linhaSuspensaMinha,
          linhaMinhaElegivelNoMesmoDono: linhaMinhaElegivelNoMesmoDono,
          idConfere: (idAtualReal === idEscolhido)
        };
        diag.motivoBloqueio = "REVALIDACAO_FALHOU";
        diag.tentativasConflito = maxTentativasConflito;
        retConflito.debugDiagPedirNovaTarefa = diag;
      }
      return retConflito;
    }
    if (!linhaAtribuivel) {
      var retConflitoSemRetry = {
        sucesso: false,
        codigoErro: "SQ_RESERVA_FALHOU",
        detalheReserva: evidenciaConcorrencia ? "CONCORRENCIA_REAL" : "INCONSISTENCIA_LOCAL",
        conflitoDetalhe: {
          idEscolhido: idEscolhido,
          idAtualReal: idAtualReal,
          estadoAtualReal: estadoAtualReal,
          responsavelAtualReal: responsavelAtualReal,
          linhaAtribuida: linhaAtribuida
        },
        mensagem: "Não foi possível reservar o caso escolhido neste ciclo. Continua em Disponível e volta a tentar dentro de instantes."
      };
      if (diag) {
        diag.estadoOperadorAtual = estadoOperadorAtual;
        diag.lojaOperador = lojaOperador;
        diag.skillsOperadorCount = skillsOperador.length;
        diag.candidato = { id: t.idUnico, linha: linhaAtribuida, tier: t.tier };
        diag.revalidacao = {
          idAtualReal: idAtualReal,
          idEscolhido: idEscolhido,
          estadoAtualReal: estadoAtualReal,
          responsavelAtualReal: responsavelAtualReal,
          linhaAindaLivre: linhaAindaLivre,
          linhaAindaMinha: linhaAindaMinha,
          linhaSuspensaMinha: linhaSuspensaMinha,
          linhaMinhaElegivelNoMesmoDono: linhaMinhaElegivelNoMesmoDono,
          idConfere: (idAtualReal === idEscolhido)
        };
        diag.motivoBloqueio = "REVALIDACAO_FALHOU_SEM_RETRY";
        retConflitoSemRetry.debugDiagPedirNovaTarefa = diag;
      }
      return retConflitoSemRetry;
    }
    
    // Grava o "Em Tratamento" (+ coluna P em RAM) num único setValues A:P
    var linhaGravar = _linhaDistrib16(linhaAtualReal);
    linhaGravar[9] = "Em Tratamento";
    linhaGravar[10] = emailOperador;
    linhaGravar[11] = agora;
    // Prioridade manual (Flash, col. P): consumir ao cair ao colaborador. Antes só limpava em tier -2;
    // com SIM em tier -1 (ou noutros tiers elegíveis) a flag ficava na folha e o caso voltava a furar a fila.
    var priCel = data[t.linhaIndex][15];
    var tinhaFlash = (priCel === true || (priCel != null && priCel !== "" && priCel.toString().toUpperCase().trim() === "SIM"));
    if (t.tier === -2 || tinhaFlash) {
      linhaGravar[15] = "";
    }
    for (var cg = 0; cg < 16; cg++) {
      data[t.linhaIndex][cg] = linhaGravar[cg];
    }
    _perfMark("before_write");
    sheet.getRange(linhaAtribuida, 1, 1, 16).setValues([linhaGravar]);
    _perfMark("after_write");
    
    if (!opcoes.skipSpreadsheetFlush) {
      _perfMark("before_flush");
      SpreadsheetApp.flush();
      _perfMark("after_flush");
    }
    _perfMark("before_return");
    var strCriacao = Utilities.formatDate(t.dataCriacao, timezone, "dd/MM/yyyy");
    var strRqs = Utilities.formatDate(t.dataRqs, timezone, "dd/MM/yyyy");
    
    // --- ESTE É O RETURN QUE FALTAVA ATUALIZAR ---
    var resposta = { 
      sucesso: true, 
      tarefa: { 
        idUnico: t.idUnico, 
        loja: t.loja, 
        canal: t.canal, 
        pn: t.pn, // <--- PASSA O PN PARA A TAREFA
        linha: linhaAtribuida, 
        dataRqsIso: t.dataRqs.toISOString(),
        dataDespertadorIso: t.dataDespertador.toISOString(),
        intercalar: t.intercalar,
        dataCriacaoVisivel: strCriacao,
        dataRqsVisivel: strRqs,
        observacoes: t.observacoes,
        isSlaAtrasado: t.isSlaAtrasado // <--- ERA ISTO QUE FALTAVA NA ENCOMENDA!
      } 
    };
    if (perfDetalhe) {
      perfDetalhe.total_ms = Date.now() - startTsHealth;
      resposta.debugPedirNovaTarefa = perfDetalhe;
    }
    if (diag) {
      diag.estadoOperadorAtual = estadoOperadorAtual;
      diag.lojaOperador = lojaOperador;
      diag.skillsOperadorCount = skillsOperador.length;
      diag.candidato = { id: t.idUnico, linha: linhaAtribuida, tier: t.tier };
      diag.motivoBloqueio = "SUCESSO";
      resposta.debugDiagPedirNovaTarefa = diag;
    }
    return resposta;

  } catch (e) { 
    return { sucesso: false, codigoErro: "SQ_PROCESSAMENTO_ERRO", mensagem: "Erro no processamento: " + e.message }; 
  } finally { 
    if (!assumeLocked) {
      lock.releaseLock();
    }
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("pedirNovaTarefa", new Date().getTime() - startTsHealth);
  }
}

function pedirTarefaEspecifica(idTarefa) {
  var startTsHealth = new Date().getTime();
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(30000); 
    
    // 1. Ler o e-mail da sessão ativa
    var emailOperador = Session.getActiveUser().getEmail();
    
    // 2. CORREÇÃO: Bloquear se a sessão estiver vazia (em vez de herdar super admin)
    if (!emailOperador || emailOperador === "") {
      return { sucesso: false, mensagem: "Sessão não identificada. Por favor, faz F5 na página para atualizar a ligação." };
    }
    emailOperador = emailOperador.toLowerCase().trim();

    var sheet = getSpreadsheet().getSheetByName(ABA_DISTRIBUICAO);
    var data = sheet.getRange(1, 1, sheet.getLastRow(), 11).getValues(); 
    var timezone = Session.getScriptTimeZone();
    var linhaEncontrada = -1; 
    var tarefa = null;

    var idNormEsp = _normIdCaso_(idTarefa);
    if (idNormEsp === "") {
      return { sucesso: false, mensagem: "Identificador do caso em falta." };
    }

    for (var i = 1; i < data.length; i++) {
      if (_normIdCaso_(data[i][1]) !== idNormEsp) continue;
      var est = data[i][9] ? data[i][9].toString().trim() : "";
      var resp = data[i][10] ? data[i][10].toString().toLowerCase().trim() : "";

      if (est === "Concluído") return { sucesso: false, mensagem: "O caso já está Concluído." };
      
      // Regra: Se está Em Tratamento, mas não é por mim, bloqueia. Se for por mim, deixa avançar!
      if (est === "Em Tratamento" && resp !== emailOperador && resp !== "") {
        return { sucesso: false, mensagem: "O caso já está a ser tratado por " + resp + "." };
      }

      linhaEncontrada = i + 1; 
      
      // Conversão robusta: usa o Universal Parser para lidar com strings europeias
      var dCria = parseDataSegura(data[i][5]) || new Date();
      var dRqs = parseDataSegura(data[i][6]) || new Date();

      tarefa = { 
        idUnico: data[i][1], 
        loja: data[i][0], 
        canal: data[i][2], 
        pn: data[i][4] ? data[i][4].toString().trim() : "-", 
        linha: linhaEncontrada,
        dataRqsIso: dRqs.toISOString(), 
        intercalar: data[i][7] ? data[i][7].toString() : "", 
        dataCriacaoVisivel: Utilities.formatDate(dCria, timezone, "dd/MM/yyyy"), 
        // Usa dRqs (SLA) e não o despertador para a data visível
        dataRqsVisivel: Utilities.formatDate(dRqs, timezone, "dd/MM/yyyy"),
        observacoes: data[i][8] ? data[i][8].toString() : ""
      };
      break;
    }
    
    if (linhaEncontrada === -1) return { sucesso: false, mensagem: "Caso não encontrado na base de dados." };
    
    // Grava o estado como Em Tratamento com o e-mail real do operador
    sheet.getRange(linhaEncontrada, 10, 1, 3).setValues([["Em Tratamento", emailOperador, new Date()]]);
    
    return { sucesso: true, tarefa: tarefa };
    
  } catch (e) { 
    return { sucesso: false, mensagem: "Erro no processamento: " + e.message }; 
  } finally { 
    lock.releaseLock(); 
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("pedirTarefaEspecifica", new Date().getTime() - startTsHealth);
  }
}

/**
 * Validação leve de posse do caso ativo no frontend.
 * Evita manter cartão aberto quando o caso já mudou de dono/estado no servidor.
 */
function verificarPosseCasoAtivoServidor(idTarefa) {
  var startTsHealth = new Date().getTime();
  try {
    var emailSessao = Session.getActiveUser().getEmail();
    if (!emailSessao || emailSessao === "") {
      return { sucesso: false, ejetar: true, mensagem: "Sessão não identificada. Faz F5 para sincronizar." };
    }
    emailSessao = emailSessao.toLowerCase().trim();

    var idNorm = _normIdCaso_(idTarefa);
    if (idNorm === "") {
      return { sucesso: false, ejetar: true, mensagem: "Caso ativo inválido. Atualiza a página." };
    }

    var sheet = getSpreadsheet().getSheetByName(ABA_DISTRIBUICAO);
    var lr = sheet.getLastRow();
    if (lr < 2) {
      return { sucesso: false, ejetar: true, mensagem: "Caso já não existe na fila." };
    }

    var ids = sheet.getRange(1, 2, lr, 1).getValues();      // Col B
    var estadosResp = sheet.getRange(1, 10, lr, 2).getValues(); // Cols J:K
    for (var i = 1; i < ids.length; i++) {
      if (_normIdCaso_(ids[i][0]) !== idNorm) continue;
      var estado = estadosResp[i][0] ? estadosResp[i][0].toString().trim() : "";
      var resp = estadosResp[i][1] ? estadosResp[i][1].toString().toLowerCase().trim() : "";
      var emPosse = (estado === "Em Tratamento" && resp === emailSessao);
      if (emPosse) return { sucesso: true, emPosse: true };
      return {
        sucesso: true,
        emPosse: false,
        ejetar: true,
        mensagem: "⚠️ Este caso foi atualizado no servidor. O painel será sincronizado."
      };
    }

    return { sucesso: true, emPosse: false, ejetar: true, mensagem: "⚠️ O caso já não está disponível no Smart Queue." };
  } catch (e) {
    return { sucesso: false, mensagem: e.message };
  } finally {
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("verificarPosseCasoAtivoServidor", new Date().getTime() - startTsHealth);
  }
}

// --- FINALIZAR TAREFA E PREENCHER LOJA NA COLUNA A ---
// --- FINALIZAR TAREFA E PREENCHER LOJA NA COLUNA A (FLUXO 2 EM 1: batch A:P + matrizes inject + flush único) ---
function finalizarTarefaServidor(idTarefa, estadoFinal, observacoes, dataAgendamento, opcoes) {
  var startTsHealth = new Date().getTime();
  try {
    opcoes = opcoes || {};
    var suprimirAutoAtribuicao = (opcoes.suprimirAutoAtribuicao === true);
    var estadoAlvoAposFinalizacao = (opcoes.estadoAlvoAposFinalizacao || "").toString().trim();
    var emailSessao = Session.getActiveUser().getEmail();
    if (!emailSessao || emailSessao === "") {
      return { sucesso: false, erroSilencioso: true, mensagem: "Sessão não identificada. Faz F5 e tenta novamente." };
    }
    emailSessao = emailSessao.toString().toLowerCase().trim();
    var ss = getSpreadsheet();
    var sheetDist = ss.getSheetByName(ABA_DISTRIBUICAO);
    var ultLR = sheetDist.getLastRow();
    var dataDist = ultLR > 0 ? sheetDist.getRange(1, 1, ultLR, 16).getValues() : [];
    var idNormFin = _normIdCaso_(idTarefa);
    if (idNormFin === "") {
      return { sucesso: false, erroSilencioso: true, mensagem: "Identificador do caso em falta." };
    }

    var linha = -1;
    var dadosLinha = null;
    var linhaFallback = -1;
    var dadosLinhaFallback = null;

    for (var i = 1; i < dataDist.length; i++) {
      if (_normIdCaso_(dataDist[i][1]) !== idNormFin) continue;
      if (linhaFallback === -1) {
        linhaFallback = i + 1;
        dadosLinhaFallback = dataDist[i];
      }
      var estLinha = dataDist[i][9] ? dataDist[i][9].toString().trim() : "";
      var respLinha = dataDist[i][10] ? dataDist[i][10].toString().toLowerCase().trim() : "";
      // Preferir a linha efetivamente ativa do operador da sessão.
      if (estLinha === "Em Tratamento" && respLinha === emailSessao) {
        linha = i + 1;
        dadosLinha = dataDist[i];
        break;
      }
    }

    if (linha === -1 && linhaFallback !== -1) {
      linha = linhaFallback;
      dadosLinha = dadosLinhaFallback;
    }
    if (linha === -1) return { sucesso: false, erroSilencioso: true, mensagem: "Caso não encontrado." };

    // Impede fecho na linha "errada" do mesmo ID (ex.: duplicado / fallback) sem ser o responsável.
    var respLinhaEfetivo = dadosLinha[10] ? dadosLinha[10].toString().toLowerCase().trim() : "";
    if (respLinhaEfetivo !== emailSessao) {
      return {
        sucesso: false,
        ejetar: true,
        erroSilencioso: true,
        mensagem: "⚠️ Este caso não está atribuído à tua sessão. Atualiza a página ou volta a puxar o caso em Trabalhos manuais."
      };
    }

    // 🛡️ CADEADO DE SINCRONIZAÇÃO: Impede dupla gravação e dispara Ejeção do Frontend
    var estadoAtual = dadosLinha[9] ? dadosLinha[9].toString().trim() : "";
    if (estadoAtual !== "Em Tratamento") {
        return { sucesso: false, ejetar: true, erroSilencioso: true, mensagem: "⚠️ O caso já não está em tratamento. O painel foi sincronizado para continuares." };
    }

    var idx = linha - 1;

    // --- MAPA EXATO DAS TUAS COLUNAS ---
    var emailOperador = dadosLinha[10]; // Coluna K (Índice 10)
    if (!emailOperador || emailOperador.toString().trim() === "") {
      emailOperador = emailSessao;
    }
    emailOperador = emailOperador.toString().toLowerCase().trim();
    var dataInicio = dadosLinha[11];    // Coluna L (Índice 11)
    var obsAntigas = dadosLinha[8] ? dadosLinha[8].toString().trim() : ""; // Coluna I (Índice 8)
    var dataFim = new Date();

    // 1. Histórico Cumulativo de Observações
    var obsFinal = obsAntigas;
    if (observacoes && observacoes.trim() !== "") {
       var carimbo = Utilities.formatDate(dataFim, Session.getScriptTimeZone(), "dd/MM HH:mm");
       var userAbreviado = emailOperador.split('@')[0]; // Fica só com o 1º nome do email
       var novaEntrada = "[" + carimbo + " - " + userAbreviado + "]\n" + observacoes.trim();
       
       obsFinal = obsAntigas === "" ? novaEntrada : novaEntrada + "\n\n" + obsAntigas;
    }

    // Linha A:P em RAM (igual ao efeito das antigas escritas I–L, M e A)
    var baseRow = _linhaDistrib16(dadosLinha);
    baseRow[8] = obsFinal;
    baseRow[9] = estadoFinal;
    // Em estados fechados não manter responsável preenchido evita reaberturas ambíguas por dados sujos.
    baseRow[10] = (estadoFinal === "Concluído" || estadoFinal === "Cancelado") ? "" : emailOperador;
    baseRow[11] = "";

    var persistirLinhaParcialIemL = function() {
      for (var pc = 0; pc < 16; pc++) {
        dataDist[idx][pc] = baseRow[pc];
      }
      sheetDist.getRange(linha, 1, 1, 16).setValues([baseRow]);
    };

    // 1.2 Gravar a Data de Agendamento (O Despertador) para TODOS os estados que a exigem
    var estadosComRelogio = ["Agendado", "Pendente", "Por tratar", "Outro", "Suspenso", "Suspender"];
    
    if (estadosComRelogio.includes(estadoFinal) && dataAgendamento) {
      var dataAgend = parseDataSegura(dataAgendamento);
      if (!dataAgend) {
        persistirLinhaParcialIemL();
        return { sucesso: false, erroSilencioso: true, mensagem: "Erro: A data de agendamento é inválida." };
      }
      var dataRqsOriginal = parseDataSegura(dadosLinha[6]); // Coluna G (Índice 6)
      var intercalarMarker = dadosLinha[7]; // Coluna H (Índice 7)
      var hasIntercalar = (intercalarMarker && intercalarMarker.toString().trim() !== "");
      
      // Validação SLA: Não pode agendar para depois da RQS (Ignora se for uma Suspensão Automática pelo Sistema)
      var isAcaoAutomatica = (observacoes && observacoes.indexOf("[Auto]") > -1);
      if (!hasIntercalar && dataRqsOriginal && !isAcaoAutomatica) {
        var limiteRqs = new Date(dataRqsOriginal.getTime());
        limiteRqs.setHours(20, 0, 0, 0); // Permite agendar até às 20:00 do dia da RQS
        
        if (dataAgend > limiteRqs) {
          var rqsFmt = Utilities.formatDate(limiteRqs, Session.getScriptTimeZone(), "dd/MM/yyyy HH:mm");
          persistirLinhaParcialIemL();
          return { sucesso: false, erroSilencioso: true, mensagem: "⚠️ Erro: Sem intercalar marcada, só podes agendar até ao limite da RQS (" + rqsFmt + ")." };
        }
      }

      // Validação de segurança: Limite de 7 dias (Aplica-se sempre)
      var limiteMax = new Date();
      limiteMax.setDate(limiteMax.getDate() + 7);
      limiteMax.setHours(23, 59, 59, 999);
      if (dataAgend > limiteMax) {
        persistirLinhaParcialIemL();
        return { sucesso: false, erroSilencioso: true, mensagem: "Erro: Não podes agendar com mais de 7 dias de antecedência." };
      }

      baseRow[12] = dataAgend; // Coluna M — Objeto Date (igual ao setValue anterior)
    }

    // 2. Preencher a Coluna A (Loja) e Captar Estado do Operador
    var lojaOp = "";
    var estadoAtualOp = "Offline";
    var sheetU = ss.getSheetByName('Utilizadores');
    var dataU = [[]];
    if (sheetU && sheetU.getLastRow() > 0) {
      dataU = sheetU.getRange(1, 1, sheetU.getLastRow(), 13).getValues();
      for (var u = 1; u < dataU.length; u++) {
         if (dataU[u][0] && dataU[u][0].toString().toLowerCase().trim() === emailOperador.toString().toLowerCase().trim()) {
             lojaOp = dataU[u][2] ? dataU[u][2].toString().trim() : ""; // Coluna C dos Utilizadores
             estadoAtualOp = dataU[u][6] ? dataU[u][6].toString().trim() : "Offline"; // Captura estado
             break;
         }
      }
    }
    baseRow[0] = lojaOp;

    for (var fc = 0; fc < 16; fc++) {
      dataDist[idx][fc] = baseRow[fc];
    }
    sheetDist.getRange(linha, 1, 1, 16).setValues([baseRow]);

    // 3. Gravar no Log de Performance (Agora 7 Colunas com Duração)
    var sheetLog = ss.getSheetByName(ABA_LOG);
    if (sheetLog) {
      var diffS = 0;
      var duracaoFormatada = "00:00";
      
      // Lógica de cálculo da duração
      if (dataInicio && dataFim) {
         var dInicio = new Date(dataInicio);
         if (!isNaN(dInicio.getTime())) {
            diffS = Math.floor((dataFim.getTime() - dInicio.getTime()) / 1000);
            if (diffS < 0) diffS = 0; // Proteção contra tempos negativos
            
            var h = Math.floor(diffS / 3600);
            var m = Math.floor((diffS % 3600) / 60).toString().padStart(2, '0');
            var s = (diffS % 60).toString().padStart(2, '0');
            duracaoFormatada = (h > 0 ? h + ":" : "") + m + ":" + s;
         }
      }

      var skillPerf = dadosLinha.length > 14 && dadosLinha[14] != null ? dadosLinha[14].toString().trim() : "";
      sheetLog.appendRow([
        idNormFin,        // Coluna A (ID normalizado, alinhado ao lookup)
        emailOperador,    // Coluna B
        dataInicio,       // Coluna C
        dataFim,          // Coluna D
        estadoFinal,      // Coluna E
        duracaoFormatada, // Coluna F [NOVO: HH:MM:SS ou MM:SS]
        diffS,            // Coluna G [NOVO: Total Segundos para Looker Studio]
        skillPerf         // Coluna H — Skill (folha Distribuição col. O)
      ]);
    }

    try { _limparCacheStatsOperador_(emailSessao); } catch (_eClrSt) {}

    // 4. LÓGICA 2-EM-1 (AUTO-ATRIBUIÇÃO) — matrizes em RAM + sem flush intermédio
    var novaTarefaObj = null;
    var permiteAutoAtribuicaoPosFinalizacao = (estadoAlvoAposFinalizacao === "" || estadoAlvoAposFinalizacao === "Disponível");
    if (!suprimirAutoAtribuicao && estadoAtualOp === "Disponível" && permiteAutoAtribuicaoPosFinalizacao) {
        var pedido = pedirNovaTarefa(ss, {
          dataU: dataU,
          skipSpreadsheetFlush: true
        });
        if (pedido && pedido.sucesso) {
            novaTarefaObj = pedido.tarefa;
        }
    }

    SpreadsheetApp.flush();

    if (novaTarefaObj) {
        return { sucesso: true, mensagem: "Caso fechado com sucesso! Nova tarefa iniciada.", novaTarefa: novaTarefaObj };
    } else {
        return { sucesso: true, mensagem: "Caso fechado com sucesso!" };
    }

  } catch(e) {
    return { sucesso: false, erroSilencioso: true, mensagem: "Erro: " + e.message };
  } finally {
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("finalizarTarefaServidor", new Date().getTime() - startTsHealth);
  }
}

// --- DADOS PARA O DASHBOARD DE SUPERVISÃO (REFATORADO PARA ALTA PERFORMANCE) ---
// --- DADOS PARA O DASHBOARD DE SUPERVISÃO (REFATORADO PARA ALTA PERFORMANCE COM CHUNKS) ---
function obterDadosSupervisao(skillsFiltro) {
  var startTsHealth = new Date().getTime();
  try {
    // SETUP DE FILTROS E CHAVE DE CACHE
    var filtroAtivo = (skillsFiltro && skillsFiltro.length > 0);
    var skillsFiltroLower = filtroAtivo ? skillsFiltro.map(function(s) { return s.toLowerCase().trim(); }) : [];
    var cacheKey = "supervisao_" + (skillsFiltroLower.length > 0 ? skillsFiltroLower.join("_") : "all");

    // CRÍTICO: Bypass Google Cache Limit (>100KB) com vida útil de 60 segundos
    var cachedData = _getCacheChunked(cacheKey);
    if (cachedData) {
        return JSON.parse(cachedData);
    }

    var ss = getSpreadsheet();
    var agora = new Date();
    var hojeInicio = new Date(); hojeInicio.setHours(0,0,0,0);
    var hojeFim = new Date(); hojeFim.setHours(23,59,59,999);
    var inicioMes = new Date(agora.getFullYear(), agora.getMonth(), 1);
    var fimMes = new Date(agora.getFullYear(), agora.getMonth() + 1, 0);
    var dataLimiteSLA = calcularDataLimiteSLA();
    var timezone = Session.getScriptTimeZone();
    var mesAtualStr = Utilities.formatDate(agora, timezone, "MM/yyyy");
    var anoMesCache = Utilities.formatDate(agora, timezone, "yyyyMM");

    // ------------------------------------------------------------------------
    // 1. MANDAMENTO: BATCH OPERATIONS (Ler tudo para Memória no arranque c/ GETRANGE exatos)
    // ------------------------------------------------------------------------
    var sheetConfig = ss.getSheetByName('Config_Skills');
    var dataConfig = (sheetConfig && sheetConfig.getLastRow() > 0) ? sheetConfig.getDataRange().getValues() : [];

    var sheetD = ss.getSheetByName(ABA_DISTRIBUICAO);
    // [OTIMIZAÇÃO] Em vez de getDataRange(), ler exatamente as 16 colunas necessárias (A a P)
    var dataD = (sheetD && sheetD.getLastRow() > 0) ? sheetD.getRange(1, 1, sheetD.getLastRow(), 16).getValues() : [];

    var sheetLog = ss.getSheetByName(ABA_LOG);
    var dataLog = sheetLog ? sheetLog.getDataRange().getValues() : [];

    var sheetU = ss.getSheetByName('Utilizadores');
    // [OTIMIZAÇÃO] Em vez de getDataRange(), ler exatamente as 11 colunas necessárias (A a K) + M(13) porque skills = Col 13
    var dataU = (sheetU && sheetU.getLastRow() > 0) ? sheetU.getRange(1, 1, sheetU.getLastRow(), 13).getValues() : [];

    var sheetObj = ss.getSheetByName('Objetivos_Mensais');
    var dataObj = sheetObj ? sheetObj.getDataRange().getValues() : [];

    var skillsMaster = [];
    if (dataConfig.length > 0) {
      for (var c = 0; c < dataConfig.length; c++) {
        if (dataConfig[c][0]) skillsMaster.push(dataConfig[c][0].toString());
      }
    }

    // ------------------------------------------------------------------------
    // 2. PROCESSAMENTO EM MEMÓRIA: Dicionários Base
    // ------------------------------------------------------------------------
    var skillDoCaso = {};
    for (var j = 1; j < dataD.length; j++) {
       var idCasoD = dataD[j][1] ? dataD[j][1].toString().trim() : "";
       var sk = dataD[j][14] ? dataD[j][14].toString().toLowerCase().trim() : "";
       if (idCasoD !== "") skillDoCaso[idCasoD] = sk;
    }

    var emailParaLoja = {};
    for (var u = 1; u < dataU.length; u++) {
       var mailU = dataU[u][0] ? dataU[u][0].toString().toLowerCase().trim() : "";
       var ljU = dataU[u][2] ? dataU[u][2].toString().trim() : "";
       if (mailU !== "") emailParaLoja[mailU] = ljU;
    }

    // ------------------------------------------------------------------------
    // 3. LOG DE PERFORMANCE: Cálculo Global, Agentes e Lojas de uma só vez
    // ------------------------------------------------------------------------
    var statsAgentes = {};
    var globalTratadas = 0; var globalConcluidas = 0; var globalTempo = 0;
    var memoriaUnicosAgente = {}; 
    var memoriaUnicosGlobal = {};
    var memoriaConcluidosAgente = {};
    var memoriaGlobalConcluidos = {};

    var concluidasMesLoja = {};
    var concluidasHojeLoja = {};
    var unicosMesLoja = {};
    var unicosHojeLoja = {};

    for (var k = 1; k < dataLog.length; k++) {
      var idCasoLog = dataLog[k][0] ? dataLog[k][0].toString().trim() : "";
      var emailLog = dataLog[k][1] ? dataLog[k][1].toString().toLowerCase().trim() : "";
      var dInicio = new Date(dataLog[k][2]);
      var dFim = new Date(dataLog[k][3]);
      var estFinal = dataLog[k][4] ? dataLog[k][4].toString().trim() : "";
      var lojaLog = emailParaLoja[emailLog] || "";

      if (dFim >= hojeInicio && dFim <= hojeFim) {
        var skillDoH = (dataLog[k].length > 7 && dataLog[k][7] != null && dataLog[k][7].toString().trim() !== "")
          ? dataLog[k][7].toString().toLowerCase().trim() : "";
        var skillDoLog = skillDoH !== "" ? skillDoH : (skillDoCaso[idCasoLog] || "");
        // Skill em H do log quando gravada; senão Distribuição. Após limpeza de concluídos o ID pode deixar de mapear
        // e skillDoLog fica vazio — incluir nos totais do dia (trat./concl.) para alinhar com o log.
        var passaFiltroLog = !filtroAtivo || skillDoLog === "" || (skillsFiltroLower.indexOf(skillDoLog) !== -1);

        var diff = 0;
        if (!isNaN(dInicio.getTime()) && !isNaN(dFim.getTime())) {
          diff = Math.floor((dFim.getTime() - dInicio.getTime()) / 1000);
        }

        if (passaFiltroLog) {
          if (!statsAgentes[emailLog]) {
            statsAgentes[emailLog] = { tratadas: 0, concluidas: 0, tempo: 0 };
            memoriaUnicosAgente[emailLog] = {};
          }
          if (!memoriaUnicosAgente[emailLog][idCasoLog]) {
            statsAgentes[emailLog].tratadas++;
            memoriaUnicosAgente[emailLog][idCasoLog] = true;
          }
          if (estFinal === "Concluído" && idCasoLog !== "") {
            if (!memoriaConcluidosAgente[emailLog]) memoriaConcluidosAgente[emailLog] = {};
            if (!memoriaConcluidosAgente[emailLog][idCasoLog]) {
              statsAgentes[emailLog].concluidas++;
              memoriaConcluidosAgente[emailLog][idCasoLog] = true;
            }
          }
          if (diff > 0) statsAgentes[emailLog].tempo += diff;

          if (!memoriaUnicosGlobal[idCasoLog]) {
            globalTratadas++;
            memoriaUnicosGlobal[idCasoLog] = true;
          }
          if (estFinal === "Concluído" && idCasoLog !== "" && !memoriaGlobalConcluidos[idCasoLog]) {
            globalConcluidas++;
            memoriaGlobalConcluidos[idCasoLog] = true;
          }
          if (diff > 0) globalTempo += diff;
        }
      }

      if (estFinal === "Concluído" && lojaLog !== "") {
          if (dFim >= inicioMes) {
              if (!unicosMesLoja[lojaLog]) unicosMesLoja[lojaLog] = {};
              if (!concluidasMesLoja[lojaLog]) concluidasMesLoja[lojaLog] = 0;

              if (!unicosMesLoja[lojaLog][idCasoLog]) {
                  concluidasMesLoja[lojaLog]++;
                  unicosMesLoja[lojaLog][idCasoLog] = true;
              }

              if (dFim >= hojeInicio && dFim <= hojeFim) {
                  if (!unicosHojeLoja[lojaLog]) unicosHojeLoja[lojaLog] = {};
                  if (!concluidasHojeLoja[lojaLog]) concluidasHojeLoja[lojaLog] = 0;

                  if (!unicosHojeLoja[lojaLog][idCasoLog]) {
                      concluidasHojeLoja[lojaLog]++;
                      unicosHojeLoja[lojaLog][idCasoLog] = true;
                  }
              }
          }
      }
    }

    var globalTmtSegundos = globalTratadas > 0 ? Math.floor(globalTempo / globalTratadas) : 0;
    var mG = Math.floor(globalTmtSegundos / 60).toString().padStart(2, '0');
    var sG = (globalTmtSegundos % 60).toString().padStart(2, '0');
    var globalTmtFormatado = mG + ":" + sG;

    // Concluídas do mês vindas do agregado (logs arquivados / Looker) — alinha com obterObjetivosLoja
    var concluidasArquivoLoja = {};
    try {
      var dadosHistSup = _obterDadosHistoricoAgregado();
      if (dadosHistSup && dadosHistSup.mensal && dadosHistSup.mensal[anoMesCache]) {
        var histMesSup = dadosHistSup.mensal[anoMesCache];
        for (var kh in histMesSup) {
          if (!histMesSup.hasOwnProperty(kh)) continue;
          var ljHS = histMesSup[kh].loja ? histMesSup[kh].loja.toString().toLowerCase().trim() : "";
          if (ljHS === "") continue;
          var addH = parseInt(histMesSup[kh].concluidas, 10) || 0;
          concluidasArquivoLoja[ljHS] = (concluidasArquivoLoja[ljHS] || 0) + addH;
        }
      }
    } catch (eHistSup) {}

    var _lojaMapGetSup_ = function(mapa, nome) {
      if (!mapa || !nome) return 0;
      if (Object.prototype.hasOwnProperty.call(mapa, nome)) return mapa[nome];
      var t = nome.toString().toLowerCase().trim();
      for (var kk in mapa) {
        if (!Object.prototype.hasOwnProperty.call(mapa, kk) || !kk) continue;
        if (kk.toString().toLowerCase().trim() === t) return mapa[kk];
      }
      return 0;
    };

    // ------------------------------------------------------------------------
    // 4. MAPA DE OBJETIVOS DAS LOJAS (log ativo + histórico + arquivo intradia)
    // ------------------------------------------------------------------------
    var mapaObjetivosLojas = {};
    var diasUteisRestantes = calcularDiasUteis(agora, fimMes);
    if (diasUteisRestantes < 1) diasUteisRestantes = 1;

    for (var o = 1; o < dataObj.length; o++) {
      var mObj = dataObj[o][0] instanceof Date ? Utilities.formatDate(dataObj[o][0], timezone, "MM/yyyy") : dataObj[o][0];
      if (mObj === mesAtualStr) {
        var ljNome = dataObj[o][1] ? dataObj[o][1].toString().trim() : "";
        var valMensal = parseInt(dataObj[o][2]) || 0;
        
        if (ljNome !== "") {
           var ljKey = ljNome.toLowerCase().trim();
           var cMesLog = _lojaMapGetSup_(concluidasMesLoja, ljNome);
           var cMesHist = concluidasArquivoLoja[ljKey] || 0;
           // Evita dupla contagem: o log ativo já reflete o dia corrente.
           var cMes = cMesLog + cMesHist;
           var cHoje = _lojaMapGetSup_(concluidasHojeLoja, ljNome);

           var objDiario = Math.ceil((valMensal - cMes) / diasUteisRestantes);
           if (objDiario < 0 || valMensal === 0 || cMes >= valMensal) {
             objDiario = 0;
           }

           mapaObjetivosLojas[ljNome] = { 
               mensal: valMensal, 
               hoje: cHoje, 
               diarioCalculado: objDiario 
           };
        }
      }
    }

    // ------------------------------------------------------------------------
    // 5. FILA E MAPEAMENTO DOS CASOS ATIVOS
    // ------------------------------------------------------------------------
    var fila = { 
        livres: 0, emTratamento: 0, suspensos: 0, carteira: 0, outro: 0,
        atrasadosLivres: 0, atrasadosTrabalho: 0, atrasadosFechados: 0,
        rqsUltrapassadasLivres: 0, rqsUltrapassadasTrabalho: 0, rqsUltrapassadasFechados: 0,
        rqsHojeLivres: 0, rqsHojeTrabalho: 0, rqsHojeFechados: 0,
        tratadasDia: globalTratadas, concluidasDia: globalConcluidas, tmtGlobal: globalTmtFormatado,
        listaAtrasados: [], listaRqsUltrapassadas: [], listaRqsHoje: [], listaLivres: [], listaTodos: [], listaOutro: []
    };

    var casosAtivosPorOperador = {};

    for (var j = 1; j < dataD.length; j++) {
      var est = dataD[j][9] ? dataD[j][9].toString().trim() : "";
      var responsavel = dataD[j][10] ? dataD[j][10].toString().toLowerCase().trim() : "";
      
      if (est === "Em Tratamento" && responsavel !== "") {
          var tsInicioCaso = (dataD[j][11] instanceof Date) ? dataD[j][11].getTime() : agora.getTime();
          casosAtivosPorOperador[responsavel] = { idCaso: dataD[j][1].toString(), tsInicio: tsInicioCaso };
      }

      var skD = dataD[j][14] ? dataD[j][14].toString().toLowerCase().trim() : "";
      var passaFiltroD = !filtroAtivo || (skillsFiltroLower.indexOf(skD) !== -1);
      if (!passaFiltroD) continue;

      var grupoEstado = "";
      if (est === "Livre" || est === "") grupoEstado = "livres";
      else if (est === "Concluído" || est === "Cancelado") grupoEstado = "fechados";
      else grupoEstado = "trabalho"; 

      var dRqs = (dataD[j][6] instanceof Date) ? dataD[j][6] : (dataD[j][6] ? new Date(dataD[j][6]) : null);
      if (dRqs && isNaN(dRqs.getTime())) dRqs = null;
      var dCria = (dataD[j][5] instanceof Date) ? dataD[j][5] : (dataD[j][5] ? new Date(dataD[j][5]) : null);
      if (dCria && isNaN(dCria.getTime())) dCria = null;
      var dAgend = (dataD[j][12] instanceof Date) ? dataD[j][12] : (dataD[j][12] ? new Date(dataD[j][12]) : null);
      if (dAgend && isNaN(dAgend.getTime())) dAgend = null;
      var intercalar = dataD[j][7] ? dataD[j][7].toString().trim() : "";
      
      var agendIso = dAgend ? Utilities.formatDate(dAgend, timezone, "yyyy-MM-dd'T'HH:mm") : "";
      var obsCompleta = dataD[j][8] ? dataD[j][8].toString().trim() : "";
      var obsTruncada = obsCompleta.length > 30 ? obsCompleta.substring(0, 30) + "..." : obsCompleta;
      
      var casoDet = {
          id: dataD[j][1] ? dataD[j][1].toString().trim() : "-",
          linhaPlanilha: j + 1,
          skill: skD.toUpperCase() || "-",
          criacao: dCria ? Utilities.formatDate(dCria, timezone, "dd/MM/yyyy") : "-", 
          rqs: dRqs ? Utilities.formatDate(dRqs, timezone, "dd/MM/yyyy") : "-", 
          agendIso: agendIso,
          estado: est === "" ? "Livre" : est,
          resp: responsavel.split('@')[0] || "-",
          respEmail: responsavel,
          obsCompleta: obsCompleta,
          obsTruncada: obsTruncada,
          intercalar: intercalar,
          prioridade: dataD[j][15] ? dataD[j][15].toString().toUpperCase().trim() : ""
      };

      if (grupoEstado !== "fechados") { fila.carteira++; fila.listaTodos.push(casoDet); }
      if (est === "Outro") { fila.outro++; fila.listaOutro.push(casoDet); }

      if (grupoEstado === "livres") { fila.livres++; fila.listaLivres.push(casoDet); }
      else if (est === "Em Tratamento") fila.emTratamento++;
      else if (est === "Suspenso") fila.suspensos++;

      if (dCria && !isNaN(dCria.getTime())) {
        var dCriaZero = new Date(dCria.getTime());
        dCriaZero.setHours(0,0,0,0); 
        if (dCriaZero <= dataLimiteSLA) {
            if (grupoEstado === "livres") { fila.atrasadosLivres++; fila.listaAtrasados.push(casoDet); }
            else if (grupoEstado === "trabalho") { fila.atrasadosTrabalho++; fila.listaAtrasados.push(casoDet); }
            else if (grupoEstado === "fechados") fila.atrasadosFechados++;
        }
      }
      
      if (dRqs && !isNaN(dRqs.getTime())) {
        if (dRqs < hojeInicio && intercalar === "") {
            if (grupoEstado === "livres") { fila.rqsUltrapassadasLivres++; fila.listaRqsUltrapassadas.push(casoDet); }
            else if (grupoEstado === "trabalho") { fila.rqsUltrapassadasTrabalho++; fila.listaRqsUltrapassadas.push(casoDet); }
            else if (grupoEstado === "fechados") fila.rqsUltrapassadasFechados++;
        }
        if (dRqs <= hojeFim && intercalar === "") {
            if (grupoEstado === "livres") { fila.rqsHojeLivres++; fila.listaRqsHoje.push(casoDet); }
            else if (grupoEstado === "trabalho") { fila.rqsHojeTrabalho++; fila.listaRqsHoje.push(casoDet); }
            else if (grupoEstado === "fechados") fila.rqsHojeFechados++;
        }
      }
    }

    // ------------------------------------------------------------------------
    // 6. EQUIPA E ESTADOS ATUAIS
    // ------------------------------------------------------------------------
    var equipa = [];
    var quatroHorasMs = 4 * 60 * 60 * 1000; 
    var tresMinutosMs = 15 * 60 * 1000; 
    var hojeStr = Utilities.formatDate(agora, timezone, "yyyy-MM-dd");

    for (var i = 1; i < dataU.length; i++) {
      var emailVal = dataU[i][0] ? dataU[i][0].toString().toLowerCase().trim() : "";
      if (emailVal === "") continue;

      var perfilAgente = dataU[i][3] ? dataU[i][3].toString().toLowerCase() : "";
      var isAdminPuro = perfilAgente.includes("admin") && !perfilAgente.includes("supervisor") && !perfilAgente.includes("coordenador");
      if (isAdminPuro) continue; 

      var estadoVal = dataU[i][6] || "Offline";
      var dataRef = dataU[i][7];
      var timestamp = (dataRef instanceof Date) ? dataRef.getTime() : agora.getTime();
      
      var dataPing = dataU[i][10]; 
      var timestampPing = agora.getTime();
      
      if (dataPing instanceof Date) {
          timestampPing = dataPing.getTime();
      } else if (typeof dataPing === "string" && dataPing.trim() !== "") {
          timestampPing = new Date(dataPing.replace(/'/g, "")).getTime();
      } else if (!isNaN(Number(dataPing)) && dataPing.toString().trim() !== "") {
          timestampPing = Number(dataPing); 
      }

      if (estadoVal !== "Offline") {
        var motivoOffline = null;
        if ((agora.getTime() - timestampPing) >= tresMinutosMs) motivoOffline = "Offline (Janela Fechada)";
        else if ((agora.getTime() - timestamp) >= quatroHorasMs) motivoOffline = "Offline (Timeout 4H)";

        if (motivoOffline) {
          estadoVal = "Offline";
          sheetU.getRange(i + 1, 7).setValue("Offline"); 
          sheetU.getRange(i + 1, 8).setValue(agora);     
          registarLogEstado(emailVal, motivoOffline, agora);
          timestamp = agora.getTime();
          suspenderCasosEmTratamentoServidor(emailVal, undefined, dataU);
        }
      }

      var sAg = statsAgentes[emailVal] || { tratadas: 0, concluidas: 0, tempo: 0 };
      var tmtAgSegundos = sAg.tratadas > 0 ? Math.floor(sAg.tempo / sAg.tratadas) : 0;
      var m = Math.floor(tmtAgSegundos / 60).toString().padStart(2, '0');
      var s = (tmtAgSegundos % 60).toString().padStart(2, '0');

      var dataRegObj = dataU[i][8]; 
      var dataObjStr = (dataRegObj instanceof Date) ? Utilities.formatDate(dataRegObj, timezone, "yyyy-MM-dd") : "";
      
      var objetivoDia = (dataObjStr === hojeStr) ? (parseInt(dataU[i][9]) || 0) : 0;
      var percentagem = objetivoDia > 0 ? Math.round((sAg.concluidas / objetivoDia) * 100) : 0;
      var isSuper = perfilAgente.includes("supervisor") || perfilAgente.includes("coordenador");

      var casoAtivo = casosAtivosPorOperador[emailVal] || null;

      var nomeCompleto = dataU[i][1] ? dataU[i][1].toString().trim() : "Agente";
      var partesNome = nomeCompleto.split(" ");
      var nomeVisivel = partesNome.length > 1 ? partesNome[0] + " " + partesNome[partesNome.length - 1] : partesNome[0];

      equipa.push({
        email: emailVal,
        nome: nomeVisivel,
        loja: dataU[i][2] ? dataU[i][2].toString().trim() : "",
        equipaOp: dataU[i][4] ? dataU[i][4].toString().trim() : "",
        estado: estadoVal,
        horaMudanca: timestamp,
        tratadas: sAg.tratadas,
        concluidas: sAg.concluidas,
        objetivo: objetivoDia,
        percObjetivo: percentagem,
        tmtFormatado: m + ":" + s,
        tmtSegundos: tmtAgSegundos,
        isSuper: isSuper,
        casoAtivoId: casoAtivo ? casoAtivo.idCaso : null,   
        casoAtivoTs: casoAtivo ? casoAtivo.tsInicio : null  
      });
    }

    equipa.sort(function(a, b) {
      if (a.estado === "Disponível" && b.estado !== "Disponível") return -1;
      if (b.estado === "Disponível" && a.estado !== "Disponível") return 1;
      if (a.estado === "Offline" && b.estado !== "Offline") return 1;
      if (b.estado === "Offline" && a.estado !== "Offline") return -1;
      return a.horaMudanca - b.horaMudanca;
    });

    var arrayParaRanking = equipa.filter(function(ag) { return !ag.isSuper; }); 
    arrayParaRanking.sort(function(a, b) { 
      if (b.concluidas !== a.concluidas) return b.concluidas - a.concluidas;
      if (b.tratadas !== a.tratadas) return b.tratadas - a.tratadas;
      return a.tmtSegundos - b.tmtSegundos;
    });
    
    var rankingMapa = {};
    arrayParaRanking.forEach(function(ag, index) { rankingMapa[ag.email] = index + 1; });
    
    equipa.forEach(function(ag) {
        if(!ag.isSuper && ag.concluidas > 0) {
            ag.rankingDiario = rankingMapa[ag.email];
        } else {
            ag.rankingDiario = null;
        }
    });

    var finalResult = { 
      sucesso: true, 
      equipa: equipa, 
      fila: fila, 
      skillsMaster: skillsMaster, 
      mapaObjetivosLojas: mapaObjetivosLojas 
    };

    // Alojamos em Chunk Cache por apenas 25 Segundos para bypass! 
    // Garante que no próximo pedido do frontend (aos 30s) os dados já são recarregados da folha em tempo real
    _putCacheChunked(cacheKey, JSON.stringify(finalResult), 25);

    return finalResult;
  } catch(e) {
    return { sucesso: false, erroSilencioso: true, mensagem: "Erro no Servidor: " + e.message }; 
  } finally {
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("obterDadosSupervisao", new Date().getTime() - startTsHealth);
  }
}

// --- FUNÇÃO AUXILIAR: OBJETIVOS DA LOJA ---
// --- FUNÇÃO AUXILIAR: OBJETIVOS DA LOJA (COM HISTÓRICO INTEGRADO E BLINDAGEM DE TEXTO) ---
// matrizUtilizadores / matrizLog: opcionais (ex.: já lidos em obterEstatisticasOperador); null/omitido = lê da folha.
// dadosHistoricoOpt: se a 4ª posição for passada (arguments.length >= 4), usa esse valor em vez de _obterDadosHistoricoAgregado() (pode ser null).
function obterObjetivosLoja(lojaAlvo, matrizUtilizadores, matrizLog, dadosHistoricoOpt) {
  var timezone = Session.getScriptTimeZone();
  var agora = new Date();
  var mesAtualStr = Utilities.formatDate(agora, timezone, "MM/yyyy");
  var anoMesCache = Utilities.formatDate(agora, timezone, "yyyyMM");
  var objMensal = 0;
  var ss = getSpreadsheet();
  var sheetObj = ss.getSheetByName('Objetivos_Mensais');
  
  if (sheetObj) {
    var dataObj = sheetObj.getDataRange().getValues();
    for(var i=1; i<dataObj.length; i++) {
      var m = dataObj[i][0] ? dataObj[i][0].toString().trim() : "";
      if(dataObj[i][0] instanceof Date) m = Utilities.formatDate(dataObj[i][0], timezone, "MM/yyyy");
      var l = dataObj[i][1] ? dataObj[i][1].toString().trim() : "";
      // Procura a meta da Loja
      if (m === mesAtualStr && l.toLowerCase() === lojaAlvo.toLowerCase().trim()) {
        objMensal = parseInt(dataObj[i][2]) || 0;
        break;
      }
    }
  }

  // 1. BUSCAR CONCLUÍDAS DO ARQUIVO (CACHE 6H) COM COMPARAÇÃO BLINDADA
  var concluidasArquivo = 0;
  try {
      var dadosArquivo = (arguments.length >= 4) ? dadosHistoricoOpt : _obterDadosHistoricoAgregado();
      if (dadosArquivo && dadosArquivo.mensal && dadosArquivo.mensal[anoMesCache]) {
          var histMes = dadosArquivo.mensal[anoMesCache];
          for (var k in histMes) {
              var lojaHist = histMes[k].loja ? histMes[k].loja.toString().toLowerCase().trim() : "";
              var lojaAlvoLimpa = lojaAlvo.toString().toLowerCase().trim();
              
              if (lojaHist === lojaAlvoLimpa) {
                  concluidasArquivo += (parseInt(histMes[k].concluidas) || 0);
              }
          }
      }
  } catch(e) {}

  // 2. BUSCAR CONCLUÍDAS ATIVAS (LOG DE HOJE)
  var inicioMes = new Date(); inicioMes.setDate(1); inicioMes.setHours(0,0,0,0);
  var fimMes = new Date(inicioMes.getFullYear(), inicioMes.getMonth() + 1, 0);
  var hojeInicio = new Date(); hojeInicio.setHours(0,0,0,0);

  var concluidasMesAtivas = 0;
  var concluidasHoje = 0;
  var lojaAlvoLimpa = lojaAlvo.toString().toLowerCase().trim();
  var resumoLog = (arguments.length >= 5 && arguments[4]) ? arguments[4] : null;
  if (resumoLog && resumoLog.mensal && resumoLog.hoje) {
    concluidasMesAtivas = resumoLog.mensal[lojaAlvoLimpa] || 0;
    concluidasHoje = resumoLog.hoje[lojaAlvoLimpa] || 0;
  } else {
    var emailParaLoja = {};
    var dataU;
    if (matrizUtilizadores != null && matrizUtilizadores.length > 0) {
      dataU = matrizUtilizadores;
    } else {
      var sheetU = ss.getSheetByName('Utilizadores');
      dataU = sheetU ? sheetU.getRange(1, 1, sheetU.getLastRow(), 3).getValues() : [];
    }
    for (var u = 1; u < dataU.length; u++) {
      var mail = dataU[u][0] ? dataU[u][0].toString().toLowerCase().trim() : "";
      var lj = dataU[u][2] ? dataU[u][2].toString().toLowerCase().trim() : "";
      if (mail) emailParaLoja[mail] = lj;
    }

    var dataLog;
    if (matrizLog != null && matrizLog.length > 0) {
      dataLog = matrizLog;
    } else {
      var sheetLogRead = ss.getSheetByName(ABA_LOG);
      dataLog = sheetLogRead ? sheetLogRead.getDataRange().getValues() : [];
    }

    if (dataLog && dataLog.length > 1) {
      var unicosMes = {};
      var unicosHoje = {};
      for (var k=1; k<dataLog.length; k++) {
        var idCaso = dataLog[k][0] ? dataLog[k][0].toString().trim() : "";
        var emailLog = dataLog[k][1] ? dataLog[k][1].toString().toLowerCase().trim() : "";
        var dFim = new Date(dataLog[k][3]);
        var estFinal = dataLog[k][4] ? dataLog[k][4].toString().trim() : "";

        if (estFinal === "Concluído" && emailParaLoja[emailLog] === lojaAlvoLimpa) {
          if (dFim >= inicioMes) {
            if (!unicosMes[idCaso]) { concluidasMesAtivas++; unicosMes[idCaso] = true; }
            if (dFim >= hojeInicio && !unicosHoje[idCaso]) { concluidasHoje++; unicosHoje[idCaso] = true; }
          }
        }
      }
    }
  }

  // Evita dupla contagem no painel do colaborador:
  // o Log_Performance ativo já contempla as concluídas do dia.

  // 3. CONSOLIDAR E CALCULAR OBJETIVO DIÁRIO REAL
  var totalConcluidasMes = concluidasArquivo + concluidasMesAtivas;

  var diasUteisRestantes = calcularDiasUteis(agora, fimMes);
  if (diasUteisRestantes < 1) diasUteisRestantes = 1;
  
  // Impede objetivos residuais (o teu 17) quando a meta já foi atingida no geral!
  var objetivoDiario = Math.ceil((objMensal - totalConcluidasMes) / diasUteisRestantes);
  if (objetivoDiario < 0 || objMensal === 0 || totalConcluidasMes >= objMensal) {
      objetivoDiario = 0;
  }

  return { 
      mensal: objMensal, 
      concluidasMes: totalConcluidasMes, 
      concluidasHoje: concluidasHoje, 
      objetivoDiario: objetivoDiario 
  };
}

    var casosTratadosUnicosHoje = {};

                  var chaveLojaHoje = lojaEmail + "|" + idCasoLog;
                  if (!unicosLojaHoje[chaveLojaHoje]) {
                    lojasConcluidasHoje[lojaEmail] = (lojasConcluidasHoje[lojaEmail] || 0) + 1;
                    unicosLojaHoje[chaveLojaHoje] = true;
                  }
                }
              }
        }

        if (dataFim >= hoje) {
          if (estFinal === "Concluído" && emailLog !== "") {
              // Filtro Anti-Batota do Ranking: Conta apenas 1 vez por caso por pessoa
              var chaveRank = emailLog + "_" + idCasoLog;
              if (!unicosParaRanking[chaveRank]) {
                  rankingTodosHoje[emailLog] = (rankingTodosHoje[emailLog] || 0) + 1;
                  unicosParaRanking[chaveRank] = true;
              }
          }
          if (emailLog === email) {
            // Conta as Tratadas Únicas
            if (!casosTratadosUnicosHoje[idCasoLog]) { 
                totalTratadasHoje++; 
                casosTratadosUnicosHoje[idCasoLog] = true; 
            }
            
            // Conta as Concluídas Únicas do Operador (Alinha com os objetivos da Loja!)
            if (estFinal === "Concluído") {
                if (!casosConcluidosUnicosHoje[idCasoLog]) {
                    totalConcluidasHoje++;
                    casosConcluidosUnicosHoje[idCasoLog] = true;
                }
            } 
            
            var dInicio = new Date(dataLogMatriz[i][2]);
            if (!isNaN(dInicio.getTime()) && !isNaN(dataFim.getTime())) {
              var diffS = Math.floor((dataFim.getTime() - dInicio.getTime()) / 1000);
              if (diffS > 0) tempoHoje += diffS;
            }
          }
      }
    }
  }

  sqDebugPerfMaybeMarkStats_('stats_log_perf');

  // O Log_Performance ativo já contém os registos do dia (incluindo após arquivo intraday),
  // logo somar contadores fragmentados aqui duplica KPIs/ranking do operador.

  sqDebugPerfMaybeMarkStats_('stats_contadores_ranking');

  // TMT Diário do Operador
  var tmtSegundos = totalTratadasHoje > 0 ? Math.floor(tempoHoje / totalTratadasHoje) : 0;
  var m = Math.floor(tmtSegundos / 60).toString().padStart(2, '0');
  var s = (tmtSegundos % 60).toString().padStart(2, '0');

  // Ranking Diário
  var arrRanking = [];
  for (var k in rankingTodosHoje) { arrRanking.push({email: k, concluidas: rankingTodosHoje[k]}); }
    arrRanking.sort(function(a,b){ return b.concluidas - a.concluidas; });
    var minhaPosicao = 0;
    for(var r=0; r < arrRanking.length; r++){ if(arrRanking[r].email === email) { minhaPosicao = r+1; break; } }
    var totalOperadoresHoje = arrRanking.length > 0 ? arrRanking.length : 1;
    if(minhaPosicao === 0 && totalTratadasHoje > 0) { minhaPosicao = totalOperadoresHoje + 1; totalOperadoresHoje++; }
    var strRanking = minhaPosicao > 0 ? ("🏆 " + minhaPosicao + "º Lugar de " + totalOperadoresHoje) : "-";

    // Log de Estados / Pausas
    var sheetEst = ss.getSheetByName('Log_Estados');
    var temposEstados = {};
    if (sheetEst) {
      var dataEst = _lerLogEstadosDesde_(sheetEst, hoje);
      var ultimoEstado = null;
      var ultimaHora = null;

      for (var j = 0; j < dataEst.length; j++) {
        var eEst = dataEst[j][0] ? dataEst[j][0].toString().toLowerCase().trim() : "";
        if (eEst !== email) continue;
        var dRegisto = new Date(dataEst[j][2]);
        
        if (ultimoEstado && ultimaHora) {
           var inicioCalc = ultimaHora < hoje ? hoje : ultimaHora;
           if (dRegisto > hoje) {
               var diferenca = Math.floor((dRegisto.getTime() - inicioCalc.getTime()) / 1000);
               if (diferenca > 0) {
                   if (!temposEstados[ultimoEstado]) temposEstados[ultimoEstado] = 0;
                   temposEstados[ultimoEstado] += diferenca;
               }
           }
        }
        ultimoEstado = dataEst[j][1] ? dataEst[j][1].toString().trim() : "";
        ultimaHora = dRegisto;
      }
      if (ultimoEstado && ultimaHora && ultimoEstado !== "Offline") {
        var inicioCalc = ultimaHora < hoje ? hoje : ultimaHora;
        var diffAgora = Math.floor((agora.getTime() - inicioCalc.getTime()) / 1000);
        if (diffAgora > 0) {
          if (!temposEstados[ultimoEstado]) temposEstados[ultimoEstado] = 0;
          temposEstados[ultimoEstado] += diffAgora;
        }
      }
    }

    sqDebugPerfMaybeMarkStats_('stats_log_estados');

    var estadosFormatados = [];
    for (var key in temposEstados) {
      if (temposEstados[key] > 0 && !key.includes("Forçado") && !key.includes("Timeout") && key !== "Offline") {
        var hs = Math.floor(temposEstados[key] / 3600);
        var ms = Math.floor((temposEstados[key] % 3600) / 60).toString().padStart(2, '0');
        var ssStr = (temposEstados[key] % 60).toString().padStart(2, '0');
        var formato = (hs > 0 ? hs + "h " : "") + ms + "m " + ssStr + "s";
        estadosFormatados.push({ estado: key, tempoVisivel: formato });
      }
    }

    var objLoja = obterObjetivosLoja(lojaOp, dataU, dataLogMatriz, dadosHist, {
      mensal: lojasConcluidasMes,
      hoje: lojasConcluidasHoje
    });

    sqDebugPerfMaybeMarkStats_('stats_objetivos_loja');

    var payloadStats = { 
      sucesso: true, 
      tratadas: totalTratadasHoje, 
      concluidas: totalConcluidasHoje, // <--- Agora é impossível ser maior que o Ponto de Atendimento!
      tmt: m + ":" + s, 
      rankingDiario: strRanking, 
      objetivo: objLoja.objetivoDiario, 
      percObjetivo: objLoja.objetivoDiario > 0 ? Math.round((objLoja.concluidasHoje / objLoja.objetivoDiario) * 100) : 100, 
      estados: estadosFormatados,
      estadoAtualServidor: estadoAtualServidor,
      lojaNome: lojaOp, 
      lojaDiario: objLoja.objetivoDiario, 
      lojaHoje: objLoja.concluidasHoje, 
      lojaMensal: objLoja.mensal, 
      lojaConcluidasMes: objLoja.concluidasMes
    };
    _putCacheChunked(keyStats, JSON.stringify(payloadStats), 5);
    return payloadStats;
  } catch(e) { 
    return { sucesso: false, erroSilencioso: true, mensagem: e.message }; 
  } finally {
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("obterEstatisticasOperador", new Date().getTime() - startTsHealth);
  }
}

function forcarEstadoOperador(emailAlvo, novoEstado) {
  try {
    var emailSuper = Session.getActiveUser().getEmail().toLowerCase().trim();
    if (!emailSuper || emailSuper === "") return { sucesso: false, mensagem: "Sessão inválida." };

    var ss = getSpreadsheet();
    var sheetU = ss.getSheetByName('Utilizadores');
    var dataU = sheetU.getDataRange().getValues();
    
    // 1. Validar se quem está a pedir é mesmo Supervisor/Admin
    var isSuper = false;
    for (var i = 1; i < dataU.length; i++) {
      if (dataU[i][0] && dataU[i][0].toString().toLowerCase().trim() === emailSuper) {
         var perfil = dataU[i][3] ? dataU[i][3].toString().toLowerCase() : "";
         if (perfil.includes("admin") || perfil.includes("supervisor") || perfil.includes("coordenador")) {
            isSuper = true;
         }
         break;
      }
    }
    
    if (!isSuper) return { sucesso: false, mensagem: "Não tens permissão para alterar estados." };
    
    // 2. Procurar o alvo, mudar o estado e registar no log
    var agora = new Date();
    for (var j = 1; j < dataU.length; j++) {
      if (dataU[j][0] && dataU[j][0].toString().toLowerCase().trim() === emailAlvo.toLowerCase().trim()) {
        
        sheetU.getRange(j + 1, 7).setValue(novoEstado);
        sheetU.getRange(j + 1, 8).setValue(agora);
        
        // Regista no log que foi o Supervisor a mudar
        var sheetLogEst = ss.getSheetByName(ABA_LOG_ESTADOS);
        // Regista a ação do Supervisor e fecha a duração anterior
        registarLogEstado(emailAlvo, novoEstado + " (Forçado por SPV)", agora);

        // 🛡️ SALVAGUARDA: Suspende o caso SEMPRE que o SPV força um estado (Gera Interrupção Flash e Fecha o TMT)
        suspenderCasosEmTratamentoServidor(emailAlvo, "Forçado a " + novoEstado + " por SPV", dataU);

        return { sucesso: true, mensagem: "Estado " + novoEstado + " aplicado a " + emailAlvo + "." };
      }
    }
    return { sucesso: false, mensagem: "Operador alvo não encontrado." };
  } catch(e) {
    return { sucesso: false, mensagem: e.message };
  }
}

// --- FUNÇÃO AUXILIAR: REGISTAR LOG DE ESTADO COM DURAÇÃO ---
// Cursor em ScriptProperties + appendRow; nunca getDataRange na folha. Caminho rápido: 1× getRange(A:D) na linha do cursor; fallback = últimas LOG_ESTADO_LEGACY_SCAN_MAX linhas (A:D).
function registarLogEstado(email, novoEstado, dataHora) {
  try {
    var ss = getSpreadsheet();
    var sheetLogEst = ss.getSheetByName(ABA_LOG_ESTADOS);
    if (!sheetLogEst) return;

    var props = PropertiesService.getScriptProperties();
    var key = _logEstadoCursorKey_(email);
    var raw = props.getProperty(key);
    var cursor = null;
    if (raw) {
      try {
        cursor = JSON.parse(raw);
      } catch (eParse) {
        cursor = null;
      }
    }

    var em = email.toString().toLowerCase().trim();
    var fechouDuracao = false;
    var lastRowSheet = sheetLogEst.getLastRow();

    if (cursor && cursor.linha && cursor.linha > 0) {
      var lr = parseInt(cursor.linha, 10);
      if (!isNaN(lr) && lr > 0 && lr <= lastRowSheet) {
        // Cursor aponta para uma única linha anterior; ler apenas 1x4 evita custo exponencial.
        var row4 = sheetLogEst.getRange(lr, 1, 1, 4).getValues()[0];
        var cellEmail = row4[0];
        if (cellEmail && cellEmail.toString().toLowerCase().trim() === em) {
          var dataInicio = new Date(row4[2]);
          if (isNaN(dataInicio.getTime())) dataInicio = new Date(cursor.inicioMs);
          if (!isNaN(dataInicio.getTime())) {
            var diffSegundos = Math.floor((dataHora.getTime() - dataInicio.getTime()) / 1000);
            if (diffSegundos >= 0) {
              sheetLogEst.getRange(lr, 4).setValue(_formatarDuracaoLogEstado_(diffSegundos));
              fechouDuracao = true;
            }
          }
        }
      }
    }

    if (!fechouDuracao && cursor && cursor.linha && cursor.linha > 0 && lastRowSheet >= 2) {
      // Fast fallback local: em concorrência alta o índice pode deslocar alguns registos.
      // Procurar numa janela curta perto do cursor antes de cair no scan legacy grande.
      var lrBase = parseInt(cursor.linha, 10);
      if (!isNaN(lrBase)) {
        var ini = Math.max(2, lrBase - 30);
        var fim = Math.min(lastRowSheet, lrBase + 10);
        if (fim >= ini) {
          var blocoLocal = sheetLogEst.getRange(ini, 1, fim - ini + 1, 4).getValues();
          for (var li = blocoLocal.length - 1; li >= 0; li--) {
            var emLoc = blocoLocal[li][0] ? blocoLocal[li][0].toString().toLowerCase().trim() : "";
            if (emLoc !== em) continue;
            var durLoc = blocoLocal[li][3] ? blocoLocal[li][3].toString().trim() : "";
            if (durLoc !== "") continue;
            var dIniLoc = new Date(blocoLocal[li][2]);
            if (isNaN(dIniLoc.getTime())) continue;
            var diffLoc = Math.floor((dataHora.getTime() - dIniLoc.getTime()) / 1000);
            if (diffLoc < 0) continue;
            sheetLogEst.getRange(ini + li, 4).setValue(_formatarDuracaoLogEstado_(diffLoc));
            fechouDuracao = true;
            break;
          }
        }
      }
    }

    if (!fechouDuracao) {
      _registarLogEstadoFecharLinhaLegacy_(sheetLogEst, email, dataHora, lastRowSheet);
    }

    sheetLogEst.appendRow([email, novoEstado, dataHora, ""]);
    // Em cenários concorrentes, lastRowSheet+1 pode divergir; usa a linha real pós-append.
    var nLinha = sheetLogEst.getLastRow();
    props.setProperty(key, JSON.stringify({ linha: nLinha, inicioMs: dataHora.getTime() }));
  } catch (e) {
    Logger.log("Erro ao registar log de estado: " + e.message);
  }
}

      for (var i = 1; i < dataLog.length; i++) {
        var dFim = new Date(dataLog[i][3]);
        if (dFim >= hojeInicio && dFim <= hojeFim) {
          var idCaso = dataLog[i][0] ? dataLog[i][0].toString().trim() : "N/A";
          var emailLog = dataLog[i][1] ? dataLog[i][1].toString().toLowerCase().trim() : "";
          var dInicioLog = new Date(dataLog[i][2]);
          var estFinal = dataLog[i][4] ? dataLog[i][4].toString().trim() : "";
          
          var duracaoFormatada = "00:00"; var diffS = 0;
          if (!isNaN(dInicioLog.getTime()) && !isNaN(dFim.getTime())) {
            diffS = Math.floor((dFim.getTime() - dInicioLog.getTime()) / 1000);
            if (diffS < 0) diffS = 0;
            var h = Math.floor(diffS / 3600);
            var m = Math.floor((diffS % 3600) / 60).toString().padStart(2, '0');
            var s = (diffS % 60).toString().padStart(2, '0');
            duracaoFormatada = (h > 0 ? h + ":" : "") + m + ":" + s;
          }

          // NOVO: Vai "pescar" os dados extra ao dicionário infoCasos
          var extraDataDist = infoCasos[idCaso] ? infoCasos[idCaso].dataDist : "-";
          var extraSkill = infoCasos[idCaso] ? infoCasos[idCaso].skill : "-";

          if (infoUsers[emailLog]) {
            infoUsers[emailLog].teveAtividade = true;
            linhasParaExportar.push([
              strData, strAnoMes, numMes, numDia, // A, B, C, D
              emailLog, infoUsers[emailLog].nome, // E, F
              infoUsers[emailLog].userColab, infoUsers[emailLog].loja, infoUsers[emailLog].equipa, // G (User), H (Loja), I (Equipa) -> Ajustado à tua ordem
              infoUsers[emailLog].objetivo, infoUsers[emailLog].pendentes, // J, K
              idCaso, estFinal, duracaoFormatada, diffS, // L, M, N, O
              extraDataDist, extraSkill // P, Q (NOVAS COLUNAS)
            ]);
          }
        }
      }
    }

    // 4. Garantir que quem não tratou casos vai para o relatório se tiver pendentes/objetivo
    for (var key in infoUsers) {
      var uInfo = infoUsers[key];
      if (!uInfo.teveAtividade && (uInfo.pendentes > 0 || uInfo.objetivo > 0)) {
        linhasParaExportar.push([
          strData, strAnoMes, numMes, numDia, // A, B, C, D
          key, uInfo.nome, // E, F
          uInfo.userColab, uInfo.loja, uInfo.equipa, // G, H, I
          uInfo.objetivo, uInfo.pendentes, // J, K
          "Sem Casos Tratados", "-", "00:00", 0, // L, M, N, O
          "-", "-" // P, Q (NOVAS COLUNAS)
        ]);
      }
    }

    // 5. Escrever com Escudo Anti-Duplicação
    if (linhasParaExportar.length > 0) {
      var ssHistorico = SpreadsheetApp.openById(ID_HISTORICO);
      var sheetHist = ssHistorico.getSheetByName(NOME_ABA_HISTORICO);
      if (!sheetHist) return "Erro: A aba não existe.";
      
      var ultimaLinhaHist = sheetHist.getLastRow();
      if (ultimaLinhaHist > 1) {
        var datasHist = sheetHist.getRange(1, 1, ultimaLinhaHist, 1).getValues();
        var linhasApagar = 0; var primeiraLinhaApagar = -1;
        for (var r = datasHist.length - 1; r >= 1; r--) {
          var dataVal = datasHist[r][0];
          var dataFormatada = (dataVal instanceof Date) ? Utilities.formatDate(dataVal, timezone, "dd/MM/yyyy") : dataVal.toString();
          if (dataFormatada === strData) {
            linhasApagar++; primeiraLinhaApagar = r + 1;
          } else if (linhasApagar > 0) break;
        }
        if (linhasApagar > 0 && primeiraLinhaApagar !== -1) sheetHist.deleteRows(primeiraLinhaApagar, linhasApagar);
      }
      ultimaLinhaHist = sheetHist.getLastRow();
      
      // INSERE AGORA AS 17 COLUNAS!
      sheetHist.getRange(ultimaLinhaHist + 1, 1, linhasParaExportar.length, 17).setValues(linhasParaExportar);
    }
    
    return "Exportação Concluída com 17 colunas (Data Dist. e Skill incluídas).";
  } catch(e) { return "Erro no Fecho do Dia: " + e.message; }
}

function _cntChaveOperador_(emailNorm) {
  return CNT_PREFIX_OPERADOR + (emailNorm || '').toLowerCase().trim();
}

function _cntChaveLoja_(nomeLoja) {
  var raw = (nomeLoja || '').toString().trim();
  if (raw === '') return '';
  var slug = raw.replace(/\s+/g, '_').replace(/[^\w\-]/g, '_');
  if ((CNT_PREFIX_LOJA + slug).length > 220) {
    slug = Utilities.base64EncodeWebSafe(Utilities.newBlob(raw).getBytes());
  }
  return CNT_PREFIX_LOJA + slug;
}

function _cntParseMetric_(raw) {
  if (!raw || raw === '') return { t: 0, c: 0, s: 0 };
  try {
    var o = JSON.parse(raw);
    return {
      t: parseInt(o.t, 10) || 0,
      c: parseInt(o.c, 10) || 0,
      s: parseInt(o.s, 10) || 0
    };
  } catch (e) {
    return { t: 0, c: 0, s: 0 };
  }
}

/**
 * Se a data de referência mudou, apaga todas as chaves cnt_op_* e cnt_loja_*.
 * Chamado no início das escritas (arquivo) e leituras (dashboard).
 */
function _cntGarantirDiaReferencia_(props) {
  var tz = Session.getScriptTimeZone();
  var hojeStr = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd');
  var ref = props.getProperty(CNT_DATA_REFERENCIA);
  if (ref === hojeStr) return;

  var all = props.getProperties();
  for (var k in all) {
    if (k.indexOf(CNT_PREFIX_OPERADOR) === 0 || k.indexOf(CNT_PREFIX_LOJA) === 0) {
      props.deleteProperty(k);
    }
  }
  props.setProperty(CNT_DATA_REFERENCIA, hojeStr);
}

function _cntSomarMetricaChave_(props, chave, deltaT, deltaC, deltaS) {
  if (!chave || chave === '') return;
  var cur = _cntParseMetric_(props.getProperty(chave));
  cur.t += deltaT;
  cur.c += deltaC;
  cur.s += deltaS;
  props.setProperty(chave, JSON.stringify({ t: cur.t, c: cur.c, s: cur.s }));
}

/**
 * Incrementa contadores fragmentados a partir de agregados já calculados no batch.
 * aggOp: { "email@x.pt": { t: nTratadasUnicas, c: nConcluidasUnicas, s: segundos } }
 * aggLoja: { "Nome Loja": { t, c, s } }
 */
function _cntAplicarAgregadosArquivamento_(props, aggOp, aggLoja) {
  for (var mail in aggOp) {
    if (!aggOp.hasOwnProperty(mail)) continue;
    var v = aggOp[mail];
    if (!v) continue;
    _cntSomarMetricaChave_(props, _cntChaveOperador_(mail), v.t || 0, v.c || 0, v.s || 0);
  }
  for (var lj in aggLoja) {
    if (!aggLoja.hasOwnProperty(lj)) continue;
    var w = aggLoja[lj];
    if (!w || lj === '') continue;
    var chL = _cntChaveLoja_(lj);
    if (chL === '') continue;
    _cntSomarMetricaChave_(props, chL, w.t || 0, w.c || 0, w.s || 0);
  }
}

/**
 * Corpo do arquivamento: assume ScriptLock + cnt_data já alinhados.
 * Transação: Ler ativo → Validar → Append Distribuição no arquivo → Properties (fragmentado) → Limpar ativo.
 * Nunca arquiva linhas com estado "Em Tratamento".
 * Log_Performance: NÃO é movido para Log_Performance_Arq aqui (fica na app até limpezaNoturnaLogs),
 * para o executarFechoDoDia ler o log no mesmo dia. Contadores (cnt_*) usam as linhas de log elegíveis.
 */
function _executarCorpoArquivamentoIntraday_(props) {
  var ss = getSpreadsheet();
  var tz = Session.getScriptTimeZone();
  var agora = new Date();
  var hojeInicio = new Date(agora.getTime());
  hojeInicio.setHours(0, 0, 0, 0);
  var hojeFim = new Date(agora.getTime());
  hojeFim.setHours(23, 59, 59, 999);

  var sheetDist = ss.getSheetByName(ABA_DISTRIBUICAO);
  if (!sheetDist) return { sucesso: false, mensagem: 'Aba de Distribuição não encontrada.' };

  var dataDist = sheetDist.getDataRange().getValues();
  if (dataDist.length <= 1) {
    return { sucesso: true, mensagem: 'Nada a arquivar.', linhasDistribuicao: 0, linhasLog: 0 };
  }

  var cabecalhos = dataDist[0];
  var linhasParaManter = [cabecalhos];
  var linhasArquivoDist = [];
  var idsArquivar = {};

  for (var i = 1; i < dataDist.length; i++) {
    var estado = dataDist[i][9] ? dataDist[i][9].toString().trim() : '';
    if (estado === 'Em Tratamento') {
      linhasParaManter.push(dataDist[i]);
      continue;
    }
    if (estado === 'Concluído' || estado === 'Cancelado') {
      var idC = dataDist[i][1] ? dataDist[i][1].toString().trim() : '';
      if (idC !== '') idsArquivar[idC] = true;
      linhasArquivoDist.push(dataDist[i]);
    } else {
      linhasParaManter.push(dataDist[i]);
    }
  }

  var nDist = linhasArquivoDist.length;
  if (nDist === 0) {
    return { sucesso: true, mensagem: 'Não havia casos Concluído/Cancelado para arquivar.', linhasDistribuicao: 0, linhasLog: 0 };
  }
  var idsArquivadosLog = Object.keys(idsArquivar).slice(0, 50).join(",");

  var sheetU = ss.getSheetByName('Utilizadores');
  var emailParaLoja = {};
  if (sheetU) {
    var dataU = sheetU.getDataRange().getValues();
    for (var u = 1; u < dataU.length; u++) {
      var em = dataU[u][0] ? dataU[u][0].toString().toLowerCase().trim() : '';
      if (em === '') continue;
      emailParaLoja[em] = dataU[u][2] ? dataU[u][2].toString().trim() : '';
    }
  }

  var sheetLog = ss.getSheetByName(ABA_LOG);
  var aggOp = {};
  var aggLoja = {};
  var nLogParaContadores = 0;

  if (sheetLog) {
    var dataLog = sheetLog.getDataRange().getValues();

    var memT = {};
    var memC = {};

    for (var k = 1; k < dataLog.length; k++) {
      var idL = dataLog[k][0] ? dataLog[k][0].toString().trim() : '';
      var dFim = new Date(dataLog[k][3]);
      var incluir = (idL !== '' && idsArquivar[idL] && !isNaN(dFim.getTime()) && dFim >= hojeInicio && dFim <= hojeFim);

      if (!incluir) continue;

      nLogParaContadores++;

      var emailLog = dataLog[k][1] ? dataLog[k][1].toString().toLowerCase().trim() : '';
      var estFinal = dataLog[k][4] ? dataLog[k][4].toString().trim() : '';
      var dIni = new Date(dataLog[k][2]);
      var diffS = 0;
      if (!isNaN(dIni.getTime()) && !isNaN(dFim.getTime())) {
        diffS = Math.floor((dFim.getTime() - dIni.getTime()) / 1000);
        if (diffS < 0) diffS = 0;
      }

      if (emailLog === '') continue;

      if (!aggOp[emailLog]) aggOp[emailLog] = { t: 0, c: 0, s: 0 };
      if (!memT[emailLog]) memT[emailLog] = {};
      if (!memT[emailLog][idL]) {
        memT[emailLog][idL] = true;
        aggOp[emailLog].t += 1;
      }
      if (estFinal === 'Concluído') {
        if (!memC[emailLog]) memC[emailLog] = {};
        if (!memC[emailLog][idL]) {
          memC[emailLog][idL] = true;
          aggOp[emailLog].c += 1;
        }
      }
      aggOp[emailLog].s += diffS;

      var lj = emailParaLoja[emailLog] || '';
      if (lj !== '') {
        if (!aggLoja[lj]) aggLoja[lj] = { t: 0, c: 0, s: 0, _tIds: {}, _cIds: {} };
        if (!aggLoja[lj]._tIds[idL]) {
          aggLoja[lj]._tIds[idL] = true;
          aggLoja[lj].t += 1;
        }
        if (estFinal === 'Concluído' && !aggLoja[lj]._cIds[idL]) {
          aggLoja[lj]._cIds[idL] = true;
          aggLoja[lj].c += 1;
        }
        aggLoja[lj].s += diffS;
      }
    }

    for (var lj2 in aggLoja) {
      if (aggLoja.hasOwnProperty(lj2)) {
        delete aggLoja[lj2]._tIds;
        delete aggLoja[lj2]._cIds;
      }
    }
  }

  for (var iv = 0; iv < linhasArquivoDist.length; iv++) {
    var ev = linhasArquivoDist[iv][9] ? linhasArquivoDist[iv][9].toString().trim() : '';
    if (ev === 'Em Tratamento' || (ev !== 'Concluído' && ev !== 'Cancelado')) {
      return {
        sucesso: false,
        mensagem: 'Validação de arquivo: linha inválida (apenas Concluído/Cancelado; nunca Em Tratamento).',
        linhasDistribuicao: 0,
        linhasLog: 0
      };
    }
  }

  var ssHist = _arqAbrirFicheiroArquivo_();

  var shDistArq = _arqGarantirFolhaDistribuicaoArq_(ssHist, cabecalhos);
  var stamp = Utilities.formatDate(agora, tz, "yyyy-MM-dd'T'HH:mm:ss'Z'");
  var numColsCab = cabecalhos.length;
  var blocoDist = [];
  for (var a = 0; a < linhasArquivoDist.length; a++) {
    var r = linhasArquivoDist[a].slice();
    while (r.length < numColsCab) r.push('');
    if (r.length > numColsCab) r = r.slice(0, numColsCab);
    r.push(stamp);
    blocoDist.push(r);
  }
  if (blocoDist.length > 0) {
    var lr = shDistArq.getLastRow();
    shDistArq.getRange(lr + 1, 1, blocoDist.length, blocoDist[0].length).setValues(blocoDist);
  }

  _cntAplicarAgregadosArquivamento_(props, aggOp, aggLoja);

  sheetDist.getRange(2, 1, sheetDist.getMaxRows() - 1, sheetDist.getMaxColumns()).clearContent();
  sheetDist.getRange(1, 1, linhasParaManter.length, linhasParaManter[0].length).setValues(linhasParaManter);

  SpreadsheetApp.flush();
  logOrigemEscrita_("_executarCorpoArquivamentoIntraday_", "CLEAR_REWRITE_DISTRIBUICAO", idsArquivadosLog, nDist, "Arquivamento intraday de concluídos/cancelados");

  return {
    sucesso: true,
    mensagem: 'Arquivo intraday concluído. Distribuição: ' + nDist + '. Log_Performance mantido na app (fecho do dia); contadores com ' + nLogParaContadores + ' linha(s) de log de hoje.',
    linhasDistribuicao: nDist,
    linhasLog: 0
  };
}

/**
 * MÓDULO 1 — Double-lock: ScriptLock + chave curta em ScriptProperties (arq_batch_lock).
 * Respeita exclusão explícita de "Em Tratamento".
 */
function executarArquivamentoIntradaySeguro(opt) {
  var startTsHealth = new Date().getTime();
  var lock = LockService.getScriptLock();
  var props = PropertiesService.getScriptProperties();
  try {
    lock.waitLock(120000);

    _cntGarantirDiaReferencia_(props);

    var now = Date.now();
    var rawPropLock = props.getProperty(ARQ_PROP_LOCK);
    if (rawPropLock) {
      try {
        var ol = JSON.parse(rawPropLock);
        if (ol.until && ol.until > now) {
          return { sucesso: false, mensagem: 'Outro arquivamento está em execução (lock de propriedades).', skip: true };
        }
      } catch (eIgn) {}
    }
    props.setProperty(ARQ_PROP_LOCK, JSON.stringify({ until: now + 300000 }));

    var res = _executarCorpoArquivamentoIntraday_(props);
    props.deleteProperty(ARQ_PROP_LOCK);
    return res;
  } catch (e) {
    try {
      props.deleteProperty(ARQ_PROP_LOCK);
    } catch (e2) {}
    return { sucesso: false, mensagem: 'Erro no arquivamento: ' + e.message };
  } finally {
    try {
      lock.releaseLock();
    } catch (e3) {}
    if (typeof logAcaoHealthCheck === 'function') {
      logAcaoHealthCheck('executarArquivamentoIntradaySeguro', new Date().getTime() - startTsHealth);
    }
  }
}

/**
 * Uma única passagem por ScriptProperties: valida dia, getProperties, mapa cnt_op_* e métricas do email.
 */
function _contadoresArquivadosHoje_obterPacote(email) {
  var props = PropertiesService.getScriptProperties();
  _cntGarantirDiaReferencia_(props);
  var all = props.getProperties();
  var mapaArchRank = {};
  for (var k in all) {
    if (all.hasOwnProperty(k) && k.indexOf(CNT_PREFIX_OPERADOR) === 0) {
      var emk = k.substring(CNT_PREFIX_OPERADOR.length);
      mapaArchRank[emk] = _cntParseMetric_(all[k]);
    }
  }
  var em = email ? email.toString().toLowerCase().trim() : '';
  var archOp = em === '' ? { t: 0, c: 0, s: 0 } : (mapaArchRank[em] || { t: 0, c: 0, s: 0 });
  return { archOp: archOp, mapaArchRank: mapaArchRank };
}

function contadoresArquivadosHoje_lerOperador(email) {
  var props = PropertiesService.getScriptProperties();
  _cntGarantirDiaReferencia_(props);
  if (!email) return { t: 0, c: 0, s: 0 };
  var em = email.toString().toLowerCase().trim();
  return _cntParseMetric_(props.getProperty(_cntChaveOperador_(em)));
}

function contadoresArquivadosHoje_lerLoja(nomeLoja) {
  var props = PropertiesService.getScriptProperties();
  _cntGarantirDiaReferencia_(props);
  var ch = _cntChaveLoja_(nomeLoja);
  if (ch === '') return { t: 0, c: 0, s: 0 };
  return _cntParseMetric_(props.getProperty(ch));
}

/**
 * Mapa email (minúsculas) -> { t, c, s } para todos os operadores com chave no dia.
 */
function contadoresArquivadosHoje_mapaOperadores() {
  var props = PropertiesService.getScriptProperties();
  _cntGarantirDiaReferencia_(props);
  var all = props.getProperties();
  var out = {};
  for (var k in all) {
    if (all.hasOwnProperty(k) && k.indexOf(CNT_PREFIX_OPERADOR) === 0) {
      var em = k.substring(CNT_PREFIX_OPERADOR.length);
      out[em] = _cntParseMetric_(all[k]);
    }
  }
  return out;
}

/**
 * Mapa chave interna de loja (slug/base64 usado na chave, sem prefixo) -> { t, c, s }.
 * Para UI humana, preferir contadoresArquivadosHoje_lerLoja(nomeReal).
 */
function contadoresArquivadosHoje_mapaLojasPorChave() {
  var props = PropertiesService.getScriptProperties();
  _cntGarantirDiaReferencia_(props);
  var all = props.getProperties();
  var out = {};
  for (var k in all) {
    if (all.hasOwnProperty(k) && k.indexOf(CNT_PREFIX_LOJA) === 0) {
      var idLoja = k.substring(CNT_PREFIX_LOJA.length);
      out[idLoja] = _cntParseMetric_(all[k]);
    }
  }
  return out;
}

/**
 * Agrega deltas de arquivo de hoje para um conjunto de emails (ex.: equipa da supervisão).
 * emailsLista: array de emails; devolve soma { t, c, s }.
 */
function contadoresArquivadosHoje_somarParaListaEmails(emailsLista) {
  var mapa = contadoresArquivadosHoje_mapaOperadores();
  var sum = { t: 0, c: 0, s: 0 };
  if (!emailsLista || emailsLista.length === 0) return sum;
  for (var i = 0; i < emailsLista.length; i++) {
    var em = (emailsLista[i] || '').toString().toLowerCase().trim();
    if (em === '' || !mapa[em]) continue;
    sum.t += mapa[em].t;
    sum.c += mapa[em].c;
    sum.s += mapa[em].s;
  }
  return sum;
}

// --- FUNÇÃO AUXILIAR: CALCULAR DIAS ÚTEIS (EXCLUI FINS DE SEMANA E FERIADOS) ---
function calcularDiasUteis(dataInicio, dataFim) {
  var count = 0;
  var dataAtual = new Date(dataInicio.getTime());
  dataAtual.setHours(0,0,0,0);
  var dataLimite = new Date(dataFim.getTime());
  dataLimite.setHours(23,59,59,999);

  // 1. LISTA DE FERIADOS FIXOS (Mês-Dia) - Repetem-se todos os anos
  var feriadosFixos = [
    "01-01", // Ano Novo
    "04-25", // Dia da Liberdade
    "05-01", // Dia do Trabalhador
    "06-10", // Dia de Portugal
    "08-15", // Assunção de Nossa Senhora
    "10-05", // Implantação da República
    "11-01", // Dia de Todos os Santos
    "12-01", // Restauração da Independência
    "12-08", // Imaculada Conceição
    "12-25"  // Natal
  ];

  // 2. LISTA DE FERIADOS MÓVEIS (Ano-Mês-Dia) - Específicos deste ano (2026)
  // Podes adicionar aqui também o vosso Feriado Municipal, se aplicável à operação.
  var feriadosMoveis = [
    "2026-02-17", // Carnaval (Se a Randstad der tolerância de ponto)
    "2026-04-03", // Sexta-feira Santa
    "2026-06-04"  // Corpo de Deus
  ];

  while (dataAtual <= dataLimite) {
    var diaSemana = dataAtual.getDay();
    
    // Verifica se não é fim de semana (0 = Domingo, 6 = Sábado)
    if (diaSemana !== 0 && diaSemana !== 6) {
      
      // Formatar a data atual para verificar nas listas
      var mes = (dataAtual.getMonth() + 1).toString().padStart(2, '0');
      var dia = dataAtual.getDate().toString().padStart(2, '0');
      var ano = dataAtual.getFullYear().toString();
      
      var formatoFixo = mes + "-" + dia;
      var formatoMovel = ano + "-" + mes + "-" + dia;
      
      // Se NÃO estiver na lista dos fixos E NÃO estiver na lista dos móveis, então é Dia Útil!
      if (feriadosFixos.indexOf(formatoFixo) === -1 && feriadosMoveis.indexOf(formatoMovel) === -1) {
        count++;
      }
    }
    
    // Avança para o dia seguinte
    dataAtual.setDate(dataAtual.getDate() + 1);
  }
  
  return count;
}

// --- ARQUIVO E LIMPEZA NOTURNA DE LOGS (HOT VS COLD STORAGE) ---
function limpezaNoturnaLogs() {
  var startTsHealth = new Date().getTime();
  try {
    // 1. FORÇAR DESCARGA DA CACHE (RAM) PARA A FOLHA ANTES DE ARQUIVAR

// --- CÉREBRO DO SLA (Usa a tua lógica de Feriados Fixos/Móveis) ---
function calcularDataLimiteSLA() {
  var dataAtual = new Date();
  dataAtual.setHours(0,0,0,0); // Tranca na meia-noite de hoje

  // 1. LISTA DE FERIADOS FIXOS (Mês-Dia)
  var feriadosFixos = [
    "01-01", "04-25", "05-01", "06-10", "08-15", 
    "10-05", "11-01", "12-01", "12-08", "12-25"
  ];

  // 2. LISTA DE FERIADOS MÓVEIS (Ano-Mês-Dia)
  var feriadosMoveis = [
    "2026-02-17", // Carnaval
    "2026-04-03", // Sexta-feira Santa
    "2026-06-04"  // Corpo de Deus
  ];

  var diasUteisARecuar = 4; // 1 (dia útil anterior) + 3 (dias de SLA)

  // O Motor de Viagem no Tempo
  while (diasUteisARecuar > 0) {
    dataAtual.setDate(dataAtual.getDate() - 1); // Recua 1 dia no calendário
    
    var diaSemana = dataAtual.getDay();
    
    // Se NÃO for fim de semana...
    if (diaSemana !== 0 && diaSemana !== 6) {
      
      var mes = (dataAtual.getMonth() + 1).toString().padStart(2, '0');
      var dia = dataAtual.getDate().toString().padStart(2, '0');
      var ano = dataAtual.getFullYear().toString();
      
      var formatoFixo = mes + "-" + dia;
      var formatoMovel = ano + "-" + mes + "-" + dia;
      
      // E NÃO for feriado... desconta um dia útil!
      if (feriadosFixos.indexOf(formatoFixo) === -1 && feriadosMoveis.indexOf(formatoMovel) === -1) {
        diasUteisARecuar--;
      }
    }
  }
  
  // Devolve a data exata (ex: se hoje for 25/03, devolve 19/03)
  return dataAtual; 
}

// Processa a Adição ou Remoção em Massa
function atualizarSkillsEmMassa(emailsAlvo, skillsAlvo, acao) {
  try {
    if (emailsAlvo.length === 0 || skillsAlvo.length === 0) return { sucesso: false, mensagem: "Seleciona pelo menos um operador e uma skill." };

    var ss = getSpreadsheet();
    var sheetU = ss.getSheetByName('Utilizadores');
    var dataU = sheetU.getDataRange().getValues();
    var linhasAlteradas = 0;

    for (var i = 1; i < dataU.length; i++) {
      var mail = dataU[i][0] ? dataU[i][0].toString().trim() : "";
      
      if (emailsAlvo.includes(mail)) {
         var skillsAtuaisStr = dataU[i][12] ? dataU[i][12].toString().trim() : "";
         var listaSkills = skillsAtuaisStr === "" ? [] : skillsAtuaisStr.split(",").map(function(s) { return s.trim(); });

         if (acao === 'adicionar') {
            skillsAlvo.forEach(function(sk) { 
               if (!listaSkills.includes(sk)) listaSkills.push(sk); 
            });
         } else if (acao === 'remover') {
            listaSkills = listaSkills.filter(function(sk) { 
               return !skillsAlvo.includes(sk); 
            });
         }

         // Grava na Coluna M
         sheetU.getRange(i + 1, 13).setValue(listaSkills.join(", "));
         linhasAlteradas++;
      }
    }
    
    return { sucesso: true, mensagem: linhasAlteradas + " operador(es) atualizado(s) com sucesso!" };
  } catch (e) {
    return { sucesso: false, mensagem: "Erro: " + e.message };
  }
}

        if (obsTruncada.length > 40) obsTruncada = obsTruncada.substring(0, 40) + "...";

        var hasIntercalar = (data[i][7] && data[i][7].toString().trim() !== "") ? true : false;
        var isRqsAtrasada = (dRqs && !isNaN(dRqs.getTime()) && dRqs <= limiteHoje) ? true : false;

        pendentes.push({ 
          id: id, 
          rqs: rqsStr, 
          agendamento: agendStr, 
          obsCompleta: obsCompleta, 
          obsTruncada: obsTruncada, 
          estado: est,
          hasIntercalar: hasIntercalar,
          isRqsAtrasada: isRqsAtrasada
        });
      }
    }
    return { sucesso: true, dados: pendentes };
  } catch(e) {
    return { sucesso: false, mensagem: "Erro: " + e.message };
  }
}

var HIST_AGG_CACHE_KEY = "historico_agg_v4";
var HIST_AGG_LOCK_KEY = "historico_agg_lock";
var HIST_COLUNAS_ARQUIVO = 17; // A:Q

// Função para contornar o limite restrito de 100KB da Google no CacheService
function _putCacheChunked(key, str, validadeSegundos) {
  try {
    var cache = CacheService.getScriptCache();
    var expiracao = validadeSegundos || 21600; // Default 6 Horas
    var chunkSize = 90000;
    var chunks = Math.ceil(str.length / chunkSize);
    cache.put(key + "_chunks", chunks.toString(), expiracao); 
    for (var i = 0; i < chunks; i++) {
      cache.put(key + "_" + i, str.substring(i * chunkSize, (i + 1) * chunkSize), expiracao);
    }
  } catch (e) {
    return { sucesso: false, erroSilencioso: true, mensagem: e.message };
  }
}

function _getCacheChunked(key) {
  try {
    var cache = CacheService.getScriptCache();
    var chunksStr = cache.get(key + "_chunks");
    if (!chunksStr) return null;
    var chunks = parseInt(chunksStr);
    var str = "";
    for (var i = 0; i < chunks; i++) {
      var chunk = cache.get(key + "_" + i);
      if (!chunk) return null;
      str += chunk;
    }
    return str;
  } catch (e) {
    return null; // Fallback silencioso para reconstruir cache nativo se falhar
  }
}

// Motor Base que lê do Arquivo Pesado uma vez a cada 2H
function _obterDadosHistoricoAgregado() {
  var cached = _getCacheChunked(HIST_AGG_CACHE_KEY);
  if (cached) return JSON.parse(cached);

  var ID_HISTORICO = '11dDUUgq087rvSl1COrJ8ZbSKwLjEbytTsZiYKAFYZMY';
  var NOME_ABA = 'BD_Historico_SmartQueue';
  var lock = LockService.getScriptLock();
  
  try {
      // Evita "cache stampede" quando vários utilizadores pedem histórico ao mesmo tempo.
      if (!lock.tryLock(10000)) {
        Utilities.sleep(800);
        cached = _getCacheChunked(HIST_AGG_CACHE_KEY);
        return cached ? JSON.parse(cached) : null;
      }

      // Segunda leitura de cache após adquirir lock: outro pedido pode já ter preenchido.
      cached = _getCacheChunked(HIST_AGG_CACHE_KEY);
      if (cached) return JSON.parse(cached);

      var ssHist = SpreadsheetApp.openById(ID_HISTORICO);
      var sheetHist = ssHist.getSheetByName(NOME_ABA);
      if(!sheetHist) return null;
      var lastRow = sheetHist.getLastRow();
      if (lastRow <= 1) {
        var vazio = { mensal: {}, diario: {}, meses: {} };
        _putCacheChunked(HIST_AGG_CACHE_KEY, JSON.stringify(vazio), 21600);
        return vazio;
      }
      var data = sheetHist.getRange(1, 1, lastRow, HIST_COLUNAS_ARQUIVO).getValues();

      var aggMensal = {};
      var aggDiario = {};
      var mesesUnicos = {};
      var memMensal = {}; 
      var memDiario = {};
      var memObjetivo = {}; // NOVO: Impede que o objetivo se some repetidamente

      var timezone = Session.getScriptTimeZone();

      for(var i=1; i<data.length; i++) {
          var diaRaw = data[i][0];
          var dataDiaExato = "";
          if (diaRaw instanceof Date) {
              dataDiaExato = Utilities.formatDate(diaRaw, timezone, "dd/MM/yyyy");
          } else {
              dataDiaExato = diaRaw ? diaRaw.toString().trim() : "";
          }

          var anoMes = data[i][1] ? data[i][1].toString().trim() : "";       
          if(!anoMes) continue;
          
          var mesApresentacao = data[i][2] ? data[i][2].toString().padStart(2, '0') + "/" + anoMes.substring(0,4) : "";
          var email = data[i][4] ? data[i][4].toString().toLowerCase().trim() : "";

          mesesUnicos[anoMes] = mesApresentacao; 

          // Inicializa Dicionários com a variável 'objetivo'
          if(!aggMensal[anoMes]) aggMensal[anoMes] = {};
          if(email !== "" && !aggMensal[anoMes][email]) {
              aggMensal[anoMes][email] = { nome: data[i][5] || "", loja: data[i][7] || "", equipa: data[i][8] || "", tratadas: 0, concluidas: 0, tempo: 0, skills: [], skillsMap: {}, objetivo: 0 };
          }

          if(!aggDiario[dataDiaExato]) aggDiario[dataDiaExato] = { anoMes: anoMes, dados: {} };
          if(email !== "" && !aggDiario[dataDiaExato].dados[email]) {
              aggDiario[dataDiaExato].dados[email] = { nome: data[i][5] || "", loja: data[i][7] || "", equipa: data[i][8] || "", tratadas: 0, concluidas: 0, tempo: 0, skills: [], skillsMap: {}, objetivo: 0 };
          }

          var objetivoDoDia = parseInt(data[i][9]) || 0; // Coluna J (Objetivo da Loja)
          var idCaso = data[i][11] ? data[i][11].toString().trim() : "";
          var estFinal = data[i][12] ? data[i][12].toString().trim() : "";
          var diffS = parseInt(data[i][14]) || 0;
          var skill = data[i][16] ? data[i][16].toString().trim() : "";

          var chkObjetivo = dataDiaExato + "_" + email;
          var chkMensal = anoMes + "_" + email + "_" + idCaso;
          var chkDiario = dataDiaExato + "_" + email + "_" + idCaso;

          if (email !== "") {
              // 1. Acumulação do Objetivo (Corre sempre 1x por dia por Operador)
              if (!memObjetivo[chkObjetivo]) {
                  aggMensal[anoMes][email].objetivo += objetivoDoDia;
                  aggDiario[dataDiaExato].dados[email].objetivo += objetivoDoDia;
                  memObjetivo[chkObjetivo] = true;
              }

              // 2. Acumulação de Produtividade (Só contabiliza se tocou num caso)
              if (idCaso !== "Sem Casos Tratados") {
                  if (!memMensal[chkMensal]) { aggMensal[anoMes][email].tratadas++; memMensal[chkMensal] = true; }
                  if(estFinal === "Concluído") aggMensal[anoMes][email].concluidas++;
                  aggMensal[anoMes][email].tempo += diffS;
                  if(skill && skill !== "-" && !aggMensal[anoMes][email].skillsMap[skill]) {
                    aggMensal[anoMes][email].skillsMap[skill] = true;
                    aggMensal[anoMes][email].skills.push(skill);
                  }

                  if (!memDiario[chkDiario]) { aggDiario[dataDiaExato].dados[email].tratadas++; memDiario[chkDiario] = true; }
                  if(estFinal === "Concluído") aggDiario[dataDiaExato].dados[email].concluidas++;
                  aggDiario[dataDiaExato].dados[email].tempo += diffS;
                  if(skill && skill !== "-" && !aggDiario[dataDiaExato].dados[email].skillsMap[skill]) {
                    aggDiario[dataDiaExato].dados[email].skillsMap[skill] = true;
                    aggDiario[dataDiaExato].dados[email].skills.push(skill);
                  }
              }
          }
      }

      // Remove mapas auxiliares antes de serializar/cache.
      for (var ym in aggMensal) {
        for (var em in aggMensal[ym]) {
          delete aggMensal[ym][em].skillsMap;
        }
      }
      for (var d in aggDiario) {
        for (var ed in aggDiario[d].dados) {
          delete aggDiario[d].dados[ed].skillsMap;
        }
      }

      var result = { mensal: aggMensal, diario: aggDiario, meses: mesesUnicos };
      _putCacheChunked(HIST_AGG_CACHE_KEY, JSON.stringify(result), 21600);
      return result;
  } catch(e) {
      Logger.log("Erro no Histórico: " + e.message);
      return null;
  } finally {
      try { lock.releaseLock(); } catch(_e) {}
  }
}

// Utilitário de administração: cria trigger horário para manter cache quente.
function configurarTriggerPreAquecimentoHistorico() {

  var triggers = ScriptApp.getProjectTriggers();
  var jaExiste = false;

  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === alvo) {
      jaExiste = true;
      break;

  if (!jaExiste) {
    ScriptApp.newTrigger(alvo).timeBased().everyHours(1).create();
  }

  return {
    sucesso: true,
    mensagem: jaExiste ? "Trigger já existia." : "Trigger criado com sucesso.",
    handler: alvo,
    frequencia: "1 hora"
  };
}

            if(ranking[i].email === email) { pos = i+1; break; }
        }
        var tr = ranking.length > 0 ? ranking.length : 1;
        if(pos === 0 && atual.tratadas > 0) { pos = tr + 1; tr++; }

// Motor de Filtragem para a Sala de Controlo (Sem limites de frontend)
function obterHistoricoSupervisaoAsync(filtros) {
     try {
         var dataHist = _obterDadosHistoricoAgregado();
         if(!dataHist) return { sucesso: false, mensagem: "A extrair dados profundos. Tenta novamente em 5s." };

         var mesesAlvo = filtros.mesesAno || [];
         if(mesesAlvo.length === 0) return { sucesso: false, mensagem: "Por favor, seleciona pelo menos um Mês." };

         // --- NOVO: Lê a meta Mensal fixa diretamente da Base de Dados para não distorcer com as flutuações diárias ---
         var mapaObjMensal = {};
         if (filtros.tipoVisao === "mensal") {
             var ss = getSpreadsheet();
             var sheetObj = ss.getSheetByName('Objetivos_Mensais');
             if (sheetObj) {
                 var dataObj = sheetObj.getDataRange().getValues();
                 var tz = Session.getScriptTimeZone();
                 for(var i=1; i<dataObj.length; i++){
                     var m = dataObj[i][0] instanceof Date ? Utilities.formatDate(dataObj[i][0], tz, "MM/yyyy") : (dataObj[i][0] ? dataObj[i][0].toString().trim() : "");
                     var l = dataObj[i][1] ? dataObj[i][1].toString().trim() : "";
                     var v = parseInt(dataObj[i][2]) || 0;
                     if (m && l) mapaObjMensal[m + "_" + l] = v;
                 }
             }
         }

         var resultados = [];

         if (filtros.tipoVisao === "mensal") {
             mesesAlvo.forEach(function(mA) {
                 var targetAgg = dataHist.mensal[mA];
                 var mesLabel = dataHist.meses[mA] || mA; // Ex: "04/2026"
                 
                 if(targetAgg) {
                     for(var mail in targetAgg) {
                         var obj = targetAgg[mail];
                         var tmt = obj.tratadas > 0 ? Math.floor(obj.tempo / obj.tratadas) : 0;
                         var m = Math.floor(tmt / 60).toString().padStart(2, '0');
                         var s = (tmt % 60).toString().padStart(2, '0');
                         
                         // Injeta o objetivo REAL mensal (ex: 150) da loja, ignorando a soma dos diários
                         var objRealMensal = mapaObjMensal[mesLabel + "_" + obj.loja] || 0;

                         resultados.push({
                             periodo: mesLabel, 
                             nome: obj.nome, loja: obj.loja, equipa: obj.equipa,
                             objetivo: objRealMensal, // <--- OBJETIVO MENSAL REAL AQUI
                             tratadas: obj.tratadas, concluidas: obj.concluidas, tmt: m + ":" + s,
                            tempoTotalSeg: obj.tempo || 0,
                            tmtSeg: tmt,
                            skills: obj.skills.join(", ")
                         });
                     }
                 }
             });
         } else {
             // Visão Diária (Mantém a leitura do objetivo diário exato que foi gravado naquele dia)
             for (var diaExato in dataHist.diario) {
                 var rootDia = dataHist.diario[diaExato];
                 if (mesesAlvo.indexOf(rootDia.anoMes) !== -1) {
                     var targetAgg = rootDia.dados;
                     for(var mail in targetAgg) {
                         var obj = targetAgg[mail];
                         var tmt = obj.tratadas > 0 ? Math.floor(obj.tempo / obj.tratadas) : 0;
                         var m = Math.floor(tmt / 60).toString().padStart(2, '0');
                         var s = (tmt % 60).toString().padStart(2, '0');

                         resultados.push({
                             periodo: diaExato, 
                             nome: obj.nome, loja: obj.loja, equipa: obj.equipa,
                             objetivo: obj.objetivo || 0, // <--- OBJETIVO DIÁRIO GRAVADO
                             tratadas: obj.tratadas, concluidas: obj.concluidas, tmt: m + ":" + s,
                            tempoTotalSeg: obj.tempo || 0,
                            tmtSeg: tmt,
                            skills: obj.skills.join(", ")
                         });
                     }
                 }
             }
         }

         resultados.sort(function(a,b){ 
            if (a.periodo !== b.periodo) {
                var pA = a.periodo.split('/').reverse().join('');
                var pB = b.periodo.split('/').reverse().join('');
                return pB.localeCompare(pA);
            }
            if (b.concluidas !== a.concluidas) return b.concluidas - a.concluidas;
            if (b.tratadas !== a.tratadas) return b.tratadas - a.tratadas;
            return (a.tmtSeg || 0) - (b.tmtSeg || 0);
         });

         for (var r = 0; r < resultados.length; r++) {
            delete resultados[r].tmtSeg;
         }
         
         return { sucesso: true, dados: resultados };
     } catch(e) { return { sucesso: false, mensagem: e.message }; }
}

function obterObjetivosParaEdicao(mesStr) {
  try {
    var ss = getSpreadsheet();
    var sheetU = ss.getSheetByName('Utilizadores');
    if (!sheetU) return { sucesso: false, mensagem: "Aba Utilizadores não encontrada." };
    
    var matrizUsers = sheetU.getDataRange().getValues();
    var lojasMap = {};
    for(var i=1; i<matrizUsers.length; i++) {
      var l = matrizUsers[i][2] ? matrizUsers[i][2].toString().trim() : "";
      if(l !== "") lojasMap[l] = 0; // Prepara todas as lojas ativas a zeros
    }
    
    var sheetObj = ss.getSheetByName('Objetivos_Mensais');
    if (sheetObj) {
      var matrizObj = sheetObj.getDataRange().getValues();
      for(var j=1; j<matrizObj.length; j++) {
         var m = matrizObj[j][0] ? matrizObj[j][0].toString().trim() : "";
         if(matrizObj[j][0] instanceof Date) {
             m = Utilities.formatDate(matrizObj[j][0], Session.getScriptTimeZone(), "MM/yyyy");
         }
         if (m === mesStr) {
             var nomeLoja = matrizObj[j][1] ? matrizObj[j][1].toString().trim() : "";
             var valor = parseInt(matrizObj[j][2]) || 0;
             if (nomeLoja !== "") lojasMap[nomeLoja] = valor; 
         }
      }
    }
    
    var listaFinal = [];
    for (var key in lojasMap) {
        listaFinal.push({ loja: key, objetivo: lojasMap[key] });
    }
    listaFinal.sort(function(a,b) { return a.loja.localeCompare(b.loja); });
    
    return { sucesso: true, dados: listaFinal };
  } catch(e) { return { sucesso: false, mensagem: "Erro no servidor: " + e.message }; }
}

function salvarObjetivosEmMassa(mesStr, arrayObjetivos) {
  try {
    var ss = getSpreadsheet();
    var sheetObj = ss.getSheetByName('Objetivos_Mensais');
    if (!sheetObj) {
      sheetObj = ss.insertSheet('Objetivos_Mensais');
      sheetObj.appendRow(["Mês", "Ponto de Atendimento", "Objetivo"]);
    }
    
    var matrizSheet = sheetObj.getDataRange().getValues();
    var timezone = Session.getScriptTimeZone();
    var linhasAlteradas = 0;
    
    for (var k=0; k < arrayObjetivos.length; k++) {
        var lojaAlvo = arrayObjetivos[k].loja;
        var valorAlvo = arrayObjetivos[k].objetivo;
        var encontrou = false;
        
        for(var i=1; i<matrizSheet.length; i++) {
           var m = matrizSheet[i][0] ? matrizSheet[i][0].toString().trim() : "";
           if(matrizSheet[i][0] instanceof Date) m = Utilities.formatDate(matrizSheet[i][0], timezone, "MM/yyyy");
           
           if (m === mesStr && matrizSheet[i][1].toString().trim() === lojaAlvo) {
               sheetObj.getRange(i+1, 3).setValue(valorAlvo);
               encontrou = true;
               linhasAlteradas++;
               break;
           }
        }
        
        if (!encontrou) {
            sheetObj.appendRow([mesStr, lojaAlvo, valorAlvo]);
            // Atualiza a memória para impedir duplicação em loop rápido
            matrizSheet = sheetObj.getDataRange().getValues(); 
            linhasAlteradas++;
        }
    }
    
    return { sucesso: true, mensagem: "Objetivos de " + linhasAlteradas + " lojas gravados com sucesso!" };
  } catch(e) { return { sucesso: false, mensagem: "Erro ao gravar: " + e.message }; }
}

/**
 * Observação da supervisão no drilldown: histórico cumulativo na coluna I (mesmo padrão de prepend que finalizarTarefaServidor).
 * Exige id do caso + número de linha da folha; valida que a coluna B nessa linha coincide (evita gravar em linha errada).
 */
function adicionarObservacaoSupervisaoServidor(idCaso, linhaPlanilha, textoObservacao) {
  var startTsHealth = new Date().getTime();
  var lock = LockService.getScriptLock();
  try {
    var perfOk = _temPerfilSupervisaoSalaControlo_();
    if (!perfOk.ok) {
      return { sucesso: false, mensagem: perfOk.mensagem };
    }

    var idLimpo = idCaso != null ? idCaso.toString().trim() : "";
    if (idLimpo === "") {
      return { sucesso: false, mensagem: "Identificador do caso em falta." };
    }

    var linha = parseInt(linhaPlanilha, 10);
    if (isNaN(linha) || linha < 2) {
      return { sucesso: false, mensagem: "Número de linha da folha inválido. Atualiza a Sala de Controlo e tenta de novo." };
    }

    var texto = textoObservacao != null ? textoObservacao.toString().trim() : "";
    if (texto === "") {
      return { sucesso: false, mensagem: "A observação não pode estar vazia." };
    }
    if (texto.length > 4000) {
      return { sucesso: false, mensagem: "Observação demasiado longa (máx. 4000 caracteres)." };
    }

    lock.waitLock(30000);

    var ss = getSpreadsheet();
    var sheetDist = ss.getSheetByName(ABA_DISTRIBUICAO);
    if (!sheetDist) {
      return { sucesso: false, mensagem: "Folha de distribuição não encontrada." };
    }

    var ultLR = sheetDist.getLastRow();
    if (linha > ultLR) {
      return { sucesso: false, mensagem: "A linha indicada já não existe na folha. Atualiza o drilldown." };
    }

    var idNaFolha = sheetDist.getRange(linha, 2).getValue();
    var idFolhaStr = idNaFolha != null ? idNaFolha.toString().trim() : "";
    if (idFolhaStr !== idLimpo) {
      return {
        sucesso: false,
        mensagem: "O caso foi atualizado: o ID na linha " + linha + " não coincide com o selecionado. Fecha o drilldown e atualiza a Sala de Controlo."
      };
    }

    var obsAntigas = sheetDist.getRange(linha, 9).getValue();
    var obsAntigasStr = obsAntigas != null ? obsAntigas.toString().trim() : "";

    var agora = new Date();
    var carimbo = Utilities.formatDate(agora, Session.getScriptTimeZone(), "dd/MM HH:mm");
    var emailSup = perfOk.email || Session.getActiveUser().getEmail().toLowerCase().trim();
    var userAbreviado = emailSup.split("@")[0];
    var novaEntrada = "[" + carimbo + " - " + userAbreviado + "]\n" + texto;
    var obsFinal = obsAntigasStr === "" ? novaEntrada : novaEntrada + "\n\n" + obsAntigasStr;

    sheetDist.getRange(linha, 9).setValue(obsFinal);
    SpreadsheetApp.flush();

    return { sucesso: true, mensagem: "Observação registada no histórico do caso." };
  } catch (e) {
    return { sucesso: false, mensagem: "Erro ao gravar: " + e.message };
  } finally {
    try {
      lock.releaseLock();
    } catch (eRel) {}
    if (typeof logAcaoHealthCheck === "function") {
      logAcaoHealthCheck("adicionarObservacaoSupervisaoServidor", new Date().getTime() - startTsHealth);
    }
  }
}

function importarNovosCasos() {
  var ui = SpreadsheetApp.getUi();
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  var sheetBase = ss.getSheetByName("Distribuição"); 
  var sheetImport = ss.getSheetByName("Importar"); 
  
  if (!sheetBase || !sheetImport) {
    ui.alert("⚠️ Erro de Sistema", "Não encontrei as abas 'Distribuição' e/ou 'Importar'.", ui.ButtonSet.OK);
    return;
  }
  
  var dadosImport = sheetImport.getDataRange().getValues();
  if (dadosImport.length <= 1) {
    ui.alert("ℹ️ Informação", "A aba 'Importar' está vazia (ou só tem o cabeçalho).", ui.ButtonSet.OK);
    return;
  }
  
  var dadosBase = sheetBase.getDataRange().getValues();
  
  // 1. MAPEAMENTO DE IDs EXISTENTES E HISTÓRICOS (Blindagem Anti-Zombie Cases)
  var idsExistentes = {};
  
  // A) Varre a grelha atual de Distribuição
  for (var i = 1; i < dadosBase.length; i++) {
    var idBase = dadosBase[i][1] ? dadosBase[i][1].toString().trim().toUpperCase() : "";
    if (idBase !== "") {
      idsExistentes[idBase] = true;
    }
  }

  // B) Varre o Log de Performance do dia atual (Previne ressurreição após limpeza manual a meio do dia)
  var sheetLog = ss.getSheetByName('Log_Performance');
  if (sheetLog && sheetLog.getLastRow() > 1) {
    var dadosLog = sheetLog.getRange(2, 1, sheetLog.getLastRow() - 1, 1).getValues(); // Coluna A (Índice 1)
    for (var l = 0; l < dadosLog.length; l++) {
      var idLog = dadosLog[l][0] ? dadosLog[l][0].toString().trim().toUpperCase() : "";
      if (idLog !== "") idsExistentes[idLog] = true;
    }
  }
  
  // 2. FILTRAGEM DOS DADOS NOVOS
  var novosCasos = [];
  var duplicadosApanhados = 0;
  var falhasDatas = [];
  var falhasSkill = [];
  var colunasParaInjetar = sheetBase.getLastColumn();
  
  for (var j = 1; j < dadosImport.length; j++) {
    var linha = dadosImport[j];
    // O ID na aba "Importar" DEVE estar na Coluna B (Índice 1)
    var idNovo = linha[1] ? linha[1].toString().trim().toUpperCase() : ""; 
    
    if (idNovo !== "") {
      if (!idsExistentes[idNovo]) {
        // Uniformiza o comprimento da matriz consoante a aba 'Distribuição' para evitar erros de limite
        var linhaNormalizada = new Array(colunasParaInjetar).fill("");
        for(var c = 0; c < colunasParaInjetar; c++) {
           linhaNormalizada[c] = linha[c] !== undefined ? linha[c] : "";
        }

        // 🛡️ HIGIENIZAÇÃO DE DATAS: Força a conversão das strings coladas para Objetos Data do Google Sheets
        // Coluna F (Índice 5) = Data de Criação | Coluna G (Índice 6) = Data RQS
        var dataCriaLimpa = parseDataSegura(linhaNormalizada[5]);
        var dataRqsLimpa = parseDataSegura(linhaNormalizada[6]);
        if (!dataCriaLimpa || !dataRqsLimpa) {
          var faltas = [];
          if (!dataCriaLimpa) faltas.push("Data criação");
          if (!dataRqsLimpa) faltas.push("Data RQS");
          falhasDatas.push({
            linha: j + 1,
            id: idNovo,
            erros: faltas
          });
          continue;
        }
        var skillLimpa = linhaNormalizada[14] ? linhaNormalizada[14].toString().trim() : "";
        if (skillLimpa === "") {
          falhasSkill.push({
            linha: j + 1,
            id: idNovo,
            erros: ["Skill"]
          });
          continue;
        }
        linhaNormalizada[5] = dataCriaLimpa;
        linhaNormalizada[6] = dataRqsLimpa;
        linhaNormalizada[14] = skillLimpa;
        
        // Garante que o estado e responsável entram limpos por defeito
        linhaNormalizada[9] = linhaNormalizada[9] ? linhaNormalizada[9] : ""; // Coluna J
        linhaNormalizada[10] = linhaNormalizada[10] ? linhaNormalizada[10] : ""; // Coluna K
        
        novosCasos.push(linhaNormalizada);
        idsExistentes[idNovo] = true; // Bloqueia duplicados na própria colagem
      } else {
        duplicadosApanhados++;
      }
    }
  }

  if (falhasDatas.length > 0) {
    var maxShow = Math.min(3, falhasDatas.length);
    var partes = [];
    for (var fx = 0; fx < maxShow; fx++) {
      var f = falhasDatas[fx];
      partes.push("linha " + f.linha + " (ID " + f.id + "): " + f.erros.join(", "));
    }
    var msgDatas = "Importação bloqueada — Data criação e Data RQS são obrigatórias e válidas (" + falhasDatas.length + " caso(s)). " + partes.join(" | ");
    if (falhasDatas.length > maxShow) {
      msgDatas += " (+" + (falhasDatas.length - maxShow) + " outros)";
    }
    logOrigemEscrita_("importarNovosCasos", "BLOQUEIO_VALIDACAO_DATAS", "", falhasDatas.length, msgDatas);
    ui.alert("⚠️ Importação bloqueada", msgDatas, ui.ButtonSet.OK);
    return;
  }
  if (falhasSkill.length > 0) {
    var maxShowSkill = Math.min(3, falhasSkill.length);
    var partesSkill = [];
    for (var sx = 0; sx < maxShowSkill; sx++) {
      var fs = falhasSkill[sx];
      partesSkill.push("linha " + fs.linha + " (ID " + fs.id + "): " + fs.erros.join(", "));
    }
    var msgSkill = "Importação bloqueada — Skill é obrigatória nas linhas novas (" + falhasSkill.length + " caso(s)). " + partesSkill.join(" | ");
    if (falhasSkill.length > maxShowSkill) {
      msgSkill += " (+" + (falhasSkill.length - maxShowSkill) + " outros)";
    }
    logOrigemEscrita_("importarNovosCasos", "BLOQUEIO_VALIDACAO_SKILL", "", falhasSkill.length, msgSkill);
    ui.alert("⚠️ Importação bloqueada", msgSkill, ui.ButtonSet.OK);
    return;
  }
  
  // 3. INJEÇÃO SEGURA NA BASE DE DADOS
  if (novosCasos.length > 0) {
    var ultimaLinhaBase = sheetBase.getLastRow();
    // Injeta tudo num só segundo (Batch Operation para evitar Race Conditions!)
    sheetBase.getRange(ultimaLinhaBase + 1, 1, novosCasos.length, colunasParaInjetar).setValues(novosCasos);
    SpreadsheetApp.flush(); // Tranca os dados na base de forma irreversível
    var idsInseridos = novosCasos.slice(0, 50).map(function(r) { return r[1] || ""; }).filter(String).join(",");
    logOrigemEscrita_("importarNovosCasos", "SETVALUES_BATCH", idsInseridos, novosCasos.length, "Importação via aba Importar");
    
    // Limpa a aba de importação, preservando o cabeçalho (Linha 1)
    sheetImport.getRange(2, 1, sheetImport.getMaxRows() - 1, sheetImport.getMaxColumns()).clearContent();
    
    ui.alert("✅ Importação Concluída com Sucesso!", 
             "Foram importados e adicionados " + novosCasos.length + " novos casos à fila.\n" + 
             "Foram retidos e ignorados " + duplicadosApanhados + " casos duplicados.", 
             ui.ButtonSet.OK);
  } else {
    // Se era tudo lixo ou duplicado, limpa a staging area na mesma para não empatar a operação
    sheetImport.getRange(2, 1, sheetImport.getMaxRows() - 1, sheetImport.getMaxColumns()).clearContent();
    ui.alert("🛡️ Carga Bloqueada (Duplicados)", "Todos os " + duplicadosApanhados + " casos que colaste já existiam na fila de Distribuição. A aba 'Importar' foi limpa.", ui.ButtonSet.OK);
  }
}

function guardarUtilizadorServidor(dados) {
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(15000);
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName('Utilizadores');
    var data = sheet.getDataRange().getValues();
    var linhaEncontrada = -1;
    
    var emailOriginal = (dados.emailOriginal || "").toLowerCase().trim();
    var emailNovo = dados.emailNovo.toLowerCase().trim();

    // 1. Procura se estamos a editar um utilizador existente
    if (emailOriginal !== "") {
      for (var i = 1; i < data.length; i++) {
        if (data[i][0] && data[i][0].toString().toLowerCase().trim() === emailOriginal) {
          linhaEncontrada = i + 1;
          break;
        }
      }
    } else {
       // 2. Se for uma criação, garante que o e-mail não existe já
       for (var j = 1; j < data.length; j++) {
        if (data[j][0] && data[j][0].toString().toLowerCase().trim() === emailNovo) {
          return { sucesso: false, mensagem: "⚠️ Já existe um utilizador com este e-mail!" };
        }
      }
    }

    if (linhaEncontrada > -1) {
      // ATUALIZAR (Altera o e-mail se foi modificado e guarda o resto)
      sheet.getRange(linhaEncontrada, 1).setValue(emailNovo); 
      sheet.getRange(linhaEncontrada, 2).setValue(dados.nome);
      sheet.getRange(linhaEncontrada, 3).setValue(dados.loja);
      sheet.getRange(linhaEncontrada, 4).setValue(dados.perfil);
      sheet.getRange(linhaEncontrada, 5).setValue(dados.equipa);
      return { sucesso: true, mensagem: "✅ Utilizador atualizado com sucesso!" };
    } else {
      // CRIAR NOVO
      var novaLinha = new Array(sheet.getLastColumn()).fill("");
      novaLinha[0] = emailNovo;
      novaLinha[1] = dados.nome;
      novaLinha[2] = dados.loja;
      novaLinha[3] = dados.perfil;
      novaLinha[4] = dados.equipa;
      sheet.appendRow(novaLinha);
      return { sucesso: true, mensagem: "✅ Novo utilizador criado com sucesso!" };
    }
  } catch(e) {
    return { sucesso: false, mensagem: "Erro ao guardar: " + e.message };
  } finally { lock.releaseLock(); }
}

/** Validação obrigatória de negócio (importação grelha): índices alinhados com matriz do frontend (17 colunas). */
function _validarCamposObrigatoriosImportacaoLinha_(linha) {
  var erros = [];
  function g(i) {
    return (linha[i] != null) ? linha[i].toString().trim() : "";
  }
  if (!g(1)) erros.push("ID caso");
  if (!g(2)) erros.push("Canal de entrada");
  if (!g(14)) erros.push("Skill");
  if (!parseDataSegura(linha[5])) erros.push("Data criação");
  if (!parseDataSegura(linha[6])) erros.push("Data RQS");
  return erros;
}

function importarCasosProcessados(dadosGrelha) {
  var startTsHealth = new Date().getTime();
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(45000);

    }
    var dadosBase = sheetBase.getDataRange().getValues();
    var idsExistentes = {};
    for (var i = 1; i < dadosBase.length; i++) {
      var idB = dadosBase[i][1] ? dadosBase[i][1].toString().trim().toUpperCase() : "";
      if (idB !== "") idsExistentes[idB] = true;
    }
    var novosCasos = [];

        if (errosObr.length > 0) {
          falhasCamposObrigatorios.push({ linha: j + 1, id: idN, erros: errosObr });
          continue;
        }
        var linhaNorm = new Array(16).fill("");
        linhaNorm[0]  = linha[0] || "";  // Loja
        linhaNorm[1]  = idN;             // ID CASO
        linhaNorm[2]  = linha[2] || "";  // Canal
        linhaNorm[3]  = linha[3] || "";  // Email
        linhaNorm[4]  = linha[4] || "";  // PN
        linhaNorm[5]  = parseDataSegura(linha[5]); 
        linhaNorm[6]  = parseDataSegura(linha[6]); 
        linhaNorm[7]  = linha[7] || "";  
        linhaNorm[8]  = linha[8] || "";  
        linhaNorm[9]  = linha[9] || "";  
        linhaNorm[10] = linha[10] || ""; 
        linhaNorm[11] = "";              
        linhaNorm[12] = parseDataSegura(linha[12]); 
        linhaNorm[13] = new Date();      
        linhaNorm[14] = linha[14] || ""; 
        linhaNorm[15] = linha[15] || ""; 
        var respMail = linhaNorm[10].toString().toLowerCase().trim();
        if (respMail !== "" && mapaLojas[respMail]) {
          linhaNorm[0] = mapaLojas[respMail];
          if (!linhaNorm[9]) linhaNorm[9] = "Por tratar";
        }
        novosCasos.push(linhaNorm);
        idsExistentes[idN] = true;
      }
    }
    if (falhasCamposObrigatorios.length > 0) {
      var maxShow = Math.min(3, falhasCamposObrigatorios.length);
      var partes = [];
      for (var fx = 0; fx < maxShow; fx++) {
        var f = falhasCamposObrigatorios[fx];
        partes.push("linha " + f.linha + " (ID " + f.id + "): " + f.erros.join(", "));
      }
      var msgBloq = "Importação bloqueada — campos obrigatórios em falta ou inválidos (" + falhasCamposObrigatorios.length + " caso(s)). " + partes.join(" | ");
      if (falhasCamposObrigatorios.length > maxShow) {
        msgBloq += " (+" + (falhasCamposObrigatorios.length - maxShow) + " outros)";
      }
      logOrigemEscrita_("importarCasosProcessados", "BLOQUEIO_VALIDACAO_CAMPOS", "", falhasCamposObrigatorios.length, msgBloq);
      return { sucesso: false, mensagem: msgBloq };
    }
    if (novosCasos.length > 0) {
      sheetBase.getRange(sheetBase.getLastRow() + 1, 1, novosCasos.length, 16).setValues(novosCasos);
      SpreadsheetApp.flush(); 
      var idsInseridos = novosCasos.slice(0, 50).map(function(r) { return r[1] || ""; }).filter(String).join(",");
      logOrigemEscrita_("importarCasosProcessados", "SETVALUES_BATCH", idsInseridos, novosCasos.length, "Importação via grelha frontend");
      return { sucesso: true, mensagem: "✅ " + novosCasos.length + " casos importados!" };
    }
    return { sucesso: false, mensagem: "⚠️ Nenhum caso novo (duplicados)." };
  } catch (e) { 
    return { sucesso: false, mensagem: e.message }; 
  } finally { 
    lock.releaseLock(); 
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("importarCasosProcessados", new Date().getTime() - startTsHealth);
  }
}

/** Normaliza ID ou Contacto Aux para comparação (alinhado com obterTodosIdsDistribuicao). */
function _normalizarIdImportacao_(v) {
  if (v === "" || v === null || v === undefined) return "";
  var s = String(v).replace(/\s+/g, '').toUpperCase().trim();
  if (s.includes('E+') || s.length > 10) {
    var num = Number(String(v).replace(',', '.'));
    if (!isNaN(num)) s = BigInt(Math.round(num)).toString();
  }
  if (s === "" || s === "NULL") return "";
  return s;
}

/** Mapa de IDs existentes na coluna B da Distribuição (lookup O(1)). */
function _obterMapaIdsExistentesDistribuicao_(dadosBase) {
  var ids = {};
  for (var i = 1; i < dadosBase.length; i++) {
    var id = _normalizarIdImportacao_(dadosBase[i][1]);
    if (id !== "") ids[id] = true;
  }
  return ids;
}

/** Abre o spreadsheet Evalyze pelo nome fixo na pasta Drive configurada. */
function _abrirSpreadsheetEvalyze_() {
  var folder = DriveApp.getFolderById(EVALYZE_FOLDER_ID);
  var files = folder.getFilesByName(EVALYZE_FILE_NAME);
  if (!files.hasNext()) {
    throw new Error("Ficheiro '" + EVALYZE_FILE_NAME + "' não encontrado na pasta de origem do Smart Queue.");
  }
  return SpreadsheetApp.openById(files.next().getId());
}

/**
 * Importa casos novos do Relatório Evalyze (aba Lojas) para a Distribuição.
 * - Colunas A:P gravadas; coluna Q (Contacto Aux) só para anti-duplicados.
 * - Campos obrigatórios em falta: ignora a linha e continua com as restantes.
 * - Lógica de normalização alinhada com importarCasosProcessados.
 * @returns {{sucesso:boolean, mensagem:string, importados:number, duplicados:number, ignoradosCampos:number}}
 */
function importarCasosEvalyzeAutomatico() {
  var startTsHealth = new Date().getTime();
  var lock = LockService.getScriptLock();
  var importados = 0;
  var duplicados = 0;
  var ignoradosCampos = 0;
  try {
    lock.waitLock(45000);

    var ssEvalyze = _abrirSpreadsheetEvalyze_();
    var sheetEvalyze = ssEvalyze.getSheetByName(EVALYZE_SHEET_NAME);
    if (!sheetEvalyze) {
      throw new Error("Aba '" + EVALYZE_SHEET_NAME + "' não encontrada no relatório Smart Queue.");
    }

    var lastRowEvalyze = sheetEvalyze.getLastRow();
    if (lastRowEvalyze < 2) {
      return {
        sucesso: true,
        mensagem: "ℹ️ Relatório Smart Queue sem linhas de dados.",
        importados: 0,
        duplicados: 0,
        ignoradosCampos: 0
      };
    }

    var dadosEvalyze = sheetEvalyze.getRange(2, 1, lastRowEvalyze - 1, 17).getValues();
    var ss = getSpreadsheet();
    var sheetBase = ss.getSheetByName(ABA_DISTRIBUICAO);
    if (!sheetBase) throw new Error("Aba de Distribuição não encontrada.");

    var sheetUsers = ss.getSheetByName('Utilizadores');
    var mapaLojas = {};
    if (sheetUsers) {
      var dataU = sheetUsers.getDataRange().getValues();
      for (var u = 1; u < dataU.length; u++) {
        if (dataU[u][0]) mapaLojas[dataU[u][0].toString().toLowerCase().trim()] = dataU[u][2];
      }
    }

    var dadosBase = sheetBase.getDataRange().getValues();
    var idsExistentes = _obterMapaIdsExistentesDistribuicao_(dadosBase);
    var novosCasos = [];
    var dataDistribuicao = new Date();

    for (var j = 0; j < dadosEvalyze.length; j++) {
      var row = dadosEvalyze[j];
      var linha = new Array(17);
      for (var c = 0; c < 17; c++) {
        linha[c] = (row[c] !== undefined && row[c] !== null) ? row[c] : "";
      }

      var idN = _normalizarIdImportacao_(linha[1]);
      if (idN === "") continue;

      var contactoAux = _normalizarIdImportacao_(linha[16]);
      if (idsExistentes[idN] || (contactoAux !== "" && idsExistentes[contactoAux])) {
        duplicados++;
        continue;
      }

      var errosObr = _validarCamposObrigatoriosImportacaoLinha_(linha);
      if (errosObr.length > 0) {
        ignoradosCampos++;
        continue;
      }

      var linhaNorm = new Array(16).fill("");
      linhaNorm[0]  = linha[0] || "";
      linhaNorm[1]  = idN;
      linhaNorm[2]  = linha[2] || "";
      linhaNorm[3]  = linha[3] || "";
      linhaNorm[4]  = linha[4] || "";
      linhaNorm[5]  = parseDataSegura(linha[5]);
      linhaNorm[6]  = parseDataSegura(linha[6]);
      linhaNorm[7]  = linha[7] || "";
      linhaNorm[8]  = linha[8] || "";
      linhaNorm[9]  = linha[9] || "";
      linhaNorm[10] = linha[10] || "";
      linhaNorm[11] = "";
      linhaNorm[12] = parseDataSegura(linha[12]);
      linhaNorm[13] = dataDistribuicao;
      linhaNorm[14] = linha[14] || "";
      linhaNorm[15] = linha[15] || "";

      var respMail = linhaNorm[10].toString().toLowerCase().trim();
      if (respMail !== "" && mapaLojas[respMail]) {
        linhaNorm[0] = mapaLojas[respMail];
        if (!linhaNorm[9]) linhaNorm[9] = "Por tratar";
      }

      novosCasos.push(linhaNorm);
      idsExistentes[idN] = true;
      if (contactoAux !== "") idsExistentes[contactoAux] = true;
    }

    if (novosCasos.length > 0) {
      sheetBase.getRange(sheetBase.getLastRow() + 1, 1, novosCasos.length, 16).setValues(novosCasos);
      SpreadsheetApp.flush();
      importados = novosCasos.length;
      var idsInseridos = novosCasos.slice(0, 50).map(function(r) { return r[1] || ""; }).filter(String).join(",");
      logOrigemEscrita_(
        "importarCasosEvalyzeAutomatico",
        "SETVALUES_BATCH",
        idsInseridos,
        importados,
        "Evalyze Lojas | dup:" + duplicados + " | ignorados:" + ignoradosCampos
      );
    } else if (duplicados > 0 || ignoradosCampos > 0) {
      logOrigemEscrita_(
        "importarCasosEvalyzeAutomatico",
        "SEM_NOVOS",
        "",
        0,
        "dup:" + duplicados + " | ignorados campos:" + ignoradosCampos
      );
    }

    var msg = "✅ " + importados + " importados";
    if (duplicados > 0) msg += ", " + duplicados + " duplicados ignorados";
    if (ignoradosCampos > 0) msg += ", " + ignoradosCampos + " ignorados (campos obrigatórios em falta)";
    if (importados === 0 && duplicados === 0 && ignoradosCampos === 0) {
      msg = "ℹ️ Nenhum caso novo encontrado no relatório Smart Queue.";
    }

    return {
      sucesso: true,
      mensagem: msg,
      importados: importados,
      duplicados: duplicados,
      ignoradosCampos: ignoradosCampos
    };
  } catch (e) {
    logOrigemEscrita_("importarCasosEvalyzeAutomatico", "ERRO", "", 0, e.message);
    return {
      sucesso: false,
      mensagem: "❌ " + e.message,
      importados: importados,
      duplicados: duplicados,
      ignoradosCampos: ignoradosCampos
    };
  } finally {
    try { lock.releaseLock(); } catch (_e) {}
    if (typeof logAcaoHealthCheck === "function") {
      logAcaoHealthCheck("importarCasosEvalyzeAutomatico", new Date().getTime() - startTsHealth);
    }
  }
}

/**
 * Repara datas em falta na Distribuição usando uma extração manual (cruzamento por ID).
 * Segurança:
 * - NUNCA sobrescreve Data Criação (F) nem Data RQS (G) já preenchidas na Distribuição.
 * - Só atualiza quando F/G estão vazias e a extração tem data válida.
 *
 * Aba de extração esperada (default: "Extracao_Datas"):
 * - Col A: ID Caso
 * - Col B: Data Criação
 * - Col C: Data RQS
 *
 * @param {string=} nomeAbaExtracao opcional (default "Extracao_Datas")
 * @returns {{sucesso:boolean, mensagem:string, analisadas:number, atualizadas:number, semMatch:number, semDataValida:number}}
 */
function repararDatasDistribuicaoViaExtracaoManual(nomeAbaExtracao) {
  var startTsHealth = new Date().getTime();
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(45000);
    var ss = getSpreadsheet();
    var sheetDist = ss.getSheetByName(ABA_DISTRIBUICAO);
    if (!sheetDist) {
      return { sucesso: false, mensagem: "Aba de Distribuição não encontrada.", analisadas: 0, atualizadas: 0, semMatch: 0, semDataValida: 0 };
    }

    var nomeAba = (nomeAbaExtracao || "Extracao_Datas").toString().trim();
    var sheetExt = ss.getSheetByName(nomeAba);
    if (!sheetExt) {
      return { sucesso: false, mensagem: "Aba de extração '" + nomeAba + "' não encontrada.", analisadas: 0, atualizadas: 0, semMatch: 0, semDataValida: 0 };
    }

    var dataDist = sheetDist.getDataRange().getValues();
    var dataExt = sheetExt.getDataRange().getValues();
    if (dataDist.length <= 1) {
      return { sucesso: false, mensagem: "Distribuição vazia.", analisadas: 0, atualizadas: 0, semMatch: 0, semDataValida: 0 };
    }
    if (dataExt.length <= 1) {
      return { sucesso: false, mensagem: "A extração não tem linhas para cruzamento.", analisadas: 0, atualizadas: 0, semMatch: 0, semDataValida: 0 };
    }

    // Mapa da extração por ID normalizado -> melhor linha com datas válidas
    var mapaExt = {};
    for (var e = 1; e < dataExt.length; e++) {
      var idExt = _normIdCaso_(dataExt[e][0]);
      if (!idExt) continue;
      var dCriaExt = parseDataSegura(dataExt[e][1]);
      var dRqsExt = parseDataSegura(dataExt[e][2]);
      if (!dCriaExt && !dRqsExt) continue;

      if (!mapaExt[idExt]) {
        mapaExt[idExt] = { cria: dCriaExt || null, rqs: dRqsExt || null };
      } else {
        // Se já existe, completa apenas o que estiver em falta.
        if (!mapaExt[idExt].cria && dCriaExt) mapaExt[idExt].cria = dCriaExt;
        if (!mapaExt[idExt].rqs && dRqsExt) mapaExt[idExt].rqs = dRqsExt;
      }
    }

    var atualizadas = 0;
    var semMatch = 0;
    var semDataValida = 0;
    var analisadas = 0;
    var idsAtualizados = [];

    for (var i = 1; i < dataDist.length; i++) {
      var idDist = _normIdCaso_(dataDist[i][1]);
      if (!idDist) continue;

      var criaAtual = dataDist[i][5];
      var rqsAtual = dataDist[i][6];
      var criaVazia = (criaAtual === "" || criaAtual == null);
      var rqsVazia = (rqsAtual === "" || rqsAtual == null);
      if (!criaVazia && !rqsVazia) continue; // Já está completo, não mexe.

      analisadas++;
      var fonte = mapaExt[idDist];
      if (!fonte) {
        semMatch++;
        continue;
      }

      var mudou = false;
      if (criaVazia && fonte.cria) {
        dataDist[i][5] = fonte.cria;
        mudou = true;
      }
      if (rqsVazia && fonte.rqs) {
        dataDist[i][6] = fonte.rqs;
        mudou = true;
      }

      if (mudou) {
        atualizadas++;
        if (idsAtualizados.length < 50) idsAtualizados.push(idDist);
      } else {
        semDataValida++;
      }
    }

    if (atualizadas > 0) {
      sheetDist.getRange(1, 1, dataDist.length, 16).setValues(dataDist);
      SpreadsheetApp.flush();
      logOrigemEscrita_("repararDatasDistribuicaoViaExtracaoManual", "SETVALUES_BATCH_REPARACAO_DATAS", idsAtualizados.join(","), atualizadas, "Fonte: " + nomeAba + " (só F/G vazias)");
    }

    return {
      sucesso: true,
      mensagem: "Reparação concluída. Atualizadas: " + atualizadas + ". Sem match: " + semMatch + ". Sem data válida na extração: " + semDataValida + ".",
      analisadas: analisadas,
      atualizadas: atualizadas,
      semMatch: semMatch,
      semDataValida: semDataValida
    };
  } catch (e) {
    return { sucesso: false, mensagem: "Erro na reparação: " + e.message, analisadas: 0, atualizadas: 0, semMatch: 0, semDataValida: 0 };
  } finally {
    try { lock.releaseLock(); } catch (eRel) {}
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("repararDatasDistribuicaoViaExtracaoManual", new Date().getTime() - startTsHealth);
  }
}

function obterDadosGestaoEquipa() {
  try {
    var ss = getSpreadsheet();
    
    // 1. Obter TODOS os dados dos Utilizadores (Aba Utilizadores)
    var sheetUsers = ss.getSheetByName('Utilizadores');
    var dataUsers = sheetUsers.getDataRange().getValues();
    var users = [];
    for (var i = 1; i < dataUsers.length; i++) {
      if (dataUsers[i][0] && dataUsers[i][0].toString().trim() !== "") {
        users.push({
          email: dataUsers[i][0].toString().trim(),
          nome: dataUsers[i][1] ? dataUsers[i][1].toString().trim() : "",
          loja: dataUsers[i][2] ? dataUsers[i][2].toString().trim() : "",
          perfil: dataUsers[i][3] ? dataUsers[i][3].toString().trim() : "",
          equipa: dataUsers[i][4] ? dataUsers[i][4].toString().trim() : ""
        });
      }
    }

    // 2. Obter Listas de Configuração (Lojas, Perfis, Equipas)
    var sheetListas = garantirAbaConfigListas();
    var dataListas = sheetListas.getDataRange().getValues();
    var listas = { lojas: [], perfis: [], equipas: [] };
    for (var j = 1; j < dataListas.length; j++) {
      if (dataListas[j][0]) listas.lojas.push(dataListas[j][0].toString().trim());
      if (dataListas[j][1]) listas.perfis.push(dataListas[j][1].toString().trim());
      if (dataListas[j][2]) listas.equipas.push(dataListas[j][2].toString().trim());
    }

    // 3. Obter Skills (Aba Config_Skills, Coluna A)
    var sheetSkills = ss.getSheetByName('Config_Skills');
    var skillsMaster = [];
    if (sheetSkills && sheetSkills.getLastRow() > 0) {
       skillsMaster = sheetSkills.getRange(1, 1, sheetSkills.getLastRow(), 1).getValues().flat().filter(String);
    }

    return { sucesso: true, utilizadores: users, listas: listas, skills: skillsMaster };
  } catch(e) { return { sucesso: false, mensagem: "Erro: " + e.message }; }
}

function atualizarListaSistema(tipo, acao, valorAnterior, valorNovo) {
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(15000);
    var sheet = garantirAbaConfigListas();
    var data = sheet.getDataRange().getValues();
    var colIndex = tipo === 'loja' ? 1 : (tipo === 'perfil' ? 2 : 3);
    
    if (acao === 'adicionar') {
       // Procura a primeira linha vazia nessa coluna
       var linhaVazia = -1;
       for (var i = 1; i < data.length; i++) {
         if (!data[i][colIndex - 1] || data[i][colIndex - 1].toString().trim() === "") { linhaVazia = i + 1; break; }
       }
       if (linhaVazia === -1) linhaVazia = data.length + 1;
       sheet.getRange(linhaVazia, colIndex).setValue(valorNovo.trim());
       return { sucesso: true, mensagem: "Item adicionado!" };
    } 
    else if (acao === 'remover') {
       for (var j = 1; j < data.length; j++) {
         if (data[j][colIndex - 1] && data[j][colIndex - 1].toString().trim() === valorAnterior.trim()) {
           sheet.getRange(j + 1, colIndex).setValue("");
           return { sucesso: true, mensagem: "Item removido!" };
         }
       }
    }
    return { sucesso: false, mensagem: "Operação não reconhecida." };
  } catch(e) { return { sucesso: false, mensagem: e.message }; } finally { lock.releaseLock(); }
}

    if (linha === -1) return { sucesso: false, mensagem: "Caso não encontrado." };
    
    // Grava na Coluna P (Índice 16)
    sheetDist.getRange(linha, 16).setValue(novaPrioridade);
    
    // Registo de Auditoria nas Observações
    var obsAntigas = data[linha-1][8] ? data[linha-1][8].toString().trim() : "";
    var carimbo = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "dd/MM HH:mm");
    var estadoTxt = novaPrioridade === "SIM" ? "ativada (Flash)" : "removida";
    var txtAcao = "[Auto] Prioridade " + estadoTxt + " pela Supervisão (" + emailSuper.split('@')[0] + ").";
    
    sheetDist.getRange(linha, 9).setValue("[" + carimbo + "]\n" + txtAcao + "\n\n" + obsAntigas); 
    
    return { sucesso: true, mensagem: "Prioridade atualizada!" };
  } catch(e) { 
    return { sucesso: false, mensagem: e.message }; 
  } finally {
    if (typeof logAcaoHealthCheck === "function") logAcaoHealthCheck("alterarPrioridadeServidor", new Date().getTime() - startTsHealth);
  }
}

// --- ARRANQUE EM DOIS PEDIDOS (Web App): crítico (perfil + offline + skills) vs estatísticas em segundo plano ---

/** Perfil + Offline no servidor + skills. O cliente deve bloquear «Disponível» até esta chamada concluir com sucesso. */
function obterArranqueCritico() {
  var startTsHealth = new Date().getTime();
  var perfTop = SQ_DEBUG_PERF_PAYLOAD ? sqDebugPerfStart_() : null;
  try {
    var perfilResult = getPerfilUtilizador();
    if (perfTop) sqDebugPerfMark_(perfTop, 'perfil');
    if (!perfilResult.sucesso) {
      var failRet = {
        sucesso: false,
        perfil: perfilResult,
        skills: [],
        estadoArranque: null,
        arranqueCriticoOk: false
      };
      if (perfTop) {
        failRet.debugPerfPayload = {
          payload: sqDebugPerfBuildReport_(perfTop, ['perfil']),
          obterEstatisticasOperador: null,
          debugOffline: null
        };
      }
      return failRet;
    }

    var perfCtxOffline = perfTop ? sqDebugPerfStart_() : null;
    var resOffline = atualizarEstadoOperador('Offline', perfCtxOffline);
    if (perfTop) sqDebugPerfMark_(perfTop, 'offline');

    var skillsResult = obterListaSkills();
    if (perfTop) sqDebugPerfMark_(perfTop, 'skills');

    var okRet = {
      sucesso: true,
      perfil: perfilResult,
      skills: skillsResult,
      estadoArranque: 'Offline',
      arranqueCriticoOk: true
    };
    if (perfTop) {
      okRet.debugPerfPayload = {
        payload: sqDebugPerfBuildReport_(perfTop, ['perfil', 'offline', 'skills']),
        obterEstatisticasOperador: null,
        debugOffline:
          resOffline && resOffline.debugOfflineMs
            ? resOffline.debugOfflineMs
            : perfCtxOffline && perfCtxOffline._sqOfflinePerf
              ? perfCtxOffline._sqOfflinePerf
              : null
      };
    }
    return okRet;
  } catch (e) {
    return {
      sucesso: false,
      erroSilencioso: true,
      mensagem: 'Erro no arranque crítico: ' + e.message,
      perfil: null,
      skills: [],
      estadoArranque: null,
      arranqueCriticoOk: false
    };
  } finally {
    if (typeof logAcaoHealthCheck === 'function') {
      logAcaoHealthCheck('obterArranqueCritico', new Date().getTime() - startTsHealth);
    }
  }
}

/** Estatísticas do operador (segundo pedido; não bloqueia mudança para Disponível após obterArranqueCritico). */
function obterStatsBackground() {
  var startTsHealth = new Date().getTime();
  var perfTop = SQ_DEBUG_PERF_PAYLOAD ? sqDebugPerfStart_() : null;
  var subEstat = perfTop ? sqDebugPerfStart_() : null;
  try {
    if (perfTop) sqDebugPerfMark_(perfTop, 'stats_start');
    _sqPerfEstatSub_ = subEstat;
    var statsResult = obterEstatisticasOperador();
    if (perfTop) sqDebugPerfMark_(perfTop, 'stats_end');
    var out = { sucesso: !!(statsResult && statsResult.sucesso), stats: statsResult };
    if (perfTop) {
      out.debugPerfPayload = {
        payload: sqDebugPerfBuildReport_(perfTop, ['stats_start', 'stats_end']),
        obterEstatisticasOperador: subEstat
          ? sqDebugPerfBuildReport_(subEstat, [
              'stats_utilizadores_loja',
              'stats_log_perf',
              'stats_contadores_ranking',
              'stats_log_estados',
              'stats_objetivos_loja'
            ])
          : null,
        debugOffline: null
      };
    }
    return out;
  } catch (e) {
    return {
      sucesso: false,
      stats: null,
      erroSilencioso: true,
      mensagem: 'Erro nas estatísticas: ' + e.message
    };
  } finally {
    _sqPerfEstatSub_ = null;
    if (typeof logAcaoHealthCheck === 'function') {
      logAcaoHealthCheck('obterStatsBackground', new Date().getTime() - startTsHealth);
    }
  }
}

/** Compatibilidade: um único round-trip (crítico + stats). Preferir obterArranqueCritico + obterStatsBackground no cliente. */
function obterPayloadUnicoArranque() {
  var startTsHealth = new Date().getTime();
  try {
    var c = obterArranqueCritico();
    if (!c.sucesso) {
      return {
        sucesso: false,
        perfil: c.perfil,
        skills: c.skills || [],
        stats: null,
        estadoArranque: null,
        debugPerfPayload: c.debugPerfPayload || null
      };
    }
    var b = obterStatsBackground();
    var merged = {
      sucesso: b.sucesso !== false && c.sucesso,
      perfil: c.perfil,
      skills: c.skills,
      stats: b.stats,
      estadoArranque: c.estadoArranque
    };
    if (SQ_DEBUG_PERF_PAYLOAD && (c.debugPerfPayload || b.debugPerfPayload)) {
      merged.debugPerfPayload = {
        payloadCritico: c.debugPerfPayload && c.debugPerfPayload.payload,
        payloadStatsPedido: b.debugPerfPayload && b.debugPerfPayload.payload,
        obterEstatisticasOperador: b.debugPerfPayload && b.debugPerfPayload.obterEstatisticasOperador,
        debugOffline: c.debugPerfPayload && c.debugPerfPayload.debugOffline
      };
    }
    return merged;
  } finally {
    if (typeof logAcaoHealthCheck === 'function') {
      logAcaoHealthCheck('obterPayloadUnicoArranque', new Date().getTime() - startTsHealth);
    }
  }
}

/** @deprecated Usar obterArranqueCritico / obterPayloadUnicoArranque; mantido para compatibilidade. */
function obterDadosArranqueCompleto() {
  return obterPayloadUnicoArranque();
}
