/**
 * FUNÇÃO DE EMERGÊNCIA
 * Executar manualmente apenas se a app ficar "a pensar" 
 * ou se houver suspeita de Deadlock após erro de servidor.
 */
function ADMIN_ResetCadeados() {
  const scriptLock = LockService.getScriptLock();
  const docLock = LockService.getDocumentLock();
  const userLock = LockService.getUserLock();
  
  scriptLock.releaseLock();
  docLock.releaseLock();
  userLock.releaseLock();
  
  console.log("🔓 [ADMIN] Todos os níveis de cadeados foram libertados com sucesso.");
}
