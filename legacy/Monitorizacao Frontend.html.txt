        <div class="kpi-card"><div class="kpi-lbl">Picos Perigosos (>30)</div><div class="kpi-val" id="kpi-alerts" style="color:var(--alerta-vermelho)">-</div></div>

    <script>
       // Configurações globais Chart.js (Dark Mode)
       Chart.defaults.color = '#E5E7EB';
       Chart.defaults.borderColor = 'rgba(255,255,255,0.1)';
       
       let cLine = null, cUsers = null, cScripts = null;

       window.onload = function() {
           var today = new Date().toISOString().split('T')[0];
           document.getElementById('data-filtro').value = today;
           carregarDashboard();
       };

       function carregarDashboard() {
           var dateVal = document.getElementById('data-filtro').value;
           if(!dateVal) return;

           document.getElementById('kpi-totais').innerText = "⏳";
           
           google.script.run.withSuccessHandler(renderizarDados).obterDadosHealthCheck(dateVal);
       }

       function renderizarDados(res) {
           if(!res.sucesso) { alert(res.mensagem); return; }

           document.getElementById('kpi-totais').innerText = res.totais;
           document.getElementById('kpi-maxuser').innerText = res.users.length > 0 ? res.users[0].v : 0;
           document.getElementById('kpi-maxscript').innerText = res.scripts.length > 0 ? res.scripts[0].v : 0;

           // Verificar se na timeline alguma barreira de 30 calls simultâneas foi furada
           var picosAlerts = res.timeline.data.filter(v => v >= 30).length;
           document.getElementById('kpi-alerts').innerText = picosAlerts;

           renderChartLine(res.timeline.labels, res.timeline.data);
           renderChartBars('chartUsers', cUsers, res.users.map(u=>u.u.split('@')[0]), res.users.map(u=>u.v), '#2175D9');
           renderChartBars('chartScripts', cScripts, res.scripts.map(s=>s.s), res.scripts.map(s=>s.v), '#F59E0B');
       }

       function renderChartLine(labels, data) {
           if(cLine) cLine.destroy();
           const ctx = document.getElementById('chartLine').getContext('2d');
           cLine = new Chart(ctx, {
               type: 'line',
               data: { labels: labels, datasets: [{ label: 'Execuções p/ Minuto', data: data, borderColor: '#10B981', tension: 0.3, fill: true, backgroundColor: 'rgba(16, 185, 129, 0.1)' }] },
               options: { 
                   plugins: { 
                       annotation: { annotations: { line1: { type: 'line', yMin: 30, yMax: 30, borderColor: '#EF4444', borderWidth: 2, borderDash: [5, 5] } } }
                   } 
               }
           });
       }

       function renderChartBars(canvasId, refObj, labels, data, color) {
           if(refObj) refObj.destroy();
           const ctx = document.getElementById(canvasId).getContext('2d');
           var chart = new Chart(ctx, {
               type: 'bar',
               data: { labels: labels, datasets: [{ label: 'Total Calls', data: data, backgroundColor: color, borderRadius: 4 }] },
               options: { indexAxis: 'y', plugins: { legend: { display: false } } }
           });
           if(canvasId === 'chartUsers') cUsers = chart;
           if(canvasId === 'chartScripts') cScripts = chart;
       }
    </script>
